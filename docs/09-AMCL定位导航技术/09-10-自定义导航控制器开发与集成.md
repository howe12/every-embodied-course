## 1. 控制器接口概述

Navigation2框架采用插件化架构，控制器（Controller）负责将规划出的路径转换为机器人底层的速度指令。一个好的控制器需要考虑机器人的运动学约束、动态避障、轨迹跟踪精度等因素。

### 1.1 Navigation2控制器架构

Navigation2的控制器通过`nav2_controller`包管理，核心接口定义在`nav2_core`包中。控制器插件需要实现`Controller`接口，该接口定义了控制器的基本方法。

### 1.2 核心接口类

控制器接口的核心定义如下：

```cpp
// nav2_core/controller.hpp
#include "rclcpp/rclcpp.hpp"
#include "geometry_msgs/msg/twist.hpp"
#include "geometry_msgs/msg/pose_stamped.hpp"
#include "nav_msgs/msg/path.hpp"
#include "nav2_util/lifecycle_node.hpp"

namespace nav2_core
{

class Controller : public rclcpp::Node
{
public:
  using Ptr = std::shared_ptr<Controller>;

  virtual ~Controller() = default;

  // 配置控制器
  virtual void configure(
    const rclcpp_lifecycle::LifecycleNode::WeakPtr & parent,
    std::string name,
    std::shared_ptr<tf2_ros::Buffer> tf,
    std::shared_ptr<nav2_costmap_2d::Costmap2DROS> costmap_ros) = 0;

  // 设置路径
  virtual void setPlan(const nav_msgs::msg::Path & path) = 0;

  // 计算速度命令（核心方法）
  virtual geometry_msgs::msg::Twist::SharedPtr computeVelocityCommands(
    const geometry_msgs::msg::PoseStamped & pose,
    const geometry_msgs::msg::Twist & velocity) = 0;

  // 清理资源
  virtual void cleanup() = 0;

  // 激活节点
  virtual void activate() = 0;

  // 停用节点
  virtual void deactivate() = 0;
};

}  // namespace nav2_core
```

关键方法说明：

| 方法 | 功能 |
|------|------|
| `configure()` | 初始化控制器参数、TF转换器、代价地图 |
| `setPlan()` | 接收全局规划器生成的路径 |
| `computeVelocityCommands()` | 根据当前位姿和速度，计算下一时刻的速度命令 |
| `cleanup()` | 释放资源 |
| `activate()/deactivate()` | 生命周期管理 |

## 3. PID控制器实现

PID（Proportional-Integral-Derivative）控制器是经典的闭环控制算法，具有结构简单、参数易于调节的优点。在机器人导航中，PID控制器常用于轨迹跟踪和速度调节。

### 3.1 PID控制原理

PID控制器的控制律由三项组成：

$$u(t) = K_p e(t) + K_i \int_0^t e(\tau) d\tau + K_d \frac{de(t)}{dt}$$

其中：
- $K_p$：比例增益
- $K_i$：积分增益
- $K_d$：微分增益
- $e(t)$：误差信号（目标值 - 实际值）

各项作用：
- **比例项**：根据当前误差产生控制作用，响应速度快
- **积分项**：消除稳态误差，提高控制精度
- **微分项**：根据误差变化趋势产生控制作用，改善动态性能

### 3.2 PID控制器类实现

```cpp
// pid_controller.hpp
#ifndef PID_CONTROLLER_HPP
#define PID_CONTROLLER_HPP

#include <memory>
#include <cmath>

class PIDController
{
public:
  PIDController() = default;

  PIDController(double kp, double ki, double kd, double max_output, double min_output)
    : kp_(kp), ki_(ki), kd_(kd), max_output_(max_output), min_output_(min_output)
  {}

  void setGains(double kp, double ki, double kd)
  {
    kp_ = kp;
    ki_ = ki;
    kd_ = kd;
  }

  void reset()
  {
    integral_ = 0.0;
    prev_error_ = 0.0;
    prev_timestamp_ = 0.0;
  }

  double compute(double error, double dt)
  {
    if (dt <= 0.0) return 0.0;

    // 积分项（带抗饱和）
    integral_ += error * dt;
    integral_ = std::clamp(integral_, min_integral_, max_integral_);

    // 微分项
    double derivative = (error - prev_error_) / dt;

    // PID输出
    double output = kp_ * error + ki_ * integral_ + kd_ * derivative;

    // 输出限幅
    output = std::clamp(output, min_output_, max_output_);

    // 保存状态
    prev_error_ = error;
    prev_timestamp_ = dt;

    return output;
  }

private:
  double kp_ = 1.0;
  double ki_ = 0.0;
  double kd_ = 0.0;
  double max_output_ = 1.0;
  double min_output_ = -1.0;
  double integral_ = 0.0;
  double prev_error_ = 0.0;
  double prev_timestamp_ = 0.0;
  double min_integral_ = -10.0;
  double max_integral_ = 10.0;
};

#endif  // PID_CONTROLLER_HPP
```

### 3.3 差速机器人轨迹跟踪控制器

基于PID的差速机器人轨迹跟踪控制器实现：

```cpp
// pid_dwb_controller.cpp
#include "pid_controller.hpp"
#include "nav2_core/controller.hpp"
#include "rclcpp/rclcpp.hpp"
#include "geometry_msgs/msg/twist.hpp"
#include "geometry_msgs/msg/pose_stamped.hpp"
#include "nav_msgs/msg/path.hpp"
#include "nav2_util/lifecycle_node.hpp"
#include "nav2_costmap_2d/costmap_2d_ros.hpp"
#include "tf2_ros/buffer.h"
#include "tf2_geometry_msgs/tf2_geometry_msgs.hpp"
#include "pluginlib/class_list_macros.hpp"

namespace pid_controller
{

class PIDTrajectoryController : public nav2_core::Controller
{
public:
  void configure(
    const rclcpp_lifecycle::LifecycleNode::WeakPtr & parent,
    std::string name,
    std::shared_ptr<tf2_ros::Buffer> tf,
    std::shared_ptr<nav2_costmap_2d::Costmap2DROS> costmap_ros) override
  {
    node_ = parent.lock();
    name_ = name;
    tf_ = tf;
    costmap_ros_ = costmap_ros;

    // 读取参数
    node_->declare_parameter("kp_linear", 1.0);
    node_->declare_parameter("ki_linear", 0.0);
    node_->declare_parameter("kd_linear", 0.0);
    node_->declare_parameter("kp_angular", 2.0);
    node_->declare_parameter("ki_angular", 0.0);
    node_->declare_parameter("kd_angular", 0.5);
    node_->declare_parameter("max_linear_speed", 0.5);
    node_->declare_parameter("max_angular_speed", 1.0);
    node_->declare_parameter("goal_tolerance", 0.1);
    node_->declare_parameter("lookahead_distance", 0.3);

    node_->get_parameter("kp_linear", kp_linear_);
    node_->get_parameter("ki_linear", ki_linear_);
    node_->get_parameter("kd_linear", kd_linear_);
    node_->get_parameter("kp_angular", kp_angular_);
    node_->get_parameter("ki_angular", ki_angular_);
    node_->get_parameter("kd_angular", kd_angular_);
    node_->get_parameter("max_linear_speed", max_linear_speed_);
    node_->get_parameter("max_angular_speed", max_angular_speed_);
    node_->get_parameter("goal_tolerance", goal_tolerance_);
    node_->get_parameter("lookahead_distance", lookahead_distance_);

    // 初始化PID控制器
    pid_linear_ = std::make_unique<PIDController>(
      kp_linear_, ki_linear_, kd_linear_, max_linear_speed_, -max_linear_speed_);
    pid_angular_ = std::make_unique<PIDController>(
      kp_angular_, ki_angular_, kd_angular_, max_angular_speed_, -max_angular_speed_);

    RCLCPP_INFO(node_->get_logger(), "PID Trajectory Controller configured");
  }

  void setPlan(const nav_msgs::msg::Path & path) override
  {
    path_ = path;
    pid_linear_->reset();
    pid_angular_->reset();
    current_point_index_ = 0;
  }

  geometry_msgs::msg::Twist::SharedPtr computeVelocityCommands(
    const geometry_msgs::msg::PoseStamped & pose,
    const geometry_msgs::msg::Twist & velocity) override
  {
    auto cmd = geometry_msgs::msg::Twist::make();

    if (path_.poses.empty()) {
      return cmd;
    }

    // 找到目标跟踪点
    auto target_point = getTargetPoint(pose);

    // 计算误差
    double dx = target_point.pose.position.x - pose.pose.position.x;
    double dy = target_point.pose.position.y - pose.pose.position.y;

    // 转换到机器人坐标系
    double yaw;
    tf2::Quaternion q;
    tf2::fromMsg(pose.pose.orientation, q);
    tf2::Matrix3x3 m(q);
    m.getRPY({}, {}, yaw);

    double local_x = dx * cos(yaw) + dy * sin(yaw);
    double local_y = -dx * sin(yaw) + dy * cos(yaw);

    // 计算角度误差
    double target_yaw;
    tf2::Quaternion target_q;
    tf2::fromMsg(target_point.pose.orientation, target_q);
    tf2::Matrix3x3 target_m(target_q);
    target_m.getRPY({}, {}, target_yaw);

    double angular_error = target_yaw - yaw;
    // 归一化到 [-π, π]
    while (angular_error > M_PI) angular_error -= 2 * M_PI;
    while (angular_error < -M_PI) angular_error += 2 * M_PI;

    // 计算线性速度和角速度
    double linear_error = sqrt(local_x * local_x + local_y * local_y);
    double dt = 0.1;  // 控制周期

    cmd->linear.x = pid_linear_->compute(linear_error, dt);
    cmd->angular.z = pid_angular_->compute(angular_error, dt);

    // 速度限幅
    cmd->linear.x = std::clamp(cmd->linear.x, -max_linear_speed_, max_linear_speed_);
    cmd->angular.z = std::clamp(cmd->angular.z, -max_angular_speed_, max_angular_speed_);

    return cmd;
  }

  geometry_msgs::msg::PoseStamped getTargetPoint(const geometry_msgs::msg::PoseStamped & pose)
  {
    // 简单实现：选择路径上下一个未到达的点
    for (size_t i = current_point_index_; i < path_.poses.size(); ++i) {
      double dx = path_.poses[i].pose.position.x - pose.pose.position.x;
      double dy = path_.poses[i].pose.position.y - pose.pose.position.y;
      double dist = sqrt(dx * dx + dy * dy);

      if (dist > lookahead_distance_) {
        current_point_index_ = i;
        return path_.poses[i];
      }
    }

    // 如果没有找到，返回路径终点
    return path_.poses.back();
  }

  void cleanup() override
  {
    RCLCPP_INFO(node_->get_logger(), "Cleaning up PID controller");
  }

  void activate() override
  {
    RCLCPP_INFO(node_->get_logger(), "Activating PID controller");
  }

  void deactivate() override
  {
    RCLCPP_INFO(node_->get_logger(), "Deactivating PID controller");
  }

private:
  rclcpp_lifecycle::LifecycleNode::SharedPtr node_;
  std::string name_;
  std::shared_ptr<tf2_ros::Buffer> tf_;
  std::shared_ptr<nav2_costmap_2d::Costmap2DROS> costmap_ros_;
  nav_msgs::msg::Path path_;

  // PID参数
  double kp_linear_, ki_linear_, kd_linear_;
  double kp_angular_, ki_angular_, kd_angular_;
  double max_linear_speed_, max_angular_speed_;
  double goal_tolerance_;
  double lookahead_distance_;

  std::unique_ptr<PIDController> pid_linear_;
  std::unique_ptr<PIDController> pid_angular_;
  size_t current_point_index_ = 0;
};

}  // namespace pid_controller

PLUGINLIB_EXPORT_CLASS(pid_controller::PIDTrajectoryController, nav2_core::Controller)
```

## 4. MPC控制器实现

模型预测控制（Model Predictive Control，MPC）是一种先进的控制策略，通过在线求解有限时域的优化问题来获得控制序列。MPC能够处理多变量约束控制问题，适用于机器人轨迹跟踪。

### 4.1 MPC基本原理

MPC的核心思想是：
1. 在每个控制周期，基于当前状态，预测系统未来$N$步的行为
2. 构建优化问题，最小化目标函数（通常包含跟踪误差和控制 effort）
3. 求解优化问题，得到最优控制序列
4. 只执行控制序列的第一个元素
5. 重复上述过程

```
        ┌─────────────────────────────────────────────┐
        │              MPC 滚动优化                     │
        ├─────────────────────────────────────────────┤
        │                                             │
        │   预测时域 N                                 │
        │   ├─────────┬─────────┬─────────┬─────────┐ │
        │   │         │         │         │         │ │
        │   ▼         ▼         ▼         ▼         ▼ │
        │  x(k) ──▶ x(k+1) ──▶ x(k+2) ──▶ ... ──▶ x(k+N)│
        │                                             │
        │  u(k) ──▶ u(k+1) ──▶ u(k+2) ──▶ ... ──▶ u(k+N-1)
        │                                             │
        │   ┌─────────────────────────────────────┐   │
        │   │  min Σ||x(k+i) - x_ref(k+i)||² +    │   │
        │   │      Σ||u(k+i)||²                   │   │
        │   │  s.t. 动力学约束 + 输入约束          │   │
        │   └─────────────────────────────────────┘   │
        │              ↓                               │
        │         只执行 u(k)                          │
        └─────────────────────────────────────────────┘
```

### 4.2 运动学模型

对于差速机器人，采用运动学模型：

$$
\begin{bmatrix} \dot{x} \\ \dot{y} \\ \dot{\theta} \end{bmatrix} = 
\begin{bmatrix} \cos\theta & 0 \\ \sin\theta & 0 \\ 0 & 1 \end{bmatrix}
\begin{bmatrix} v \\ \omega \end{bmatrix}
$$

离散化后：

$$
\begin{aligned}
x_{k+1} &= x_k + v_k \cos(\theta_k) \Delta t \\
y_{k+1} &= y_k + v_k \sin(\theta_k) \Delta t \\
\theta_{k+1} &= \theta_k + \omega_k \Delta t
\end{aligned}
$$

### 4.3 MPC控制器Python实现

```python
# mpc_controller.py
import rclpy
from rclpy.node import Node
from geometry_msgs.msg import Twist, PoseStamped
from nav_msgs.msg import Path
import numpy as np
from scipy import interpolate

class MPCTrajectoryController(Node):
    def __init__(self):
        super().__init__('mpc_controller')
        
        # 声明参数
        self.declare_parameter('prediction_horizon', 10)
        self.declare_parameter('control_horizon', 5)
        self.declare_parameter('dt', 0.1)
        self.declare_parameter('max_linear_vel', 0.5)
        self.declare_parameter('max_angular_vel', 1.0)
        self.declare_parameter('weight_error', 1.0)
        self.declare_parameter('weight_control', 0.1)
        
        # 获取参数
        self.N = self.get_parameter('prediction_horizon').value
        self.M = self.get_parameter('control_horizon').value
        self.dt = self.get_parameter('dt').value
        self.max_v = self.get_parameter('max_linear_vel').value
        self.max_w = self.get_parameter('max_angular_vel').value
        self.Q = self.get_parameter('weight_error').value
        self.R = self.get_parameter('weight_control').value
        
        # 状态变量
        self.path = Path()
        self.current_pose = None
        
        # 发布者和订阅者
        self.cmd_pub = self.create_publisher(Twist, '/cmd_vel', 1)
        self.path_sub = self.create_subscription(
            Path, '/plan', self.path_callback, 1)
        self.pose_sub = self.create_subscription(
            PoseStamped, '/goal_pose', self.pose_callback, 1)
        
        self.get_logger().info('MPC Controller initialized')
        
    def path_callback(self, msg):
        self.path = msg
        
    def pose_callback(self, msg):
        self.current_pose = msg
        if self.path.poses:
            cmd = self.compute_mpc()
            self.cmd_pub.publish(cmd)
    
    def compute_mpc(self):
        """MPC控制计算"""
        if not self.current_pose or not self.path.poses:
            return Twist()
            
        # 简化的MPC实现（实际需要求解优化问题）
        cmd = Twist()
        
        # 获取目标点
        target = self.get_nearest_path_point(self.current_pose)
        
        # 计算误差
        dx = target.pose.position.x - self.current_pose.pose.position.x
        dy = target.pose.position.y - self.current_pose.pose.position.y
        
        # 简单的比例控制（近似MPC效果）
        dist_error = np.sqrt(dx**2 + dy**2)
        
        # 角度误差
        current_yaw = self.get_yaw(self.current_pose.pose.orientation)
        target_yaw = self.get_yaw(target.pose.orientation)
        angle_error = target_yaw - current_yaw
        angle_error = np.arctan2(np.sin(angle_error), np.cos(angle_error))
        
        # 速度计算
        cmd.linear.x = min(self.max_v, dist_error * 2.0)
        cmd.angular.z = np.clip(angle_error * 2.0, -self.max_w, self.max_w)
        
        return cmd
    
    def get_nearest_path_point(self, pose):
        """找到路径上最近的点"""
        if not self.path.poses:
            return None
            
        min_dist = float('inf')
        nearest_idx = 0
        
        for i, p in enumerate(self.path.poses):
            dx = p.pose.position.x - pose.pose.position.x
            dy = p.pose.position.y - pose.pose.position.y
            dist = np.sqrt(dx**2 + dy**2)
            if dist < min_dist:
                min_dist = dist
                nearest_idx = i
        
        return self.path.poses[min(nearest_idx + 3, len(self.path.poses) - 1)]
    
    @staticmethod
    def get_yaw(quaternion):
        """从四元数获取偏航角"""
        x = quaternion.x
        y = quaternion.y
        z = quaternion.z
        w = quaternion.w
        
        # Roll (x-axis rotation)
        sinr_cosp = 2 * (w * x + y * z)
        cosr_cosp = 1 - 2 * (x * x + y * y)
        roll = np.arctan2(sinr_cosp, cosr_cosp)
        
        # Pitch (y-axis rotation)
        sinp = 2 * (w * y - z * x)
        if np.abs(sinp) >= 1:
            pitch = np.copysign(np.pi / 2, sinp)
        else:
            pitch = np.arcsin(sinp)
        
        # Yaw (z-axis rotation)
        siny_cosp = 2 * (w * z + x * y)
        cosy_cosp = 1 - 2 * (y * y + z * z)
        yaw = np.arctan2(siny_cosp, cosy_cosp)
        
        return yaw

def main(args=None):
    rclpy.init(args=args)
    node = MPCTrajectoryController()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()

if __name__ == '__main__':
    main()
```

### 4.4 基于OSQP的完整MPC实现

对于需要求解约束优化问题的完整MPC，可以使用OSQP求解器：

```python
# advanced_mpc_controller.py
import rclpy
from rclpy.node import Node
from geometry_msgs.msg import Twist, PoseStamped
from nav_msgs.msg import Path
import numpy as np
import osqp

class AdvancedMPCController(Node):
    def __init__(self):
        super().__init__('advanced_mpc_controller')
        
        # 参数
        self.declare_parameter('nx', 3)  # 状态维度: [x, y, theta]
        self.declare_parameter('nu', 2)  # 控制维度: [v, w]
        self.declare_parameter('N', 10)  # 预测时域
        self.declare_parameter('dt', 0.1)
        
        self.nx = self.get_parameter('nx').value
        self.nu = self.get_parameter('nu').value
        self.N = self.get_parameter('N').value
        self.dt = self.get_parameter('dt').value
        
        # 代价矩阵
        self.Q = np.diag([1.0, 1.0, 0.1])  # 状态代价
        self.R = np.diag([0.1, 0.1])       # 控制代价
        
        # 约束
        self.v_max = 0.5
        self.v_min = -0.5
        self.w_max = 1.0
        self.w_min = -1.0
        
        self.path = Path()
        self.current_pose = None
        
        # 创建OSQP问题
        self.setup_osqp()
        
        # 订阅者和发布者
        self.cmd_pub = self.create_publisher(Twist, '/cmd_vel', 1)
        self.path_sub = self.create_subscription(Path, '/plan', self.path_callback, 1)
        
        self.get_logger().info('Advanced MPC Controller initialized')
        
    def setup_osqp(self):
        """设置OSQP优化问题"""
        n_vars = self.N * self.nu  # 控制变量数量
        n_constraints = self.N * self.nx  # 约束数量
        
        # 稀疏矩阵P (Hessian)
        P = np.zeros((n_vars, n_vars))
        for i in range(self.N):
            P[i*self.nu:(i+1)*self.nu, i*self.nu:(i+1)*self.nu] = self.R
        
        # 线性约束A (动力学)
        A = np.zeros((n_constraints, n_vars))
        
        # 求解器
        self.osqp = osqp.OSQP()
        
    def path_callback(self, msg):
        self.path = msg
        
    def compute_mpc(self, pose):
        """求解MPC优化问题"""
        if not self.path.poses or not pose:
            return Twist()
        
        # 初始化
        x0 = np.array([
            pose.position.x,
            pose.position.y,
            0.0  # theta
        ])
        
        # 简化的线性MPC（实际需要完整的非线性模型）
        cmd = Twist()
        
        # 获取目标状态序列
        ref_states = self.get_reference_states(pose)
        
        # 求解QP问题（简化版：使用简单的梯度下降）
        u = np.zeros(self.nu)  # 初始控制
        
        for _ in range(10):  # 迭代求解
            # 预测状态
            x_pred = self.predict_states(x0, u)
            
            # 计算代价梯度
            grad = self.compute_gradient(x_pred, ref_states, u)
            
            # 线搜索
            alpha = 0.1
            u = u - alpha * grad
            
            # 约束投影
            u[0] = np.clip(u[0], self.v_min, self.v_max)
            u[1] = np.clip(u[1], self.w_min, self.w_max)
        
        cmd.linear.x = u[0]
        cmd.angular.z = u[1]
        
        return cmd
    
    def predict_states(self, x0, u):
        """预测状态序列"""
        x = np.zeros((self.N + 1, self.nx))
        x[0] = x0
        
        for i in range(self.N):
            v = u[0]
            w = u[1]
            theta = x[i, 2]
            
            x[i+1, 0] = x[i, 0] + v * np.cos(theta) * self.dt
            x[i+1, 1] = x[i, 1] + v * np.sin(theta) * self.dt
            x[i+1, 2] = x[i, 2] + w * self.dt
        
        return x
    
    def compute_gradient(self, x_pred, ref_states, u):
        """计算代价梯度"""
        grad = np.zeros(self.nu)
        
        for i in range(self.N):
            # 状态误差
            error = x_pred[i+1] - ref_states[i]
            
            # 对控制的梯度（简化）
            grad[0] += error[0] * np.cos(x_pred[i, 2]) * self.dt
            grad[1] += error[2] * self.dt
            
            # 控制代价梯度
            grad += self.R @ u * self.dt
        
        return grad
    
    def get_reference_states(self, pose):
        """从路径获取参考状态序列"""
        ref = []
        # 简化：使用路径上等间距的点
        step = max(1, len(self.path.poses) // self.N)
        
        for i in range(self.N):
            idx = min(i * step, len(self.path.poses) - 1)
            p = self.path.poses[idx]
            ref.append(np.array([
                p.pose.position.x,
                p.pose.position.y,
                0.0
            ]))
        
        return ref

def main(args=None):
    rclpy.init(args=args)
    node = AdvancedMPCController()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()

if __name__ == '__main__':
    main()
```

### 6.1 常见问题与解决方案

| 问题 | 原因 | 解决方案 |
|------|------|----------|
| 机器人振荡 | $K_p$过大或$K_d$过小 | 减小$K_p$，增大$K_d$ |
| 响应缓慢 | $K
| 响应缓慢 | $K_p$过小 | 增大$K_p$，适当增大$K_d$ |
| 稳态误差 | 积分项不足 | 增大$K_i$ |
| 超调过大 | $K_p$过大或$K_i$过大 | 减小$K_p$和$K_i$ |
| 控制震荡 | 微分项噪声放大 | 增大微分滤波常数 |

### 6.2 参数调节技巧

**PID参数调节口诀**：

| 调节顺序 | 参数 | 调节方法 |
|----------|------|----------|
| 1 | $K_p$ | 从小到大调节，直到系统出现振荡 |
| 2 | $K_d$ | 从小到大调节，消除振荡 |
| 3 | $K_i$ | 缓慢增加，消除稳态误差 |

**Ziegler-Nichols方法**：
1. 将$K_i$和$K_d$设为0
2. 逐渐增大$K_p$直到系统开始持续振荡（临界振荡）
3. 记录此时的临界增益$K_u$和振荡周期$T_u$
4. 按下表计算参数：

| 控制器类型 | $K_p$ | $K_i$ | $K_d$ |
|------------|-------|-------|-------|
| P | $0.5K_u$ | 0 | 0 |
| PI | $0.45K_u$ | $0.54K_u/T_u$ | 0 |
| PID | $0.6K_u$ | $1.2K_u/T_u$ | $0.075K_uT_u$ |

6. 编写一个简化的P控制器，实现机器人对路径点的跟踪。
   
   **参考答案**：
   
   ```python
   import rclpy
   from rclpy.node import Node
   from geometry_msgs.msg import Twist, PoseStamped
   from nav_msgs.msg import Path
   import math
   
   class PController(Node):
       def __init__(self):
           super().__init__('p_controller')
           self.declare_parameter('kp', 1.0)
           self.declare_parameter('max_speed', 0.5)
           
           self.kp = self.get_parameter('kp').value
           self.max_speed = self.get_parameter('max_speed').value
           self.path = None
           
           self.cmd_pub = self.create_publisher(Twist, '/cmd_vel', 1)
           self.path_sub = self.create_subscription(Path, '/plan', self.path_callback, 1)
           self.pose_sub = self.create_subscription(PoseStamped, '/pose', self.pose_callback, 1)
           
       def path_callback(self, msg):
           self.path = msg
           
       def pose_callback(self, msg):
           if not self.path or not self.path.poses:
               return
           
           # 找最近的路径点
           target = self.get_nearest_point(msg)
           
           # 计算误差
           dx = target.pose.position.x - msg.pose.position.x
           dy = target.pose.position.y - msg.pose.position.y
           
           # P控制
           dist = math.sqrt(dx**2 + dy**2)
           speed = min(self.kp * dist, self.max_speed)
           
           # 发布速度命令
           cmd = Twist()
           cmd.linear.x = speed
           
           # 简单角度控制
           target_yaw = math.atan2(dy, dx)
           current_yaw = self.get_yaw(msg.pose.orientation)
           angle_error = target_yaw - current_yaw
           cmd.angular.z = angle_error
           
           self.cmd_pub.publish(cmd)
           
       def get_nearest_point(self, pose):
           min_dist = float('inf')
           nearest = self.path.poses[0]
           for p in self.path.poses:
               dx = p.pose.position.x - pose.pose.position.x
               dy = p.pose.position.y - pose.pose.position.y
               dist = math.sqrt(dx**2 + dy**2)
               if dist < min_dist:
                   min_dist = dist
                   nearest = p
           return nearest
           
       def get_yaw(self, quat):
           return math.atan2(2*(quat.w*quat.z + quat.x*quat.y), 
                           1-2*(quat.y**2 + quat.z**2))
   
   def main(args=None):
       rclpy.init(args=args)
       node = PController()
       rclpy.spin(node)
       node.destroy_node()
       rclpy.shutdown()
   
   if __name__ == '__main__':
       main()
   ```
