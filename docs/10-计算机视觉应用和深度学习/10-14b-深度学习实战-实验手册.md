## 实验1：纯FC网络基准线

### 1.1 实验目的
- 掌握MNIST数据加载和预处理
- 理解数据归一化对训练的影响

### 1.2 代码模板

```python
"""
实验1：纯FC网络基准线
任务：使用单层全连接网络训练MNIST分类器
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
import torchvision
import torchvision.transforms as transforms
from torch.utils.data import DataLoader

# 数据加载
transform = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))  # 标准化到 [-1, 1]
])

trainset = torchvision.datasets.MNIST(root='./data', train=True, download=True, transform=transform)
trainloader = DataLoader(trainset, batch_size=64, shuffle=True, num_workers=2)
testset = torchvision.datasets.MNIST(root='./data', train=False, download=True, transform=transform)
testloader = DataLoader(testset, batch_size=100, shuffle=False, num_workers=2)

print(f"训练集样本数: {len(trainset)}, 测试集样本数: {len(testset)}")

# 定义FC网络
class SimpleFCNet(nn.Module):
    def __init__(self):
        super(SimpleFCNet, self).__init__()
        # ★ TODO: 输入784维(28x28展平)，输出10分类
        self.fc1 = nn.Linear(784, 256)
        self.fc2 = nn.Linear(256, 10)
    
    def forward(self, x):
        x = x.view(x.size(0), -1)  # 展平
        x = F.relu(self.fc1(x))
        return self.fc2(x)  # 输出层不加激活，配合CrossEntropyLoss

# 训练函数
def train_model(model, trainloader, testloader, num_epochs=5, lr=0.001):
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=lr)
    
    for epoch in range(num_epochs):
        model.train()
        running_loss = 0.0
        correct = 0
        total = 0
        
        for images, labels in trainloader:
            outputs = model(images)
            loss = criterion(outputs, labels)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            
            running_loss += loss.item()
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        
        train_acc = 100 * correct / total
        
        model.eval()
        correct = 0
        total = 0
        with torch.no_grad():
            for images, labels in testloader:
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        
        test_acc = 100 * correct / total
        print(f'Epoch [{epoch+1}/{num_epochs}], Train: {train_acc:.2f}%, Test: {test_acc:.2f}%')
    
    return model

# 运行实验
if __name__ == '__main__':
    model = SimpleFCNet()
    print(f"网络参数量: {sum(p.numel() for p in model.parameters()):,}")
    model = train_model(model, trainloader, testloader)
    torch.save(model.state_dict(), 'exp1_fc_net.pth')
```

## 实验2：卷积特征提取

### 2.1 实验目的
- 理解卷积层如何提取图像特征
- 掌握 Conv2d + ReLU + Pooling 的组合方式
- 对比 CNN vs 纯 FC 的参数量和精度

### 2.2 积木组合
```
输入 [1, 28, 28]
  ↓ Conv2d (1→32, 3×3, padding=1)
  ↓ ReLU
  ↓ MaxPool2d (2×2)
  ↓ Conv2d (32→64, 3×3, padding=1)
  ↓ ReLU
  ↓ MaxPool2d (2×2)
  ↓ Flatten → Linear → ReLU → Linear → 输出
```

### 2.3 代码模板

```python
testloader = DataLoader(testset, batch_size=100, shuffle=False, num_workers=2)

    def __init__(self):
        super(CNNNet, self).__init__()
        # ★ TODO: 定义卷积层和池化层
```

## 实验3：池化降维效果对比

### 3.1 实验目的
- 理解池化层在 CNN 中的降维作用
- 对比：使用池化 vs 使用 stride=2 卷积替代的效果差异

### 3.2 代码模板

```python
"""
实验3：池化降维效果对比
任务：对比使用MaxPool和使用stride=2卷积的差异
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
import torchvision
import torchvision.transforms as transforms
from torch.utils.data import DataLoader

transform = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))
])
trainset = torchvision.datasets.MNIST(root='./data', train=True, download=True, transform=transform)
trainloader = DataLoader(trainset, batch_size=64, shuffle=True, num_workers=2)
testset = torchvision.datasets.MNIST(root='./data', train=False, download=True, transform=transform)
testloader = DataLoader(testset, batch_size=100, shuffle=False, num_workers=2)

# 版本A：使用MaxPool
class NetWithPool(nn.Module):
    def __init__(self):
        super(NetWithPool, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.pool = nn.MaxPool2d(2, 2)
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        x = self.pool(F.relu(self.conv1(x)))
        x = self.pool(F.relu(self.conv2(x)))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

# 版本B：使用stride=2卷积替代池化
class NetWithStride(nn.Module):
    def __init__(self):
        super(NetWithStride, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1, stride=1)
        self.conv2_stride = nn.Conv2d(32, 64, 3, padding=1, stride=2)
        self.conv3_stride = nn.Conv2d(64, 64, 3, padding=1, stride=2)
        self.fc1 = nn.Linear(64 * 3 * 3, 128)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        x = F.relu(self.conv1(x))
        x = F.relu(self.conv2_stride(x))
        x = F.relu(self.conv3_stride(x))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

def train_and_compare(model, trainloader, testloader, name, num_epochs=5):
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    
    params = sum(p.numel() for p in model.parameters())
    print(f"\n训练模型: {name}, 参数量: {params:,}")
    
    for epoch in range(num_epochs):
        model.train()
        correct = 0
        total = 0
        for images, labels in trainloader:
            outputs = model(images)
            loss = criterion(outputs, labels)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        
        train_acc = 100 * correct / total
        
        model.eval()
        correct = 0
        total = 0
        with torch.no_grad():
            for images, labels in testloader:
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        
        test_acc = 100 * correct / total
        print(f'Epoch [{epoch+1}/{num_epochs}], Train: {train_acc:.2f}%, Test: {test_acc:.2f}%')
    
    return params, test_acc

if __name__ == '__main__':
    params_pool, acc_pool = train_and_compare(NetWithPool(), trainloader, testloader, "版本A: MaxPool")
    params_stride, acc_stride = train_and_compare(NetWithStride(), trainloader, testloader, "版本B: Stride卷积")
    
    print(f"\n对比结果:")
    print(f"MaxPool: 参数量={params_pool:,}, 精度={acc_pool:.2f}%")
    print(f"Stride:  参数量={params_stride:,}, 精度={acc_stride:.2f}%")
```

```python
"""
实验4：BatchNorm + Dropout 正则化
任务：对比有无BatchNorm和Dropout的训练效果差异
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
import torchvision
import torchvision.transforms as transforms
from torch.utils.data import DataLoader

transform = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))
])
trainset = torchvision.datasets.MNIST(root='./data', train=True, download=True, transform=transform)
trainloader = DataLoader(trainset, batch_size=64, shuffle=True, num_workers=2)
testset = torchvision.datasets.MNIST(root='./data', train=False, download=True, transform=transform)
testloader = DataLoader(testset, batch_size=100, shuffle=False, num_workers=2)

# 版本A：基础CNN
class BasicCNN(nn.Module):
    def __init__(self):
        super(BasicCNN, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.pool = nn.MaxPool2d(2, 2)
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        x = self.pool(F.relu(self.conv1(x)))
        x = self.pool(F.relu(self.conv2(x)))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

# 版本B：CNN + BatchNorm
class CNNCWithBN(nn.Module):
    def __init__(self):
        super(CNNCWithBN, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.bn1 = nn.BatchNorm2d(32)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.bn2 = nn.BatchNorm2d(64)
        self.pool = nn.MaxPool2d(2, 2)
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        x = self.pool(F.relu(self.bn1(self.conv1(x))))
        x = self.pool(F.relu(self.bn2(self.conv2(x))))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

# 版本C：CNN + BatchNorm + Dropout
class CNNWithBNDropout(nn.Module):
    def __init__(self, dropout_p=0.5):
        super(CNNWithBNDropout, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.bn1 = nn.BatchNorm2d(32)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.bn2 = nn.BatchNorm2d(64)
        self.pool = nn.MaxPool2d(2, 2)
        self.dropout = nn.Dropout(p=dropout_p)
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        x = self.pool(F.relu(self.bn1(self.conv1(x))))
        x = self.pool(F.relu(self.bn2(self.conv2(x))))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        x = self.dropout(x)
        return self.fc2(x)

def train_and_record(model, trainloader, testloader, name, num_epochs=10):
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    
    print(f"\n训练模型: {name}")
    
    for epoch in range(num_epochs):
        model.train()
        correct = 0
        total = 0
        for images, labels in trainloader:
            outputs = model(images)
            loss = criterion(outputs, labels)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        
        train_acc = 100 * correct / total
        
        model.eval()
        correct = 0
        total = 0
        with torch.no_grad():
            for images, labels in testloader:
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        
        test_acc = 100 * correct / total
        gap = train_acc - test_acc
        print(f'Epoch [{epoch+1}/{num_epochs}] Train: {train_acc:.2f}% | Test: {test_acc:.2f}% | Gap: {gap:.2f}%')

if __name__ == '__main__':
    train_and_record(BasicCNN(), trainloader, testloader, "版本A: 基础CNN", 10)
    train_and_record(CNNCWithBN(), trainloader, testloader, "版本B: +BatchNorm", 10)
    train_and_record(CNNWithBNDropout(0.5), trainloader, testloader, "版本C: +BN+Dropout", 10)
```

```python
transform_medium = transforms.Compose([

    transforms.RandomAffine(degrees=0, translate=(0.1, 0.1)),
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))
])

# 强增强（学生自定义）
transform_strong = transforms.Compose([
    transforms.RandomRotation(degrees=(-20, 20)),
    transforms.RandomAffine(degrees=0, translate=(0.15, 0.15), scale=(0.85, 1.15), shear=5),
    transforms.GaussianBlur(kernel_size=3, sigma=(0.1, 1.0)),
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))
])

class FinalCNN(nn.Module):
    def __init__(self, dropout_p=0.5):
        super(FinalCNN, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.bn1 = nn.BatchNorm2d(32)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.bn2 = nn.BatchNorm2d(64)
        self.conv3 = nn.Conv2d(64, 128, 3, padding=1)
        self.bn3 = nn.BatchNorm2d(128)
        self.pool = nn.MaxPool2d(2, 2)
        self.dropout = nn.Dropout(p=dropout_p)
        self.fc1 = nn.Linear(128 * 3 * 3, 256)
        self.fc2 = nn.Linear(256, 10)
    
    def forward(self, x):
        x = self.pool(F.relu(self.bn1(self.conv1(x))))
        x = self.pool(F.relu(self.bn2(self.conv2(x))))
        x = self.pool(F.relu(self.bn3(self.conv3(x))))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        x = self.dropout(x)
        return self.fc2(x)

def train_model(model, trainloader, testloader, name, num_epochs=15):
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    scheduler = torch.optim.lr_scheduler.StepLR(optimizer, step_size=5, gamma=0.5)
    
    print(f"\n训练策略: {name}")
    best_acc = 0.0
    
    for epoch in range(num_epochs):
        model.train()
        correct = 0
        total = 0
        for images, labels in trainloader:
            outputs = model(images)
            loss = criterion(outputs, labels)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        
        scheduler.step()
        train_acc = 100 * correct / total
        
        model.eval()
        correct = 0
        total = 0
        with torch.no_grad():
            for images, labels in testloader:
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        
        test_acc = 100 * correct / total
        if test_acc > best_acc:
            best_acc = test_acc
        print(f'Epoch [{epoch+1}/{num_epochs}] Train: {train_acc:.2f}% | Test: {test_acc:.2f}% | Best: {best_acc:.2f}%')
    
    return best_acc

if __name__ == '__main__':
    testset = torchvision.datasets.MNIST(root='./data', train=False, download=True, transform=transform_weak)
    testloader = DataLoader(testset, batch_size=100, shuffle=False)
    
    results = {}
    
    # 弱增强
```

## 实验1：纯FC网络基准线

### 📖 理论背景

#### 什么是全连接层（Fully Connected Layer）？

全连接层是神经网络中最基础的层结构。每个神经元与上一层的所有神经元都有连接。

```
输入层(784个神经元)          隐藏层(256个神经元)         输出层(10个神经元)
     ○──────○──────○                  ○                        ○
     │      │      │                 /│\                      /│\
     │      │      │                / │ \                    / │ \
     ○──────○──────○───────────────○  │  ○──────────────────○  │  ○
     │      │      │               \ │ /                    \ │ /
     │      │      │                \│/                      \│/
     ○──────○──────○─────────────────○────────────────────────○──────○
```

**数学公式：** y = W·x + b

- W：权重矩阵（学习参数）
- x：输入向量
- b：偏置向量（学习参数）
- y：输出向量

#### 数据预处理为什么重要？

MNIST图像像素值范围是 [0, 255]，直接训练会导致：
1. 梯度更新方向不一致
2. 收敛速度慢
3. 容易陷入局部最优

**归一化** 将像素值映射到 [-1, 1]，让数据分布更均衡。

### 🎯 实验目标

1. 理解数据预处理（归一化）的作用
2. 掌握MNIST数据加载方法
3. 建立纯FC网络的基准精度
4. 学会计算网络参数量

### 🔧 积木组合

```

  ↓
```

```python
"""
实验1：纯FC网络基准线
==============================
本实验是所有实验的基准线，搭建最简单的全连接网络。
"""

# ========== 步骤1：导入必要的库 ==========
import torch                    # PyTorch核心库
import torch.nn as nn           # 神经网络模块（包含全连接层、激活函数等）
import torch.nn.functional as F # 神经网络常用函数（如ReLU）
import torchvision              # 计算机视觉工具库
import torchvision.transforms as transforms  # 数据预处理工具
from torch.utils.data import DataLoader       # 数据加载工具

# ========== 步骤2：数据加载 ==========
# transform: 数据预处理流水线
transform = transforms.Compose([
    transforms.ToTensor(),              # 1. 转为张量：图片 → 浮点数张量
                                        #    像素值从 [0, 255] → [0.0, 1.0]
                                        #    形状从 (H, W, C) → (C, H, W)
    transforms.Normalize((0.5,), (0.5,))  # 2. 标准化：均值0.5，标准差0.5
                                        #    像素值从 [0.0, 1.0] → [-1.0, 1.0]
])

# 下载并加载MNIST训练集
trainset = torchvision.datasets.MNIST(
    root='./data',        # 数据保存路径（首次会自动下载）
    train=True,           # True=训练集，False=测试集
    download=True,        # True=自动下载（如果本地没有）
    transform=transform    # 数据预处理流水线
)

# DataLoader：数据批量加载器
trainloader = DataLoader(
    dataset=trainset,     # 数据集
    batch_size=64,        # 每批样本数量（显存不足时可改小）
    shuffle=True,          # True=打乱数据（训练时推荐）
    num_workers=2          # 加载数据的进程数（多线程加速）
)

testset = torchvision.datasets.MNIST(root='./data', train=False, download=True, transform=transform)
testloader = DataLoader(testset, batch_size=100, shuffle=False, num_workers=2)

print(f"训练集样本数: {len(trainset)}")  # 60000
print(f"测试集样本数: {len(testset)}")   # 10000
print(f"单张图片尺寸: {trainset[0][0].shape}")  # torch.Size([1, 28, 28])

# ========== 步骤3：定义神经网络模型 ==========
class SimpleFCNet(nn.Module):
    """
    简单的全连接神经网络
    用于MNIST手写数字分类（10分类）
    
    网络结构：
    输入(784) → 隐藏层(256) → ReLU → 输出(10)
    """
    
    def __init__(self):
        """初始化网络层"""
        super(SimpleFCNet, self).__init__()  # 调用父类初始化
        
        # 全连接层1：输入784维 → 输出256维
        # 参数形状：(in_features=784, out_features=256)
        self.fc1 = nn.Linear(784, 256)
        
        # 全连接层2：输入256维 → 输出10维（10分类）
        self.fc2 = nn.Linear(256, 10)
    
    def forward(self, x):
        """
        前向传播：数据从输入到输出的计算过程
        
        参数：
            x: 输入张量，形状 (batch_size, 1, 28, 28)
        
        返回：
            输出张量，形状 (batch_size, 10)
        """
        # 1. 展平：从 (batch, 1, 28, 28) → (batch, 784)
        x = x.view(x.size(0), -1)
        
        # 2. 第一个全连接层 + ReLU激活
        x = F.relu(self.fc1(x))
        
        # 3. 第二个全连接层（输出层，不加激活）
        # 配合CrossEntropyLoss使用时，输出层不加激活函数
        x = self.fc2(x)
        
        return x

# ========== 步骤4：训练函数 ==========
def train_model(model, trainloader, testloader, num_epochs=5, lr=0.001):
    """
    训练神经网络模型
    
    参数：
        model: 神经网络模型
        trainloader: 训练数据加载器
        testloader: 测试数据加载器
        num_epochs: 训练轮数
        lr: 学习率（learning rate）
    
    返回：
        训练好的模型
    """
    # 1. 损失函数：交叉熵损失（适用于多分类问题）
    criterion = nn.CrossEntropyLoss()
    
    # 2. 优化器：Adam（自适应学习率优化器）
    optimizer = torch.optim.Adam(model.parameters(), lr=lr)
    
    # 训练循环
    for epoch in range(num_epochs):
        # ====== 训练阶段 ======
        model.train()  # 设置为训练模式
        
        running_loss = 0.0
        correct = 0
        total = 0
        
        for batch_idx, (images, labels) in enumerate(trainloader):
            # 前向传播
            outputs = model(images)
            loss = criterion(outputs, labels)
            
            # 反向传播
            optimizer.zero_grad()  # 清除梯度
            loss.backward()        # 计算梯度
            optimizer.step()        # 更新参数
            
            running_loss += loss.item()
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        
        train_loss = running_loss / len(trainloader)
        train_acc = 100 * correct / total
        
        # ====== 测试阶段 ======
        model.eval()  # 设置为评估模式
        correct = 0
        total = 0
        with torch.no_grad():
            for images, labels in testloader:
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        
        test_acc = 100 * correct / total
        
        print(f'Epoch [{epoch+1}/{num_epochs}] Loss: {train_loss:.4f} | Train: {train_acc:.2f}% | Test: {test_acc:.2f}%')
    
    return model

# ========== 步骤5：主程序 ==========
if __name__ == '__main__':
    model = SimpleFCNet()
    
    # 参数量计算
    # fc1: 784 × 256 + 256 = 200,960
    # fc2: 256 × 10 + 10 = 2,570
    # 总计：203,530
    total_params = sum(p.numel() for p in model.parameters())
    print(f"网络参数量: {total_params:,}")
    
    print("\n开始训练...")
    print("=" * 50)
    model = train_model(model, trainloader, testloader, num_epochs=5)
    
    torch.save(model.state_dict(), 'exp1_fc_net.pth')
    print("模型已保存到 exp1_fc_net.pth")
```

## 实验2：卷积特征提取

### 📖 理论背景

#### 什么是卷积层（Convolutional Layer）？

卷积层通过**卷积核**在图像上滑动，提取局部特征。不同于全连接层的"全局连接"，卷积层只处理局部区域。

```
卷积操作示意：
输入图像 (5×5) + 卷积核 (3×3) → 输出特征图 (3×3)
卷积核在图像上滑动，逐位置做加权求和
```

#### 卷积层核心参数

| 参数 | 说明 | 常用值 |
|------|------|-------|
| in_channels | 输入通道数（灰度图=1，RGB图=3） | 1 / 3 |
| out_channels | 输出通道数（卷积核数量） | 16 / 32 / 64 |
| kernel_size | 卷积核大小 | 3 / 5 / 7 |
| stride | 滑动步长 | 1（保持尺寸）/ 2（降维） |
| padding | 边缘填充 | 0（valid）/ 1（保持尺寸） |

#### 形状计算公式

```
H_out = floor((H_in + 2×padding - kernel_size) / stride) + 1
W_out = floor((W_in + 2×padding - kernel_size) / stride) + 1
```

### 🎯 实验目标

1. 理解卷积层如何提取图像特征
2. 掌握 Conv → ReLU → Pool 的标准组合
3. 对比 CNN 与 FC 网络的参数量差异

### 🔧 积木组合

```
输入图像 [1, 28, 28]
  ↓
Conv2d(1→32, 3×3, padding=1)  ← 卷积层
  ↓
ReLU                           ← 激活函数
  ↓
MaxPool2d(2×2)               ← 池化层（尺寸减半）
  ↓
Conv2d(32→64, 3×3, padding=1)
  ↓
ReLU
  ↓
MaxPool2d(2×2)               ← 尺寸减半：14→7
  ↓
Flatten
  ↓
Linear(64×7×7 → 128)         ← 全连接层
  ↓
ReLU
  ↓
Linear(128 → 10)
  ↓
输出 [batch, 10]
```

### 💻 代码详解

```python
"""
实验2：卷积特征提取
==============================
本实验引入卷积层，理解它如何从图像中提取特征。
"""

transform = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))
])
trainset = torchvision.datasets.MNIST(root='./data', train=True, download=True, transform=transform)
trainloader = DataLoader(trainset, batch_size=64, shuffle=True, num_workers=2)
testset = torchvision.datasets.MNIST(root='./data', train=False, download=True, transform=transform)
testloader = DataLoader(testset, batch_size=100, shuffle=False, num_workers=2)

class CNNNet(nn.Module):
    """
    卷积神经网络
    
    网络结构：
    Conv → ReLU → Pool → Conv → ReLU → Pool → Flatten → FC → FC
    """
    
    def __init__(self):
        super(CNNNet, self).__init__()
        
        # 卷积层1：1通道 → 32通道
        # in_channels=1: 灰度图
        # out_channels=32: 32个卷积核，输出32通道
        # kernel_size=3: 3×3卷积核
        # padding=1: 边缘填充，保持尺寸28×28
        self.conv1 = nn.Conv2d(in_channels=1, out_channels=32, kernel_size=3, padding=1)
        
        # 卷积层2：32通道 → 64通道
        self.conv2 = nn.Conv2d(in_channels=32, out_channels=64, kernel_size=3, padding=1)
        
        # 池化层：2×2最大池化，尺寸减半
        self.pool = nn.MaxPool2d(kernel_size=2, stride=2)
        
        # 全连接层
        # 64通道 × 7×7 = 3136（经过两次2×2池化）
        self.fc1 = nn.Linear(in_features=64 * 7 * 7, out_features=128)
        self.fc2 = nn.Linear(in_features=128, out_features=10)
    
    def forward(self, x):
        # Conv1 → ReLU → Pool
        x = self.pool(F.relu(self.conv1(x)))  # 28→14
        # Conv2 → ReLU → Pool
        x = self.pool(F.relu(self.conv2(x)))  # 14→7
        # 展平
        x = x.view(x.size(0), -1)
        # FC → ReLU → FC
        x = F.relu(self.fc1(x))
        x = self.fc2(x)
        return x

# 训练函数
def train_and_evaluate(model, trainloader, testloader, num_epochs=5):
    criterion = nn.CrossEntropyLoss()

    for epoch in range(num_epochs):

    return model
```

## 实验3：池化降维效果对比

### 📖 理论背景

#### 什么是池化层（Pooling Layer）？

池化层对特征图进行下采样（降维），减少计算量和参数量，同时增强特征的平移不变性。

```
最大池化示意：
┌─────────────┐
│ 1  3  2  1  │   2×2窗口，步长2
│ 5  6  7  8  │   ↓
│ 9  7  3  2  │   取每个窗口的最大值
│ 2  4  6  1  │
└─────────────┘
     ↓
┌───────┬───────┐
│   6   │   8   │
├───────┼───────┤
│   9   │   6   │
└───────┴───────┘
```

#### 池化 vs stride卷积

| 方式 | 优点 | 缺点 |
|------|------|------|
| 池化 | 计算简单、无参数量 | 丢失信息 |
| stride卷积 | 可学习、有参数 | 计算量大 |

### 🎯 实验目标

1. 理解池化层的降维作用
2. 对比 MaxPool 和 stride 卷积的效果

### 💻 代码详解

```python
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        train_acc = 100 * correct / total
        
        model.eval()
        correct = 0
        total = 0
        with torch.no_grad():
            for images, labels in testloader:
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        test_acc = 100 * correct / total
        print(f'Epoch [{epoch+1}/{num_epochs}] Train: {train_acc:.2f}% | Test: {test_acc:.2f}%')
    return params, test_acc

if __name__ == '__main__':
    params_pool, acc_pool = train_and_compare(NetWithPool(), trainloader, testloader, "版本A: MaxPool")

    print(f"\n对比结果: MaxPool={acc_pool:.2f}%, Stride={acc_stride:.2f}%")
```

## 实验4：BatchNorm + Dropout 正则化

### 📖 理论背景

#### 什么是BatchNorm？

BatchNorm对每个batch的数据进行标准化，使数据分布稳定。

```
原始数据：均值=5, 方差=10 → BatchNorm后：均值=0, 方差=1
```

**作用：** 加速收敛、缓解梯度消失、对初始值不敏感

#### 什么是Dropout？

训练时随机丢弃一部分神经元，防止过拟合。

### 🎯 实验目标

1. 理解BatchNorm如何加速收敛
2. 理解Dropout如何防止过拟合
3. 观察训练/测试精度差距（Gap）

## 实验5：数据增强 + 精度竞赛

### 📖 理论背景

#### 什么是数据增强？

通过对训练数据施加随机变换，人工扩充数据多样性。

```
原始图片 → 随机旋转 → 随机平移 → 高斯模糊
```

### 🎯 实验目标

1. 掌握torchvision.transforms数据增强方法
2. 对比弱/中/强数据增强的效果

### 💻 代码详解

```python
"""
实验5：数据增强 + 精度竞赛
"""

# 策略A：弱增强（仅归一化）
transform_weak = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))
])

# 策略B：中等增强
transform_medium = transforms.Compose([
    transforms.RandomRotation(degrees=(-10, 10)),

    transforms.GaussianBlur(kernel_size=3, sigma=(0.1, 1.0)),
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))

class FinalCNN(nn.Module):
    def __init__(self, dropout_p=0.5):
        super(FinalCNN, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.bn1 = nn.BatchNorm2d(32)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.bn2 = nn.BatchNorm2d(64)
        self.conv3 = nn.Conv2d(64, 128, 3, padding=1)
        self.bn3 = nn.BatchNorm2d(128)
        self.pool = nn.MaxPool2d(2, 2)
        self.dropout = nn.Dropout(p=dropout_p)
        self.fc1 = nn.Linear(128 * 3 * 3, 256)
        self.fc2 = nn.Linear(256, 10)
    
    def forward(self, x):
        x = self.pool(F.relu(self.bn1(self.conv1(x))))
        x = self.pool(F.relu(self.bn2(self.conv2(x))))
        x = self.pool(F.relu(self.bn3(self.conv3(x))))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        x = self.dropout(x)
        return self.fc2(x)

def train_model(model, trainloader, testloader, name, num_epochs=15):
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    scheduler = torch.optim.lr_scheduler.StepLR(optimizer, step_size=5, gamma=0.5)
    
    print(f"\n训练策略: {name}")
    best_acc = 0.0
    for epoch in range(num_epochs):
        model.train()
        correct = 0
        total = 0
        for images, labels in trainloader:
            outputs = model(images)
            loss = criterion(outputs, labels)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        scheduler.step()
        train_acc = 100 * correct / total
        
        model.eval()
        correct = 0
        total = 0
        with torch.no_grad():
            for images, labels in testloader:
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        test_acc = 100 * correct / total
        if test_acc > best_acc:
            best_acc = test_acc
        print(f'Epoch [{epoch+1}/{num_epochs}] Train: {train_acc:.2f}% | Test: {test_acc:.2f}% | Best: {best_acc:.2f}%')
    return best_acc

if __name__ == '__main__':
```
