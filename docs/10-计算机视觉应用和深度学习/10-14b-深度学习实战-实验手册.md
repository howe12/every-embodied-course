## 实验1：纯FC网络基准线

### 1.1 实验目的

### 1.2 代码模板

```python
"""
实验1：纯FC网络基准线
任务：使用单层全连接网络训练MNIST分类器
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
import torchvision
import torchvision.transforms as transforms
from torch.utils.data import DataLoader

# 数据加载
transform = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))  # 标准化到 [-1, 1]
])

trainset = torchvision.datasets.MNIST(root='./data', train=True, download=True, transform=transform)
trainloader = DataLoader(trainset, batch_size=64, shuffle=True, num_workers=2)
testset = torchvision.datasets.MNIST(root='./data', train=False, download=True, transform=transform)
testloader = DataLoader(testset, batch_size=100, shuffle=False, num_workers=2)

print(f"训练集样本数: {len(trainset)}, 测试集样本数: {len(testset)}")

# 定义FC网络
class SimpleFCNet(nn.Module):
    def __init__(self):
        super(SimpleFCNet, self).__init__()
        # ★ TODO: 输入784维(28x28展平)，输出10分类
        self.fc1 = nn.Linear(784, 256)
        self.fc2 = nn.Linear(256, 10)
    
    def forward(self, x):
        x = x.view(x.size(0), -1)  # 展平
        x = F.relu(self.fc1(x))
        return self.fc2(x)  # 输出层不加激活，配合CrossEntropyLoss

# 训练函数
def train_model(model, trainloader, testloader, num_epochs=5, lr=0.001):
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=lr)
    
    for epoch in range(num_epochs):
        model.train()
        running_loss = 0.0
        correct = 0
        total = 0
        
        for images, labels in trainloader:
            outputs = model(images)
            loss = criterion(outputs, labels)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            
            running_loss += loss.item()
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        
        train_acc = 100 * correct / total
        
        model.eval()
        correct = 0
        total = 0
        with torch.no_grad():
            for images, labels in testloader:
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        
        test_acc = 100 * correct / total
        print(f'Epoch [{epoch+1}/{num_epochs}], Train: {train_acc:.2f}%, Test: {test_acc:.2f}%')
    
    return model

# 运行实验
if __name__ == '__main__':
    model = SimpleFCNet()
    print(f"网络参数量: {sum(p.numel() for p in model.parameters()):,}")
    model = train_model(model, trainloader, testloader)
    torch.save(model.state_dict(), 'exp1_fc_net.pth')
```

## 实验2：卷积特征提取

### 2.1 实验目的
- 理解卷积层如何提取图像特征
- 掌握 Conv2d + ReLU + Pooling 的组合方式
- 对比 CNN vs 纯 FC 的参数量和精度

### 2.2 积木组合
```
输入 [1, 28, 28]
  ↓ Conv2d (1→32, 3×3, padding=1)
  ↓ ReLU
  ↓ MaxPool2d (2×2)
  ↓ Conv2d (32→64, 3×3, padding=1)
  ↓ ReLU
  ↓ MaxPool2d (2×2)
  ↓ Flatten → Linear → ReLU → Linear → 输出
```

### 2.3 代码模板

testloader = DataLoader(testset, batch_size=100, shuffle=False, num_workers=2)

    def __init__(self):
        super(CNNNet, self).__init__()
        # ★ TODO: 定义卷积层和池化层

        self.conv2 = nn.Conv2d(32, 64, kernel_size=3, padding=1)

任务：对比使用MaxPool和使用stride=2卷积的差异

import torch
import torch.nn as nn
import torch.nn.functional as F
import torchvision
import torchvision.transforms as transforms
from torch.utils.data import DataLoader

transform = transforms.Compose([
    transforms.ToTensor(),

])
trainset = torchvision.datasets.MNIST(root='./data', train=True, download=True, transform=transform)
trainloader = DataLoader(trainset, batch_size=64, shuffle=True, num_workers=2)
testset = torchvision.datasets.MNIST(root='./data', train=False, download=True, transform=transform)
testloader = DataLoader(testset, batch_size=100, shuffle=False, num_workers=2)

# 版本A：使用MaxPool
class NetWithPool(nn.Module):
    def __init__(self):
        super(NetWithPool, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.pool = nn.MaxPool2d(2, 2)
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        x = self.pool(F.relu(self.conv1(x)))
        x = self.pool(F.relu(self.conv2(x)))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

# 版本B：使用stride=2卷积替代池化
class NetWithStride(nn.Module):
    def __init__(self):
        super(NetWithStride, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1, stride=1)
        self.conv2_stride = nn.Conv2d(32, 64, 3, padding=1, stride=2)
        self.conv3_stride = nn.Conv2d(64, 64, 3, padding=1, stride=2)
        self.fc1 = nn.Linear(64 * 3 * 3, 128)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        x = F.relu(self.conv1(x))
        x = F.relu(self.conv2_stride(x))
        x = F.relu(self.conv3_stride(x))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

def train_and_compare(model, trainloader, testloader, name, num_epochs=5):
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    
    params = sum(p.numel() for p in model.parameters())
    print(f"\n训练模型: {name}, 参数量: {params:,}")
    
    for epoch in range(num_epochs):
        model.train()
        correct = 0
        total = 0
        for images, labels in trainloader:
            outputs = model(images)
            loss = criterion(outputs, labels)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        
        train_acc = 100 * correct / total
        
        model.eval()
        correct = 0
        total = 0
        with torch.no_grad():
            for images, labels in testloader:
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        
        test_acc = 100 * correct / total
        print(f'Epoch [{epoch+1}/{num_epochs}], Train: {train_acc:.2f}%, Test: {test_acc:.2f}%')
    
    return params, test_acc

if __name__ == '__main__':
    params_pool, acc_pool = train_and_compare(NetWithPool(), trainloader, testloader, "版本A: MaxPool")
    params_stride, acc_stride = train_and_compare(NetWithStride(), trainloader, testloader, "版本B: Stride卷积")
    
    print(f"\n对比结果:")
    print(f"MaxPool: 参数量={params_pool:,}, 精度={acc_pool:.2f}%")
    print(f"Stride:  参数量={params_stride:,}, 精度={acc_stride:.2f}%")

            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        
        scheduler.step()
        train_acc = 100 * correct / total
        
        model.eval()
        correct = 0
        total = 0
        with torch.no_grad():
            for images, labels in testloader:
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        
        test_acc = 100 * correct / total
        if test_acc > best_acc:
            best_acc = test_acc
        print(f'Epoch [{epoch+1}/{num_epochs}] Train: {train_acc:.2f}% | Test: {test_acc:.2f}% | Best: {best_acc:.2f}%')
    
    return best_acc
