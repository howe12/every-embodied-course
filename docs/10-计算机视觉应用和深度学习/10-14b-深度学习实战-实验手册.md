## 实验1：纯FC网络基准线

### 📖 理论背景

#### 全连接层
每个神经元与上一层所有神经元都有连接。

#### 数据归一化
像素值 [0,255] → [0,1] → [-1,1]

### 💻 代码

```python
import torch
import torch.nn as nn
import torch.nn.functional as F
import torchvision
import torchvision.transforms as transforms
from torch.utils.data import DataLoader

transform = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))
])
trainset = torchvision.datasets.MNIST(root='./data', train=True, download=True, transform=transform)
trainloader = DataLoader(trainset, batch_size=64, shuffle=True, num_workers=2)
testset = torchvision.datasets.MNIST(root='./data', train=False, download=True, transform=transform)
testloader = DataLoader(testset, batch_size=100, shuffle=False, num_workers=2)

class SimpleFCNet(nn.Module):
    def __init__(self):
        super(SimpleFCNet, self).__init__()
        self.fc1 = nn.Linear(784, 256)
        self.fc2 = nn.Linear(256, 10)
    
    def forward(self, x):
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

def train_model(model, trainloader, testloader, num_epochs=5, lr=0.001):
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=lr)
    for epoch in range(num_epochs):
        model.train()
        correct = total = 0
        for images, labels in trainloader:
            outputs = model(images)
            loss = criterion(outputs, labels)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        train_acc = 100 * correct / total
        model.eval()
        correct = total = 0
        with torch.no_grad():
            for images, labels in testloader:
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        test_acc = 100 * correct / total
        print(f'Epoch [{epoch+1}/{num_epochs}] Train: {train_acc:.2f}% | Test: {test_acc:.2f}%')
    return model

if __name__ == '__main__':
    model = SimpleFCNet()
    print(f"参数量: {sum(p.numel() for p in model.parameters()):,}")
    model = train_model(model, trainloader, testloader)
```

## 实验2：卷积特征提取

### 📖 理论背景

#### 卷积层
通过卷积核在图像上滑动提取局部特征。

```
输入(5×5) + 卷积核(3×3) → 输出(3×3)
```

#### 核心参数

| 参数 | 说明 | 常用值 |
|------|------|-------|
| in_channels | 输入通道数 | 1(灰度)/3(RGB) |
| out_channels | 输出通道数 | 16/32/64 |
| kernel_size | 卷积核大小 | 3/5/7 |
| stride | 滑动步长 | 1(保持)/2(降维) |
| padding | 边缘填充 | 0/1 |

### 💻 代码

```python
class CNNNet(nn.Module):
    def __init__(self):
        super(CNNNet, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, kernel_size=3, padding=1)
        self.conv2 = nn.Conv2d(32, 64, kernel_size=3, padding=1)
        self.pool = nn.MaxPool2d(2, 2)
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        x = self.pool(F.relu(self.conv1(x)))  # 28→14
        x = self.pool(F.relu(self.conv2(x)))  # 14→7
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

if __name__ == '__main__':
    model = CNNNet()
    print(f"参数量: {sum(p.numel() for p in model.parameters()):,}")
    # FC: 203K → CNN: 421K（参数更少，精度更高）
```

## 实验3：池化降维效果对比

### 📖 理论背景

#### 池化层
对特征图下采样，减少计算量和参数量。

```
最大池化：2×2窗口，取最大值
```

#### 池化 vs stride卷积

| 方式 | 优点 | 缺点 |
|------|------|------|
| 池化 | 计算简单、无参数 | 丢失信息 |
| stride卷积 | 可学习 | 计算量大 |

### 💻 代码

```python
# 版本A：MaxPool池化
class NetWithPool(nn.Module):
    def __init__(self):
        super(NetWithPool, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.pool = nn.MaxPool2d(2, 2)
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        x = self.pool(F.relu(self.conv1(x)))
        x = self.pool(F.relu(self.conv2(x)))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

# 版本B：stride卷积降维
class NetWithStride(nn.Module):
    def __init__(self):
        super(NetWithStride, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1, stride=1)
        self.conv2_stride = nn.Conv2d(32, 64, 3, padding=1, stride=2)
        self.conv3_stride = nn.Conv2d(64, 64, 3, padding=1, stride=2)
        self.fc1 = nn.Linear(64 * 3 * 3, 128)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        x = F.relu(self.conv1(x))
        x = F.relu(self.conv2_stride(x))
        x = F.relu(self.conv3_stride(x))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)
```

## 实验4：BatchNorm + Dropout 正则化

### 📖 理论背景

#### BatchNorm
对batch数据标准化，稳定训练、加速收敛。

#### Dropout
训练时随机丢弃神经元，防止过拟合。

### 💻 代码

```python
# 版本A：基础CNN
class BasicCNN(nn.Module):
    def __init__(self):
        super(BasicCNN, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.pool = nn.MaxPool2d(2, 2)
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.fc2 = nn.Linear(128, 10)
    def forward(self, x):
        x = self.pool(F.relu(self.conv1(x)))
        x = self.pool(F.relu(self.conv2(x)))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

# 版本B：+BatchNorm
class CNNCWithBN(nn.Module):
    def __init__(self):
        super(CNNCWithBN, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.bn1 = nn.BatchNorm2d(32)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.bn2 = nn.BatchNorm2d(64)
        self.pool = nn.MaxPool2d(2, 2)
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.fc2 = nn.Linear(128, 10)
    def forward(self, x):
        x = self.pool(F.relu(self.bn1(self.conv1(x))))
        x = self.pool(F.relu(self.bn2(self.conv2(x))))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

# 版本C：+BN+Dropout
class CNNWithBNDropout(nn.Module):
    def __init__(self, dropout_p=0.5):
        super(CNNWithBNDropout, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.bn1 = nn.BatchNorm2d(32)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.bn2 = nn.BatchNorm2d(64)
        self.pool = nn.MaxPool2d(2, 2)
        self.dropout = nn.Dropout(p=dropout_p)
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.fc2 = nn.Linear(128, 10)
    def forward(self, x):
        x = self.pool(F.relu(self.bn1(self.conv1(x))))
        x = self.pool(F.relu(self.bn2(self.conv2(x))))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        x = self.dropout(x)
        return self.fc2(x)
```

## 实验5：数据增强 + 精度竞赛

### 📖 理论背景

数据增强通过随机变换扩充数据多样性。

```
原始图片 → 随机旋转 → 随机平移 → 高斯模糊
```

### 💻 代码

```python
# 三种增强策略
transform_weak = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))
])

transform_medium = transforms.Compose([
    transforms.RandomRotation(degrees=(-10, 10)),
    transforms.RandomAffine(degrees=0, translate=(0.1, 0.1)),
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))
])

transform_strong = transforms.Compose([
    transforms.RandomRotation(degrees=(-20, 20)),
    transforms.RandomAffine(degrees=0, translate=(0.15, 0.15), scale=(0.85, 1.15), shear=5),
    transforms.GaussianBlur(kernel_size=3, sigma=(0.1, 1.0)),
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))
])

# 完整CNN模型
class FinalCNN(nn.Module):
    def __init__(self, dropout_p=0.5):
        super(FinalCNN, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.bn1 = nn.BatchNorm2d(32)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.bn2 = nn.BatchNorm2d(64)
        self.conv3 = nn.Conv2d(64, 128, 3, padding=1)
        self.bn3 = nn.BatchNorm2d(128)
        self.pool = nn.MaxPool2d(2, 2)
        self.dropout = nn.Dropout(p=dropout_p)
        self.fc1 = nn.Linear(128 * 3 * 3, 256)
        self.fc2 = nn.Linear(256, 10)
    def forward(self, x):
        x = self.pool(F.relu(self.bn1(self.conv1(x))))
        x = self.pool(F.relu(self.bn2(self.conv2(x))))
        x = self.pool(F.relu(self.bn3(self.conv3(x))))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        x = self.dropout(x)
        return self.fc2(x)

# 训练函数
def train_model(model, trainloader, testloader, name, num_epochs=15):
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    scheduler = torch.optim.lr_scheduler.StepLR(optimizer, step_size=5, gamma=0.5)
    best_acc = 0.0
    for epoch in range(num_epochs):
        model.train()
        correct = total = 0
        for images, labels in trainloader:
            outputs = model(images)
            loss = criterion(outputs, labels)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        scheduler.step()
        train_acc = 100 * correct / total
        model.eval()
        correct = total = 0
        with torch.no_grad():
            for images, labels in testloader:
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        test_acc = 100 * correct / total
        if test_acc > best_acc:
            best_acc = test_acc
        print(f'Epoch [{epoch+1}/{num_epochs}] Train: {train_acc:.2f}% | Test: {test_acc:.2f}%')
    return best_acc

if __name__ == '__main__':
    testloader = DataLoader(testset, batch_size=100, shuffle=False)
    trainset_a = torchvision.datasets.MNIST(root='./data', train=True, download=True, transform=transform_weak)
    results['弱增强'] = train_model(FinalCNN(0.3), DataLoader(trainset_a, batch_size=64, shuffle=True), testloader, "弱增强")
    trainset_b = torchvision.datasets.MNIST(root='./data', train=True, download=True, transform=transform_medium)
    results['中等增强'] = train_model(FinalCNN(0.4), DataLoader(trainset_b, batch_size=64, shuffle=True), testloader, "中等增强")
    trainset_c = torchvision.datasets.MNIST(root='./data', train=True, download=True, transform=transform_strong)
    results['强增强'] = train_model(FinalCNN(0.5), DataLoader(trainset_c, batch_size=64, shuffle=True), testloader, "强增强")
```
