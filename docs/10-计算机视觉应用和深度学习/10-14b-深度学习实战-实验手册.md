```

```

### 模块积木对照表

| 模块 | PyTorch | 核心参数 | 作用 |
|------|---------|---------|------|
| 全连接层 | `nn.Linear` | in_features, out_features | 特征组合分类 |
| 卷积层 | `nn.Conv2d` | in/out_channels, kernel_size, stride, padding | 空间特征提取 |
| 池化层 | `nn.MaxPool2d` | kernel_size, stride | 降维防过拟合 |
| 激活函数 | `nn.ReLU` | - | 引入非线性 |
| Dropout | `nn.Dropout` | p（丢弃概率） | 防过拟合 |
| BatchNorm | `nn.BatchNorm2d` | num_features | 归一化加速收敛 |

## 实验1：纯FC网络基准线

### 📖 实验说明

**实验目标**：建立纯全连接网络的基准线，理解数据预处理的重要性

**设计思路**：
1. 使用最简单的全连接网络（2层）
2. 对比归一化/不归一化的训练效果
3. 让小白理解：为什么需要数据归一化

**预期结果**：约92%测试精度

### 💡 基础模块讲解

#### 1. nn.Linear（全连接层）

```python
self.fc1 = nn.Linear(784, 256)  # 输入784维 → 输出256维
self.fc2 = nn.Linear(256, 10)    # 输入256维 → 输出10分类
```

- **含义**：每个神经元与上一层的所有神经元都有连接
- **参数**：in_features=输入维数，out_features=输出维数
- **作用**：进行特征的线性组合变换，用于分类或回归

**数学公式**：y = W·x + b
- W：权重矩阵（形状为 out_features × in_features）
- x：输入向量（形状为 in_features）
- b：偏置向量（形状为 out_features）
- y：输出向量（形状为 out_features）

#### 2. transforms.Compose（数据预处理流水线）

```python
transform = transforms.Compose([
    transforms.ToTensor(),              # 转张量：图片 → 浮点张量
    transforms.Normalize((0.5,), (0.5,))  # 标准化：像素值 [-1, 1]
])
```

- **ToTensor**：将PIL图片或numpy数组转为PyTorch张量，自动将像素值从[0,255]缩放到[0,1]
- **Normalize**：用均值0.5和标准差0.5进行标准化，将像素范围从[0,1]映射到[-1,1]

#### 3. DataLoader（数据加载器）

```python
trainloader = DataLoader(dataset=trainset, batch_size=64, shuffle=True)
```

- **dataset**：要加载的数据集
- **batch_size**：每批加载的样本数量
- **shuffle**：是否在每个epoch开始时打乱数据
- **作用**：批量加载数据，支持多进程加速

#### 4. nn.CrossEntropyLoss（交叉熵损失）

```python
criterion = nn.CrossEntropyLoss()
```

- **含义**：多分类问题的标准损失函数
- **作用**：衡量预测概率分布与真实标签的差异
- **注意**：内部已包含Softmax，不需要在输出层加激活函数

#### 5. torch.optim.Adam（Adam优化器）

```python
optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
```

- **含义**：自适应学习率的梯度下降算法
- **参数**：model.parameters()=模型所有参数，lr=学习率
- **优点**：收敛速度快，对学习率选择不那么敏感

### 💻 完整代码（含详细注释）

```python
"""
实验1：纯FC网络基准线
==============================
本实验是所有实验的基准线，我们搭建一个最简单的全连接网络。
通过对比归一化/不归一化的效果，理解数据预处理的重要性。
"""

# ========== 步骤1：导入必要的库 ==========
import torch                    # PyTorch核心库
import torch.nn as nn           # 神经网络模块（包含全连接层、激活函数等）
import torch.nn.functional as F  # 神经网络常用函数（如ReLU）
import torchvision              # 计算机视觉工具库
import torchvision.transforms as transforms  # 数据预处理工具
from torch.utils.data import DataLoader       # 数据加载工具

# ========== 步骤2：数据加载 ==========
# transform: 数据预处理流水线，将图片转为标准化的张量
transform = transforms.Compose([
    transforms.ToTensor(),              # 1. 转为张量：像素[0,255] → [0.0,1.0]
    transforms.Normalize((0.5,), (0.5,))  # 2. 标准化：均值0.5，标准差0.5
])

# 下载并加载MNIST训练集
trainset = torchvision.datasets.MNIST(
    root='./data',        # 数据保存路径（首次会自动下载）
    train=True,           # True=训练集，False=测试集
    download=True,        # True=自动下载
    transform=transform    # 数据预处理
)

# DataLoader：批量加载数据
trainloader = DataLoader(
    dataset=trainset,     # 数据集
    batch_size=64,        # 每批64张图片
    shuffle=True,         # 打乱数据顺序（训练时推荐）
    num_workers=2         # 使用2个进程加载
)

# 同样加载测试集
testset = torchvision.datasets.MNIST(root='./data', train=False, download=True, transform=transform)
testloader = DataLoader(testset, batch_size=100, shuffle=False, num_workers=2)

print(f"训练集样本数: {len(trainset)}")  # 60000
print(f"测试集样本数: {len(testset)}")   # 10000
print(f"单张图片尺寸: {trainset[0][0].shape}")  # torch.Size([1, 28, 28])

# ========== 步骤3：定义神经网络模型 ==========
class SimpleFCNet(nn.Module):
    """
    简单的全连接神经网络
    
    网络结构：
    输入(784维) → 隐藏层(256) → ReLU → 输出层(10)
    
    为什么是784？因为MNIST图片是28×28=784个像素展平
    为什么是10？因为数字0-9共10个类别
    """
    
    def __init__(self):
        """初始化网络层"""
        super(SimpleFCNet, self).__init__()  # 调用父类初始化
        
        # 全连接层1：输入784维 → 输出256维
        self.fc1 = nn.Linear(784, 256)
        
        # 全连接层2：输入256维 → 输出10维（10分类）
        self.fc2 = nn.Linear(256, 10)
    
    def forward(self, x):
        """
        前向传播：数据从输入到输出的计算过程
        
        参数：x - 输入张量，形状 (batch_size, 1, 28, 28)
        返回：输出张量，形状 (batch_size, 10)
        """
        # 1. 展平：从 (batch, 1, 28, 28) → (batch, 784)
        x = x.view(x.size(0), -1)
        
        # 2. 第一个全连接层 + ReLU激活
        x = F.relu(self.fc1(x))
        
        # 3. 第二个全连接层（输出层，不加激活函数）
        # 注意：配合CrossEntropyLoss使用时，输出层不需要Softmax
        x = self.fc2(x)
        
        return x

# ========== 步骤4：训练函数 ==========
def train_model(model, trainloader, testloader, num_epochs=5, lr=0.001):
    """
    训练神经网络模型
    
    参数：
        model - 神经网络模型
        trainloader - 训练数据加载器
        testloader - 测试数据加载器
        num_epochs - 训练轮数
        lr - 学习率（learning rate）
    
    返回：
        训练好的模型
    """
    # 1. 损失函数：交叉熵损失（适用于多分类问题）
    criterion = nn.CrossEntropyLoss()
    
    # 2. 优化器：Adam（自适应学习率优化器）
    optimizer = torch.optim.Adam(model.parameters(), lr=lr)
    
    # 训练循环
    for epoch in range(num_epochs):
        # ====== 训练阶段 ======
        model.train()  # 设置为训练模式（启用Dropout、BatchNorm等）
        
        running_loss = 0.0  # 累计损失
        correct = 0         # 正确预测数
        total = 0           # 总样本数
        
        for images, labels in trainloader:
            # 前向传播：计算模型预测
            outputs = model(images)
            
            # 计算损失：预测 vs 真实标签
            loss = criterion(outputs, labels)
            
            # 反向传播：计算梯度
            optimizer.zero_grad()  # 清除上一次的梯度
            loss.backward()        # 反向传播计算梯度
            optimizer.step()        # 更新参数
            
            # 累计统计
            running_loss += loss.item()
            _, predicted = torch.max(outputs.data, 1)  # 取最大分数的类别
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        
        train_loss = running_loss / len(trainloader)
        train_acc = 100 * correct / total
        
        # ====== 测试阶段 ======
        model.eval()  # 设置为评估模式（禁用Dropout）
        
        correct = 0
        total = 0
        with torch.no_grad():  # 不计算梯度（节省内存加速）
            for images, labels in testloader:
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        
        test_acc = 100 * correct / total
        
        print(f'Epoch [{epoch+1}/{num_epochs}] '
              f'Loss: {train_loss:.4f} | '
              f'Train: {train_acc:.2f}% | '
              f'Test: {test_acc:.2f}%')
    
    return model

# ========== 步骤5：主程序 ==========
if __name__ == '__main__':
    # 创建模型实例
    model = SimpleFCNet()
    
    # 统计模型参数量
    # fc1: 784 × 256 + 256 = 200,960（权重 + 偏置）
    # fc2: 256 × 10 + 10 = 2,570
    # 总计：203,530
    total_params = sum(p.numel() for p in model.parameters())
    print(f"网络参数量: {total_params:,}")
    
    # 开始训练
    print("\n开始训练...")
    print("=" * 50)
    model = train_model(model, trainloader, testloader, num_epochs=5)
    
    # 保存模型（只保存参数，不保存结构）
    torch.save(model.state_dict(), 'exp1_fc_net.pth')
    print("=" * 50)
    print("模型已保存到 exp1_fc_net.pth")
```

## 实验2：卷积特征提取

### 📖 实验说明

**实验目标**：引入卷积层，理解卷积如何提取图像特征

**设计思路**：
1. 用卷积层替代全连接层处理图像
2. 对比 CNN 与 FC 的参数量和精度差异
3. 让小白理解：卷积核是如何提取特征的

**预期结果**：约97%测试精度

### 💡 基础模块讲解

#### 1. nn.Conv2d（卷积层）

```python
self.conv1 = nn.Conv2d(in_channels=1, out_channels=32, kernel_size=3, padding=1)
```

- **含义**：2D卷积层，在图像上滑动卷积核提取特征
- **参数**：
  - in_channels=1：输入通道数（灰度图=1，RGB图=3）
  - out_channels=32：输出通道数（卷积核数量）
  - kernel_size=3：卷积核大小（3×3）
  - padding=1：边缘填充1圈像素，保持尺寸
- **作用**：通过局部连接和权重共享提取空间特征

#### 2. nn.MaxPool2d（最大池化层）

```python
self.pool = nn.MaxPool2d(kernel_size=2, stride=2)
```

- **含义**：2×2窗口取最大值，尺寸减半
- **参数**：
  - kernel_size=2：2×2窗口
  - stride=2：步长2（不重叠）
- **作用**：降维、减少计算量、增强平移不变性

#### 形状计算公式

```
H_out = floor((H_in + 2×padding - kernel_size) / stride) + 1

28×28输入：
  conv1(3×3, padding=1) → floor((28+2-3)/1)+1 = 28 → [32, 28, 28]
  pool(2×2) → floor(28/2) = 14 → [32, 14, 14]
  conv2(3×3, padding=1) → [64, 14, 14]
  pool(2×2) → [64, 7, 7]
```

### 💻 完整代码

```python
"""
实验2：卷积特征提取
==============================
本实验引入卷积层，理解它如何从图像中提取特征。
卷积层通过局部连接和权重共享，大幅减少参数量。
"""

class CNNNet(nn.Module):
    """
    卷积神经网络
    
    网络结构：
    Conv → ReLU → Pool → Conv → ReLU → Pool → Flatten → FC → FC
    
    特点：
    - 卷积层提取空间特征
    - 池化层降维防过拟合
    - 参数量远小于全连接网络
    """
    
    def __init__(self):
        super(CNNNet, self).__init__()
        
        # 卷积层1：1通道 → 32通道
        # in_channels=1: 输入是灰度图（1通道）
        # out_channels=32: 使用32个卷积核，输出32通道特征图
        # kernel_size=3: 3×3的卷积核
        # padding=1: 边缘填充，保持尺寸28×28
        self.conv1 = nn.Conv2d(in_channels=1, out_channels=32, kernel_size=3, padding=1)
        
        # 卷积层2：32通道 → 64通道
        self.conv2 = nn.Conv2d(in_channels=32, out_channels=64, kernel_size=3, padding=1)
        
        # 池化层：2×2最大池化，尺寸减半
        # 28×28 → 14×14 → 7×7
        self.pool = nn.MaxPool2d(kernel_size=2, stride=2)
        
        # 全连接层
        # 64通道 × 7×7 = 3136
        self.fc1 = nn.Linear(in_features=64 * 7 * 7, out_features=128)
        self.fc2 = nn.Linear(in_features=128, out_features=10)
    
    def forward(self, x):
        """
        前向传播
        
        参数：x - 输入张量，形状 (batch, 1, 28, 28)
        返回：输出张量，形状 (batch, 10)
        """
        # Conv1 → ReLU → Pool
        # 1. conv1: (batch, 1, 28, 28) → (batch, 32, 28, 28)
        # 2. relu: 不改变形状
        # 3. pool: (batch, 32, 28, 28) → (batch, 32, 14, 14)
        x = self.pool(F.relu(self.conv1(x)))
        
        # Conv2 → ReLU → Pool
        # 1. conv2: (batch, 32, 14, 14) → (batch, 64, 14, 14)
        # 2. relu: 不改变形状
        # 3. pool: (batch, 64, 14, 14) → (batch, 64, 7, 7)
        x = self.pool(F.relu(self.conv2(x)))
        
        # 展平：从 (batch, 64, 7, 7) → (batch, 3136)
        x = x.view(x.size(0), -1)
        
        # 全连接层
        x = F.relu(self.fc1(x))  # (batch, 3136) → (batch, 128)
        x = self.fc2(x)          # (batch, 128) → (batch, 10)
        
        return x

if __name__ == '__main__':
    model = CNNNet()
    
    # 参数量对比
    # conv1: 1×32×3×3 + 32 = 320
    # conv2: 32×64×3×3 + 64 = 18,496
    # fc1: 3136×128 + 128 = 401,536
    # fc2: 128×10 + 10 = 1,290
    # 总计：约 421,000
    params = sum(p.numel() for p in model.parameters())
    print(f"CNN网络参数量: {params:,}")
    print(f"对比：实验1的FC网络参数量为 203,530")
    print(f"结论：CNN参数更少，但精度更高！")
```

## 实验3：池化降维效果对比

### 📖 实验说明

**实验目标**：理解池化层的降维作用，对比 MaxPool 和 stride 卷积的效果

**设计思路**：
1. 用两种方式实现降维：MaxPool池化 vs stride=2卷积
2. 对比参数量和精度差异
3. 让小白理解：不同降维方式的权衡

### 💡 基础模块讲解

#### 1. nn.MaxPool2d（最大池化层）

```python
self.pool = nn.MaxPool2d(kernel_size=2, stride=2)
```

- **含义**：2×2窗口取最大值，尺寸减半
- **参数**：
  - kernel_size=2：2×2窗口
  - stride=2：步长2，不重叠
- **优点**：计算简单，几乎无额外参数
- **缺点**：信息丢失（取最大值可能丢失有用信息）

#### 2. stride卷积降维

```python
self.conv_stride = nn.Conv2d(32, 64, kernel_size=3, stride=2, padding=1)
```

- **含义**：用卷积的步长来实现降维
- **优点**：降维时可学习，有参数
- **缺点**：增加参数量

### 💻 完整代码

```python
"""
实验3：池化降维效果对比
==============================
对比两种降维方式：
A. MaxPool池化：计算简单，不可学习
B. stride卷积：可学习，但参数量更大
"""

# 版本A：使用MaxPool降维
class NetWithPool(nn.Module):
    """
    使用MaxPool的CNN
    结构：Conv → ReLU → Pool → Conv → ReLU → Pool → FC
    """
    def __init__(self):
        super(NetWithPool, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        # 池化层（降维）
        self.pool = nn.MaxPool2d(2, 2)  # 尺寸减半
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        x = self.pool(F.relu(self.conv1(x)))  # 28→14
        x = self.pool(F.relu(self.conv2(x)))  # 14→7
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

# 版本B：使用stride=2卷积降维
class NetWithStride(nn.Module):
    """
    使用stride卷积降维的CNN
    结构：Conv → Conv(s=2) → Conv(s=2) → FC
    """
    def __init__(self):
        super(NetWithStride, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1, stride=1)  # 保持尺寸
        # stride=2 替代池化
        self.conv2_stride = nn.Conv2d(32, 64, 3, padding=1, stride=2)  # 28→14
        self.conv3_stride = nn.Conv2d(64, 64, 3, padding=1, stride=2)  # 14→7
        # 注意：fc1输入尺寸变化了
        self.fc1 = nn.Linear(64 * 3 * 3, 128)  # 尺寸变化
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        x = F.relu(self.conv1(x))              # 保持28
        x = F.relu(self.conv2_stride(x))       # 28→14
        x = F.relu(self.conv3_stride(x))        # 14→7
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)
```

## 实验4：BatchNorm + Dropout 正则化

### 📖 实验说明

**实验目标**：理解 BatchNorm 和 Dropout 如何防止过拟合

**设计思路**：
1. 对比三种网络：基础CNN、+BatchNorm、+BatchNorm+Dropout
2. 观察 Train-Test Gap（过拟合程度）
3. 让小白理解：正则化的作用

### 💡 基础模块讲解

#### 1. nn.BatchNorm2d（批归一化层）

```python
self.bn1 = nn.BatchNorm2d(32)
```

- **含义**：对batch内的数据进行标准化
- **参数**：num_features=通道数
- **作用**：
  - 加速收敛
  - 缓解梯度消失
  - 对初始值不敏感
- **位置**：通常在卷积层之后、激活函数之前

#### 2. nn.Dropout（随机丢弃层）

```python
self.dropout = nn.Dropout(p=0.5)
```

- **含义**：训练时随机丢弃一部分神经元
- **参数**：p=丢弃概率（0.5表示50%神经元被置零）
- **作用**：防止过拟合，让网络不依赖特定神经元
- **注意**：训练时启用，评估时自动关闭

### 💻 完整代码

```python
"""
实验4：BatchNorm + Dropout 正则化
对比三种网络：
A. 基础CNN：无正则化
B. +BatchNorm：加速收敛
C. +BatchNorm+Dropout：防止过拟合
"""

# 版本A：基础CNN
class BasicCNN(nn.Module):
    def __init__(self):
        super(BasicCNN, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.pool = nn.MaxPool2d(2, 2)
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        x = self.pool(F.relu(self.conv1(x)))  # 28→14
        x = self.pool(F.relu(self.conv2(x)))  # 14→7
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

# 版本B：+BatchNorm
class CNNCWithBN(nn.Module):
    def __init__(self):
        super(CNNCWithBN, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.bn1 = nn.BatchNorm2d(32)  # BN层
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.bn2 = nn.BatchNorm2d(64)  # BN层
        self.pool = nn.MaxPool2d(2, 2)
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        # BatchNorm在激活函数之前
        x = self.pool(F.relu(self.bn1(self.conv1(x))))  # Conv → BN → ReLU
        x = self.pool(F.relu(self.bn2(self.conv2(x))))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

# 版本C：+BatchNorm+Dropout
class CNNWithBNDropout(nn.Module):
    def __init__(self, dropout_p=0.5):
        super(CNNWithBNDropout, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.bn1 = nn.BatchNorm2d(32)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.bn2 = nn.BatchNorm2d(64)
        self.pool = nn.MaxPool2d(2, 2)
        # Dropout层
        self.dropout = nn.Dropout(p=dropout_p)
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        x = self.pool(F.relu(self.bn1(self.conv1(x))))
        x = self.pool(F.relu(self.bn2(self.conv2(x))))
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        x = self.dropout(x)  # 训练时随机丢弃
        return self.fc2(x)

# 训练函数（记录Train-Test Gap）
def train_and_record(model, trainloader, testloader, name, num_epochs=10):
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    
    print(f"\n训练模型: {name}")
    for epoch in range(num_epochs):
        model.train()
        correct = total = 0
        for images, labels in trainloader:
            outputs = model(images)
            loss = criterion(outputs, labels)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        train_acc = 100 * correct / total
        
        model.eval()
        correct = total = 0
        with torch.no_grad():
            for images, labels in testloader:
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        test_acc = 100 * correct / total
        gap = train_acc - test_acc  # 过拟合程度
        print(f'Epoch [{epoch+1}/{num_epochs}] Train: {train_acc:.2f}% | Test: {test_acc:.2f}% | Gap: {gap:.2f}%')

if __name__ == '__main__':
    train_and_record(BasicCNN(), trainloader, testloader, "基础CNN", 10)
    train_and_record(CNNCWithBN(), trainloader, testloader, "+BatchNorm", 10)
    train_and_record(CNNWithBNDropout(0.5), trainloader, testloader, "+BN+Dropout", 10)
```

| 基础CNN | | | |
| +BatchNorm | | | |
| +BN+Dropout | | | |

## 实验5：数据增强 + 精度竞赛

### 📖 实验说明

**实验目标**：掌握数据增强方法，进行精度竞赛

**设计思路**：
1. 掌握三种数据增强策略：弱/中/强
2. 对比不同增强策略的效果
3. 学生自由设计增强方案冲击更高精度

### 💡 基础模块讲解

#### 1. transforms.RandomRotation（随机旋转）

```python
transforms.RandomRotation(degrees=(-10, 10))
```

- **含义**：随机旋转图片±10度
- **参数**：degrees=旋转角度范围
- **作用**：增加方向多样性

#### 2. transforms.RandomAffine（随机仿射变换）

```python
transforms.RandomAffine(degrees=0, translate=(0.1, 0.1), scale=(0.9, 1.1), shear=5)
```

- **参数**：
  - translate：平移比例（0.1表示移动10%）
  - scale：缩放范围
  - shear：剪切角度
- **作用**：模拟真实世界的变换

#### 3. transforms.GaussianBlur（高斯模糊）

```python
transforms.GaussianBlur(kernel_size=3, sigma=(0.1, 1.0))
```

- **含义**：对图片进行高斯模糊
- **作用**：模拟失焦、对抗模糊

### 💻 完整代码

```python
"""
实验5：数据增强 + 精度竞赛
==============================
三种增强策略对比，设计自己的方案冲击更高精度！
"""

# 三种增强策略
transform_weak = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))
])

transform_medium = transforms.Compose([
    transforms.RandomRotation(degrees=(-10, 10)),    # 随机旋转±10度
    transforms.RandomAffine(degrees=0, translate=(0.1, 0.1)),  # 平移10%
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))
])

transform_strong = transforms.Compose([
    transforms.RandomRotation(degrees=(-20, 20)),    # 强旋转±20度
    transforms.RandomAffine(degrees=0, translate=(0.15, 0.15), scale=(0.85, 1.15), shear=5),
    transforms.GaussianBlur(kernel_size=3, sigma=(0.1, 1.0)),  # 模糊
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))
])

# 完整CNN模型（3层卷积）
class FinalCNN(nn.Module):
    def __init__(self, dropout_p=0.5):
        super(FinalCNN, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, 3, padding=1)
        self.bn1 = nn.BatchNorm2d(32)
        self.conv2 = nn.Conv2d(32, 64, 3, padding=1)
        self.bn2 = nn.BatchNorm2d(64)
        self.conv3 = nn.Conv2d(64, 128, 3, padding=1)
        self.bn3 = nn.BatchNorm2d(128)
        self.pool = nn.MaxPool2d(2, 2)
        self.dropout = nn.Dropout(p=dropout_p)
        self.fc1 = nn.Linear(128 * 3 * 3, 256)
        self.fc2 = nn.Linear(256, 10)
    
    def forward(self, x):
        x = self.pool(F.relu(self.bn1(self.conv1(x))))  # 28→14
        x = self.pool(F.relu(self.bn2(self.conv2(x))))  # 14→7
        x = self.pool(F.relu(self.bn3(self.conv3(x))))  # 7→3
        x = x.view(x.size(0), -1)
        x = F.relu(self.fc1(x))
        x = self.dropout(x)
        return self.fc2(x)

# 训练函数
def train_model(model, trainloader, testloader, name, num_epochs=15):
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    scheduler = torch.optim.lr_scheduler.StepLR(optimizer, step_size=5, gamma=0.5)
    
    print(f"\n训练策略: {name}")
    best_acc = 0.0
    for epoch in range(num_epochs):
        model.train()
        correct = total = 0
        for images, labels in trainloader:
            outputs = model(images)
            loss = criterion(outputs, labels)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        scheduler.step()
        train_acc = 100 * correct / total
        
        model.eval()
        correct = total = 0
        with torch.no_grad():
            for images, labels in testloader:
                outputs = model(images)
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()
        test_acc = 100 * correct / total
        if test_acc > best_acc:
            best_acc = test_acc
        print(f'Epoch [{epoch+1}/{num_epochs}] Train: {train_acc:.2f}% | Test: {test_acc:.2f}%')
    return best_acc

if __name__ == '__main__':
    testloader = DataLoader(testset, batch_size=100, shuffle=False)
    results = {}
    
    trainset_a = torchvision.datasets.MNIST(root='./data', train=True, download=True, transform=transform_weak)
    results['弱增强'] = train_model(FinalCNN(0.3), DataLoader(trainset_a, batch_size=64, shuffle=True), testloader, "弱增强")
    
    trainset_b = torchvision.datasets.MNIST(root='./data', train=True, download=True, transform=transform_medium)
    results['中等增强'] = train_model(FinalCNN(0.4), DataLoader(trainset_b, batch_size=64, shuffle=True), testloader, "中等增强")
    
    trainset_c = torchvision.datasets.MNIST(root='./data', train=True, download=True, transform=transform_strong)
    results['强增强'] = train_model(FinalCNN(0.5), DataLoader(trainset_c, batch_size=64, shuffle=True), testloader, "强增强")
    
    print(f"\n🏆 最终精度对比:")
    for strategy, acc in sorted(results.items(), key=lambda x: x[1], reverse=True):
        status = "✅ 达标" if acc >= 95 else f"❌ 差{95-acc:.2f}%"
        print(f"  {strategy}: {acc:.2f}% {status}")
```
