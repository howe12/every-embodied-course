## 1. 项目概述与系统架构

### 1.1 项目目标

本项目旨在构建一个**完整的大模型具身智能系统**，让机器人能够理解人类的自然语言指令，通过多模态感知理解环境和任务，并自主完成复杂的操作任务。项目将整合以下核心技术：

- **大语言模型（LLM）**：理解自然语言指令，进行任务推理和规划
- **多模态感知**：视觉、语言、传感器数据的融合理解
- **Agent框架**：将复杂任务分解为可执行的子任务
- **机器人运动控制**：将决策转化为具体的物理动作

最终，机器人将能够：
- 接收"把桌子上的红色杯子拿给我"这样的自然语言指令
- 通过视觉定位目标物体
- 自主规划抓取策略和运动路径
- 执行抓取动作并反馈结果

## 2. 系统组件详解

### 2.1 视觉感知模块

视觉感知是具身智能的"眼睛"，负责从环境中提取关键信息。

#### 2.1.1 YOLO目标检测

YOLO（You Only Look Once）是一种单阶段目标检测算法，能够在单次前向传播中同时预测目标的类别和位置，非常适合机器人实时感知需求。

YOLO的核心优势：
- **速度快**：端到端检测，FPS可达数十甚至上百
- **精度高**：YOLOv8在COCO数据集上达到SOTA
- **易部署**：支持ONNX、TensorRT等加速格式

```python
# vision.py - YOLO视觉感知模块

import cv2
import numpy as np
from typing import List, Dict, Tuple, Optional
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import Image, CameraInfo
from vision_msgs.msg import Detection2DArray, BoundingBox2D
from geometry_msgs.msg import Pose2D
import torch

class YOLOVisionPerception(Node):
    """
    基于YOLOv8的视觉感知模块

    功能：
    1. 接收RGB图像流
    2. 运行YOLOv8目标检测
    3. 结合深度相机获取目标3D位置
    4. 发布检测结果
    """

    def __init__(self, model_name: str = "yolov8n.pt"):
        super().__init__('yolo_vision_perception')

        # 加载YOLOv8模型
        # 支持 yolov8n.pt (nano), yolov8s.pt (small),
        # yolov8m.pt (medium), yolov8l.pt (large), yolov8x.pt (xlarge)
        self.model = torch.hub.load('ultralytics/yolov8', model_name)
        self.model.eval()  # 设置为推理模式

        # 目标类别过滤（只关注机器人操作相关的物体）
        # COCO数据集部分类别索引
        self.relevant_classes = {
            0: 'person',      # 人
            41: 'cup',        # 杯子
            42: 'bottle',     # 瓶子
            43: 'bowl',       # 碗
            44: 'apple',      # 苹果
            45: 'orange',     # 橙子
            46: 'book',       # 书
            47: 'remote',     # 遥控器
            48: 'keyboard',   # 键盘
            49: 'laptop',     # 笔记本电脑
            63: 'chair',      # 椅子
        }

        # 订阅相机话题
        self.rgb_subscriber = self.create_subscription(
            Image,
            '/camera/rgb/image_raw',
            self.rgb_callback,
            10
        )
        self.depth_subscriber = self.create_subscription(
            Image,
            '/camera/depth/image_raw',
            self.depth_callback,
            10
        )

        # 发布检测结果
        self.detection_publisher = self.create_publisher(
            Detection2DArray,
            '/vision/detections',
            10
        )

        # 相机内参（需要标定获取）
        self.camera_intrinsics = {
            'fx': 615.0,  # x方向焦距
            'fy': 615.0,  # y方向焦距
            'cx': 320.0,  # 主点x
            'cy': 240.0,  # 主点y
        }

        # 最新的深度图像缓存
        self.latest_depth_image = None

        self.get_logger().info('YOLOVisionPerception 初始化完成')

    def rgb_callback(self, msg: Image):
        """处理RGB图像，进行目标检测"""
        # 将ROS Image消息转换为numpy数组
        np_image = self._image_msg_to_numpy(msg)

        # 运行YOLOv8推理
        results = self.model(np_image)

        # 解析检测结果
        detections = self._parse_yolo_results(results, msg.header.stamp)

        # 发布检测结果
        if detections:
            detection_array_msg = Detection2DArray()
            detection_array_msg.header = msg.header
            detection_array_msg.detections = detections
            self.detection_publisher.publish(detection_array_msg)

    def depth_callback(self, msg: Image):
        """缓存深度图像，供后续3D定位使用"""
        self.latest_depth_image = self._image_msg_to_numpy(msg)

    def _image_msg_to_numpy(self, msg: Image) -> np.ndarray:
        """将ROS Image消息转换为numpy数组"""
        # 根据编码格式解码图像
        if msg.encoding == 'rgb8':
            np_image = np.frombuffer(msg.data, dtype=np.uint8).reshape(
                msg.height, msg.width, 3
            )
            np_image = cv2.cvtColor(np_image, cv2.COLOR_RGB2BGR)
        elif msg.encoding == 'bgr8':
            np_image = np.frombuffer(msg.data, dtype=np.uint8).reshape(
                msg.height, msg.width, 3
            )
        elif msg.encoding == 'mono8':
            np_image = np.frombuffer(msg.data, dtype=np.uint8).reshape(
                msg.height, msg.width
            )
        elif msg.encoding == '16UC1':
            np_image = np.frombuffer(msg.data, dtype=np.uint16).reshape(
                msg.height, msg.width
            )
        else:
            self.get_logger().warn(f'不支持的图像编码格式: {msg.encoding}')
            np_image = np.zeros((msg.height, msg.width, 3), dtype=np.uint8)

        return np_image

    def _parse_yolo_results(self, results, stamp) -> List:
        """解析YOLOv8检测结果"""
        detections = []

        # 遍历所有检测框
        for result in results:
            boxes = result.boxes
            if boxes is None:
                continue

            for box in boxes:
                cls_id = int(box.cls.item())
                confidence = float(box.conf.item())

                # 只处理相关类别
                if cls_id not in self.relevant_classes:
                    continue

                # 过滤低置信度检测
                if confidence < 0.5:
                    continue

                # 获取边界框坐标 [x1, y1, x2, y2]
                x1, y1, x2, y2 = box.xyxy[0].tolist()

                # 计算边界框中心点
                cx = (x1 + x2) / 2
                cy = (y1 + y2) / 2

                # 获取3D位置（如果有深度图）
                x_3d, y_3d, z_3d = self._get_3d_position(cx, cy)

                # 创建检测消息
                detection = Detection2DArray()  # 简化示例，实际用vision_msgs
                detections.append({
                    'class_id': cls_id,
                    'class_name': self.relevant_classes[cls_id],
                    'confidence': confidence,
                    'bbox': {'x1': x1, 'y1': y1, 'x2': x2, 'y2': y2},
                    'center_2d': {'x': cx, 'y': cy},
                    'center_3d': {'x': x_3d, 'y': y_3d, 'z': z_3d},
                })

        return detections

    def _get_3d_position(self, px: float, py: float) -> Tuple[float, float, float]:
        """
        根据像素坐标和深度图计算3D位置

        参数:
            px, py: 像素坐标（2D图像中心点）
        返回:
            x, y, z: 相机坐标系下的3D坐标（单位：米）
        """
        if self.latest_depth_image is None:
            return 0.0, 0.0, 0.0

        h, w = self.latest_depth_image.shape

        # 确保像素坐标在图像范围内
        px = int(np.clip(px, 0, w - 1))
        py = int(np.clip(py, 0, h - 1))

        # 获取深度值（16UC1格式，单位为毫米，需转换为米）
        depth_mm = self.latest_depth_image[py, px]
        if isinstance(depth_mm, np.ndarray):
            depth_mm = depth_mm[0]
        depth_m = depth_mm / 1000.0

        # 针孔相机模型反投影
        x_3d = (px - self.camera_intrinsics['cx']) * depth_m / self.camera_intrinsics['fx']
        y_3d = (py - self.camera_intrinsics['cy']) * depth_m / self.camera_intrinsics['fy']
        z_3d = depth_m

        return x_3d, y_3d, z_3d
```

#### 2.1.2 深度相机点云处理

深度相机（如Intel RealSense、Azure Kinect）能够获取环境的稠密深度信息，通过点云处理可以更精确地感知环境结构。

```python
# point_cloud_processor.py - 深度点云处理

import numpy as np
import open3d as o3d
from typing import Tuple, List, Optional

class PointCloudProcessor:
    """
    深度点云处理器

    功能：
    1. 深度图转点云
    2. 点云降采样（体素滤波）
    3. 平面分割（桌面/地面检测）
    4. 物体点云提取
    """

    def __init__(self, fx: float = 615.0, fy: float = 615.0,
                 cx: float = 320.0, cy: float = 240.0):
        self.fx = fx
        self.fy = fy
        self.cx = cx
        self.cy = cy

    def depth_to_point_cloud(self, depth_image: np.ndarray) -> np.ndarray:
        """
        将深度图转换为点云

        公式（针孔相机模型）：
        X = (u - cx) * Z / fx
        Y = (v - cy) * Z / fy
        Z = depth

        参数:
            depth_image: 深度图像 (H x W)，单位：米
        返回:
            points: 点云 (N x 3)，N为有效点数
        """
        h, w = depth_image.shape

        # 生成像素坐标网格
        u_coords, v_coords = np.meshgrid(np.arange(w), np.arange(h))

        # 转换为3D坐标
        z = depth_image.flatten()
        x = ((u_coords.flatten() - self.cx) * z / self.fx)
        y = ((v_coords.flatten() - self.cy) * z / self.fy)

        # 过滤无效深度（深度为0表示无效）
        valid_mask = z > 0.01  # 最小深度阈值1cm
        points = np.stack([x, y, z], axis=-1)
        points = points[valid_mask]

        return points

    def voxel_downsample(self, points: np.ndarray, voxel_size: float = 0.005) -> np.ndarray:
        """
        体素滤波降采样

        将点云划分为体素网格，每个体素只保留一个代表点
        降低点云密度的同时保留空间结构
        """
        pcd = o3d.geometry.PointCloud()
        pcd.points = o3d.utility.Vector3dVector(points)
        pcd_down = pcd.voxel_down_sample(voxel_size)
        return np.asarray(pcd_down.points)

    def segment_plane(self, points: np.ndarray,
                      distance_threshold: float = 0.01,
                      ransac_n: int = 3,
                      num_iterations: int = 1000) -> Tuple[np.ndarray, np.ndarray]:
        """
        分割点云中的平面（使用RANSAC算法）

        适用于检测桌面、地面等平面，用于确定物体放置位置

        参数:
            distance_threshold: RANSAC距离阈值
            ransac_n: 随机采样点数
            num_iterations: RANSAC迭代次数
        返回:
            (plane_points, remaining_points): 平面点云和非平面点云
        """
        pcd = o3d.geometry.PointCloud()
        pcd.points = o3d.utility.Vector3dVector(points)

        # RANSAC平面分割
        plane_model, inliers = pcd.segment_plane(
            distance_threshold=distance_threshold,
            ransac_n=ransac_n,
            num_iterations=num_iterations
        )

        # 提取平面内点和剩余点云
        inlier_cloud = pcd.select_by_index(inliers)
        outlier_cloud = pcd.select_by_index(inliers, invert=True)

        return np.asarray(inlier_cloud.points), np.asarray(outlier_cloud.points)

    def extract_object_clusters(self, points: np.ndarray,
                                 eps: float = 0.02,
                                 min_points: int = 10) -> List[np.ndarray]:
        """
        从点云中提取物体簇（使用DBSCAN聚类）

        将空间上接近的点归为同一物体

        参数:
            eps: DBSCAN邻域半径
            min_points: 最小点数阈值
        返回:
            clusters: 物体点云列表
        """
        pcd = o3d.geometry.PointCloud()
        pcd.points = o3d.utility.Vector3dVector(points)

        # 使用Open3D的DBSCAN聚类
        labels = np.array(
            pcd.cluster_dbscan(eps=eps, min_points=min_points, print_progress=False)
        )

        # 按标签分组
        clusters = []
        for label_id in set(labels):
            if label_id == -1:  # -1表示噪声点
                continue
            mask = labels == label_id
            cluster_points = points[mask]
            clusters.append(cluster_points)

        return clusters

    def compute_bounding_box(self, cluster_points: np.ndarray) -> dict:
        """
        计算点云簇的轴对齐包围盒（AABB）

        用于获取物体的位置和尺寸信息
        """
        min_coords = np.min(cluster_points, axis=0)
        max_coords = np.max(cluster_points, axis=0)

        center = (min_coords + max_coords) / 2
        size = max_coords - min_coords
        volume = np.prod(size)

        return {
            'center': center,
            'min': min_coords,
            'max': max_coords,
            'size': size,
            'volume': volume,
            'length': size[0],
            'width': size[1],
            'height': size[2],
        }
```

### 2.2 自然语言理解模块

自然语言理解是连接用户指令和机器人行为的桥梁。系统需要理解用户想要什么，同时结合知识库（RAG）来补充领域知识。

#### 2.2.1 LLM指令理解

```python
# language_understanding.py - 自然语言理解模块

import json
from typing import Dict, List, Optional
from dataclasses import dataclass
from enum import Enum

class Intent(Enum):
    """机器人可理解的意图类型"""
    GRASP = "grasp"              # 抓取物体
    PLACE = "place"              # 放置物体
    MOVE = "move"                # 移动到位置
    NAVIGATE = "navigate"        # 导航
    PICK_AND_PLACE = "pick_and_place"  # 抓取并放置
    POUR = "pour"                # 倾倒
    PUSH = "push"                # 推动
    OPEN = "open"                # 打开
    CLOSE = "close"              # 关闭
    FIND = "find"                # 查找物体
    TELL = "tell"                # 告诉用户信息
    ASK = "ask"                  # 询问用户
    UNKNOWN = "unknown"           # 未知意图


@dataclass
class ParsedCommand:
    """解析后的指令结构"""
    raw_text: str                          # 原始文本
    intent: Intent                          # 意图类型
    target_object: Optional[str] = None     # 目标物体（如"红色杯子"）
    source_location: Optional[str] = None   # 源位置（如"桌子上"）
    destination: Optional[str] = None       # 目标位置（如"柜子里"）
    constraints: Optional[Dict] = None      # 约束条件（如"轻拿轻放"）
    reference_object: Optional[str] = None # 参考物体（如"放在书的左边"）
    confidence: float = 1.0                 # 解析置信度


class LanguageUnderstandingModule:
    """
    自然语言理解模块

    使用LLM解析用户的自然语言指令，
    提取关键信息：意图、目标物体、位置、约束等
    """

    def __init__(self, llm_client):
        self.llm = llm_client

        # 意图识别提示模板
        self.intent_prompt_template = """你是一个家庭服务机器人的指令解析器。
请将用户的自然语言指令解析为结构化的机器人动作指令。

支持的意图类型：
- grasp: 抓取指定物体
- place: 将物体放置到指定位置
- move: 将物体从一处移动到另一处
- navigate: 导航到指定位置
- find: 查找指定物体
- tell: 告诉用户某些信息

请根据用户指令判断意图，并提取相关信息。

用户指令：{user_input}

请以JSON格式输出：
{{
    "intent": "意图类型",
    "target_object": "目标物体（如果有）",
    "source_location": "源位置（如果有）",
    "destination": "目标位置（如果有）",
    "constraints": {{"约束名": "约束值", ...}},
    "confidence": 0.0-1.0置信度
}}
"""

    def parse_instruction(self, user_input: str) -> ParsedCommand:
        """
        解析用户指令

        参数:
            user_input: 用户的自然语言输入
        返回:
            ParsedCommand: 结构化的指令对象
        """
        prompt = self.intent_prompt_template.format(user_input=user_input)

        response = self.llm.generate(prompt)
        response_text = self.llm.extract_json(response)

        try:
            intent_str = response_text.get('intent', 'unknown')
            intent = self._str_to_intent(intent_str)

            parsed = ParsedCommand(
                raw_text=user_input,
                intent=intent,
                target_object=response_text.get('target_object'),
                source_location=response_text.get('source_location'),
                destination=response_text.get('destination'),
                constraints=response_text.get('constraints', {}),
                confidence=response_text.get('confidence', 0.5),
            )
            return parsed

        except Exception as e:
            # 解析失败时返回默认结构
            return ParsedCommand(
                raw_text=user_input,
                intent=Intent.UNKNOWN,
                confidence=0.0,
            )

    def _str_to_intent(self, intent_str: str) -> Intent:
        """将字符串转换为Intent枚举"""
        intent_mapping = {
            'grasp': Intent.GRASP,
            'place': Intent.PLACE,
            'move': Intent.MOVE,
            'navigate': Intent.NAVIGATE,
            'pick_and_place': Intent.PICK_AND_PLACE,
            'pour': Intent.POUR,
            'push': Intent.PUSH,
            'open': Intent.OPEN,
            'close': Intent.CLOSE,
            'find': Intent.FIND,
            'tell': Intent.TELL,
            'ask': Intent.ASK,
        }
        return intent_mapping.get(intent_str.lower(), Intent.UNKNOWN)


    def understand_with_context(self, user_input: str,
                                conversation_history: List[Dict],
                                environment_state: Dict) -> ParsedCommand:
        """
        带上下文的指令理解

        结合对话历史和环境状态来理解当前指令

        参数:
            user_input: 当前用户输入
            conversation_history: 对话历史 [{'role': 'user/assistant', 'content': '...'}]
            environment_state: 当前环境状态（如 {objects: ['杯子', '书'], location: '厨房'}）
        """
        context_prompt = f"""你是一个家庭服务机器人的指令解析器。

当前环境状态：
{json.dumps(environment_state, ensure_ascii=False, indent=2)}

对话历史：
{self._format_history(conversation_history)}

用户当前指令：{user_input}

请结合上下文理解用户指令，返回结构化的动作指令（JSON格式）：
"""

        response = self.llm.generate(context_prompt)
        response_text = self.llm.extract_json(response)

        intent = self._str_to_intent(response_text.get('intent', 'unknown'))
        return ParsedCommand(
            raw_text=user_input,
            intent=intent,
            target_object=response_text.get('target_object'),
            source_location=response_text.get('source_location'),
            destination=response_text.get('destination'),
            constraints=response_text.get('constraints', {}),
            confidence=response_text.get('confidence', 0.5),
        )

    def _format_history(self, history: List[Dict]) -> str:
        """格式化对话历史"""
        formatted = ""
        for turn in history[-5:]:  # 只保留最近5轮对话
            role = turn.get('role', 'unknown')
            content = turn.get('content', '')
            formatted += f"{role}: {content}\n"
        return formatted
```

```python
import numpy as np
from typing import List, Dict, Tuple, Optional
import faiss
import json

class EmbodiedRAGRetriever:
    """
    具身智能知识检索系统

    基于向量相似度的知识检索，
    为LLM提供任务执行所需的领域知识
    """

    def __init__(self, embedding_model, dimension: int = 768):
        self.embedding_model = embedding_model
        self.dimension = dimension

        # 构建FAISS索引（高效的向量相似度搜索）
        self.index = faiss.IndexFlatL2(dimension)
        self.knowledge_base = []  # 存储原始知识文本
        self.metadata = []       # 存储知识元数据

    def add_knowledge(self, text: str, metadata: Dict):
        """
        向知识库添加知识条目

        参数:
            text: 知识文本内容
            metadata: 元数据（如 {'type': 'grasp_strategy', 'object': 'cup'}）
        """
        embedding = self.embedding_model.encode(text)
        self.index.add(np.array([embedding]).astype('float32'))
        self.knowledge_base.append(text)
        self.metadata.append(metadata)

    def retrieve(self, query: str, top_k: int = 5) -> List[Dict]:
        """
        检索与查询最相关的知识

        参数:
            query: 查询文本（如 "如何抓取杯子"）
            top_k: 返回前k条最相关结果
        返回:
            results: [{'text': '...', 'metadata': {...}, 'score': 0.0}, ...]
        """
        query_embedding = self.embedding_model.encode(query)
        query_vector = np.array([query_embedding]).astype('float32')

        # FAISS最近邻搜索
        distances, indices = self.index.search(query_vector, top_k)

        results = []
        for dist, idx in zip(distances[0], indices[0]):
            if idx < len(self.knowledge_base):  # 有效索引
                results.append({
                    'text': self.knowledge_base[idx],
                    'metadata': self.metadata[idx],
                    'score': float(dist),
                })

        return results

    def build_knowledge_base(self):
```

### 2.3 任务规划Agent模块

Agent是大模型具身智能的核心，它将高层指令分解为可执行的低层动作序列，并能够根据执行反馈进行动态调整。

#### 2.3.1 基于ReAct的Task Planner

```python
# task_planner.py - 基于ReAct的任务规划Agent

import json
from typing import List, Dict, Optional, Callable
from dataclasses import dataclass, field
from enum import Enum
import re

class ActionType(Enum):
    """机器人可执行的动作类型"""
    # 感知类动作
    DETECT_OBJECT = "detect_object"       # 检测目标物体
    GET_SCENE_INFO = "get_scene_info"     # 获取场景信息
    CHECK_GRASP_SUCCESS = "check_grasp_success"  # 检查抓取是否成功

    # 导航类动作
    NAVIGATE_TO = "navigate_to"             # 导航到位置

    # 机械臂类动作
    MOVE_ARM_TO = "move_arm_to"             # 移动机械臂到位置
    MOVE_ARM_RELATIVE = "move_arm_relative" # 相对移动机械臂
    GRASP = "grasp"                         # 抓取
    RELEASE = "release"                      # 释放
    OPEN_GRIPPER = "open_gripper"            # 打开夹爪
    CLOSE_GRIPPER = "close_gripper"         # 关闭夹爪

    # 反馈类动作
    SAY = "say"                             # 语音反馈
    REPORT = "report"                       # 报告状态

    # 规划类动作
    THINK = "think"                         # 思考（不执行动作）
    PLAN = "plan"                           # 制定计划
    FINISH = "finish"                       # 任务完成


@dataclass
class Action:
    """动作结构"""
    action_type: ActionType
    parameters: Dict = field(default_factory=dict)
    reasoning: str = ""  # 动作执行的推理过程


@dataclass
class ExecutionResult:
    """动作执行结果"""
    success: bool
    observation: str    # 执行后的观察结果
    data: Optional[Dict] = None  # 额外返回数据（如检测到的物体位置）
    error: Optional[str] = None


class ReActTaskPlanner:
    """
    基于ReAct（Reasoning + Acting）范式的任务规划Agent

    ReAct核心思想：在执行动作的同时进行推理，
    通过"思考-行动-观察"的循环来完成任务

    格式：
    Thought: 当前推理...
    Action: 动作名称(参数)
    Observation: 执行结果
    ...

    最终：
    Thought: 现在我已经完成了任务
    Action: finish()
    """

    def __init__(self, llm_client, tools: Dict[str, Callable]):
        self.llm = llm_client
        self.tools = tools  # 可用工具字典 {tool_name: callable}
        self.max_iterations = 15  # 最大迭代次数，防止无限循环

    def plan(self, task: str, context: Dict = None) -> List[Action]:
        """
        制定任务执行计划

        参数:
            task: 任务描述
            context: 额外的上下文信息（如环境状态、知识库等）
        返回:
            action_sequence: 动作序列
        """
        context = context or {}

        # 构建ReAct提示
        prompt = self._build_react_prompt(task, context)

        # LLM生成推理过程
        llm_output = self.llm.generate(prompt)

        # 解析动作序列
        actions = self._parse_llm_output(llm_output)

        return actions

    def execute_plan(self, actions: List[Action]) -> List[ExecutionResult]:
        """
        执行动作序列

        参数:
            actions: 动作序列
        返回:
            results: 每个动作的执行结果列表
        """
        results = []

        for action in actions:
            # 调用对应的工具函数
            tool_name = action.action_type.value
            if tool_name in self.tools:
                try:
                    result = self.tools[tool_name](**action.parameters)
                    results.append(ExecutionResult(
                        success=True,
                        observation=str(result),
                        data=result if isinstance(result, dict) else None,
                    ))
                except Exception as e:
                    results.append(ExecutionResult(
                        success=False,
                        observation=f"执行失败: {str(e)}",
                        error=str(e),
                    ))
            else:
                results.append(ExecutionResult(
                    success=False,
                    observation=f"未知动作类型: {tool_name}",
                    error=f"Tool not found: {tool_name}",
                ))

        return results

    def plan_and_execute(self, task: str, context: Dict = None) -> Dict:
        """
        完整的计划+执行流程

        参数:
            task: 任务描述
            context: 上下文信息
        返回:
            包含动作序列、执行结果和最终状态的字典
        """
        # Step 1: 制定计划
        actions = self.plan(task, context)

        # Step 2: 执行计划
        results = self.execute_plan(actions)

        # Step 3: 汇总结果
        return {
            'task': task,
            'plan': [{'action': a.action_type.value, 'params': a.parameters, 'reasoning': a.reasoning} for a in actions],
            'results': results,
            'success': all(r.success for r in results),
            'final_observation': results[-1].observation if results else "",
        }

    def _build_react_prompt(self, task: str, context: Dict) -> str:
        """构建ReAct推理提示"""
        # 可用动作列表
        available_actions = "\n".join([
            f"  - {action.value}: {self._get_action_description(action)}"
            for action in ActionType
        ])

        # 上下文信息
        context_str = json.dumps(context, ensure_ascii=False, indent=2) if context else "{}"

        prompt = f"""你是一个智能服务机器人的任务规划Agent。

## 你的任务
{task}

## 可用工具（动作）
{available_actions}

## 当前环境上下文
{context_str}

## 输出格式
请按照以下格式输出推理过程和动作序列：

首先，进行推理思考（Thought），然后执行相应动作（Action），
观察执行结果（Observation），继续推理直到任务完成。

示例格式：
Thought: 我需要先检测桌面上有哪些物体，找到红色杯子
Action: detect_object(object_name="红色杯子")
Observation: 在图像坐标 (320, 240) 处检测到红色杯子，距离 0.45米

Thought: 检测到了杯子，现在需要将机械臂移动到杯子附近
Action: move_arm_to(position={{'x': 0.3, 'y': 0.0, 'z': 0.2}})
Observation: 机械臂已移动到目标位置

...（继续推理和执行）

Thought: 任务已完成，杯子已被成功放置到目标位置
Action: finish()
"""

        return prompt

    def _get_action_description(self, action_type: ActionType) -> str:
        """获取动作的描述"""
        descriptions = {
            ActionType.DETECT_OBJECT: "检测指定物体，返回物体位置",
            ActionType.GET_SCENE_INFO: "获取当前场景信息",
            ActionType.NAVIGATE_TO: "导航到指定位置",
            ActionType.MOVE_ARM_TO: "移动机械臂到绝对位置",
            ActionType.MOVE_ARM_RELATIVE: "相对移动机械臂",
            ActionType.GRASP: "抓取物体",
            ActionType.RELEASE: "释放物体",
            ActionType.OPEN_GRIPPER: "打开夹爪",
            ActionType.CLOSE_GRIPPER: "关闭夹爪",
            ActionType.SAY: "语音反馈",
            ActionType.REPORT: "报告状态",
            ActionType.THINK: "进行思考推理",
            ActionType.PLAN: "制定子计划",
            ActionType.FINISH: "任务完成",
            ActionType.CHECK_GRASP_SUCCESS: "检查抓取是否成功",
        }
        return descriptions.get(action_type, "未知动作")

    def _parse_llm_output(self, llm_output: str) -> List[Action]:
        """解析LLM输出，提取动作序列"""
        actions = []

        # 使用正则表达式匹配 Action 块
        action_pattern = r'Thought:\s*(.*?)\nAction:\s*(\w+)\s*\((.*?)\)\s*\n*Observation:?\s*(.*?)(?=\n(?:Thought:|Action:|finish\(\)|$))'

        matches = re.findall(action_pattern, llm_output, re.DOTALL)

        for match in matches:
            reasoning = match[0].strip()
            action_name = match[1].strip()
            params_str = match[2].strip()
            observation = match[3].strip()

            # 跳过 finish 动作（作为结束标记处理）
            if action_name.lower() == 'finish':
                break

            # 解析参数
            parameters = self._parse_parameters(params_str)

            # 转换为ActionType枚举
            try:
                action_type = ActionType(action_name)
            except ValueError:
                continue  # 跳过未知动作

            actions.append(Action(
                action_type=action_type,
                parameters=parameters,
                reasoning=reasoning,
            ))

        return actions

    def _parse_parameters(self, params_str: str) -> Dict:
        """解析参数字符串为字典"""
        if not params_str.strip():
            return {}

        params = {}

        # 处理 dict 格式参数 {"key": value, ...}
        if params_str.strip().startswith('{'):
            try:
                # 提取JSON部分
                json_str = params_str.strip()
                # 使用正则提取JSON
                json_match = re.search(r'\{.*\}', json_str, re.DOTALL)
                if json_match:
                    params = json.loads(json_match.group())
            except json.JSONDecodeError:
                pass

        # 处理 key=value, key=value 格式
        if not params:
            param_pattern = r'(\w+)\s*=\s*["\']?([^"\',\)]+)["\']?'
            matches = re.findall(param_pattern, params_str)
            for key, value in matches:
                # 尝试转换为数值
                try:
                    if '.' in value:
                        params[key] = float(value)
                    else:
                        params[key] = int(value)
                except ValueError:
                    params[key] = value.strip()

        return params
```

```python
    def compute_grasp_poses(self, object_position: Tuple[float, float, float],
                           object_size: Tuple[float, float, float],
                           grasp_direction: str = "top") -> List[Pose6D]:
        """
        计算抓取位姿

        根据物体位置和尺寸，计算合适的抓取位姿

        参数:
            object_position: 物体3D位置（米）
            object_size: 物体尺寸 (长, 宽, 高)（米）
            grasp_direction: 抓取方向 ("top", "side", "front")
        返回:
            grasp_poses: 可能的抓取位姿列表
        """
        x, y, z = object_position
        width, depth, height = object_size

        grasp_poses = []

        if grasp_direction == "top":
            # 从上方抓取（适合大多数桌面物体）
            # 抓取高度在物体上方，距离物体顶面约1cm
            grasp_z = z + height / 2 + 0.01

            # 计算抓取偏移（夹爪需要偏移半个物体宽度）
            offset_x = width / 2

            grasp_pose = Pose6D(
                position=(x, y, grasp_z),
                orientation=(0.0, 0.707, 0.0, 0.707),  # 默认朝下抓取
            )
            grasp_poses.append(grasp_pose)

        elif grasp_direction == "side":
            # 从侧面抓取
            grasp_pose = Pose6D(
                position=(x - width/2 - 0.02, y, z),
                orientation=(0.0, 0.383, 0.0, 0.924),  # 旋转90度
            )
            grasp_poses.append(grasp_pose)

        return grasp_poses

    def compute_pick_and_place_trajectory(self,
                                        pick_pose: Pose6D,
                                        place_pose: Pose6D,
                                        pre_grasp_height: float = 0.15) -> Dict:
        """
        计算抓取-放置完整轨迹

        包含：移动到预抓取位姿 → 抓取 → 抬起 → 移动到预放置位姿 → 放置

        参数:
            pick_pose: 抓取位姿
            place_pose: 放置位姿
            pre_grasp_height: 预抓取高度（安全抬起高度）
        返回:
            trajectory: 包含各阶段路径点的字典
        """
        # 预抓取位姿（抓取点上方）
        pre_pick = Pose6D(
            position=(pick_pose.position[0],
                     pick_pose.position[1],
                     pick_pose.position[2] + pre_grasp_height),
            orientation=pick_pose.orientation,
        )

        # 移动后预抓取位姿（提起后）
        lift_pose = Pose6D(
            position=(pick_pose.position[0],
                     pick_pose.position[1],
                     pick_pose.position[2] + pre_grasp_height + 0.1),
            orientation=pick_pose.orientation,
        )

        # 预放置位姿（放置点上方）
        pre_place = Pose6D(
            position=(place_pose.position[0],
                     place_pose.position[1],
                     place_pose.position[2] + pre_grasp_height),
            orientation=place_pose.orientation,
        )

        trajectory = {
            'pre_pick': pre_pick,       # 移动到预抓取
            'pick': pick_pose,          # 下降到抓取点
            'lift': lift_pose,          # 提起
            'transport': [lift_pose, pre_place],  # 移动路径
            'pre_place': pre_place,     # 移动到预放置
            'place': place_pose,        # 下降到放置点
            'retreat': pre_place,       # 退回安全距离
        }

        return trajectory
```

```python
        self.vision = YOLOVisionPerception()

        # 6. 运动规划模块
        self.motion_planner = MotionPlanner(group_name="panda_arm")

        # 7. 机械臂控制器
        self.arm = ArmController()
        self.arm.initialize()

        # 8. 夹爪控制器
        self.gripper = GripperController()
        self.gripper.initialize()

        # 9. 安全监控器
        self.safety_monitor = SafetyMonitor(self.arm, self.gripper)
        self.safety_monitor.start_monitoring()

        # 10. ROS2桥接
        self.ros2_sub = ROS2Subscriber()
        self.ros2_pub = ROS2Publisher()

        # ========== Agent工具函数注册 ==========
        self.tools = self._register_tools()

        # 11. 任务规划Agent
        self.task_planner = ReActTaskPlanner(self.llm_client, self.tools)

        # ========== 状态变量 ==========
        self.current_task: Optional[str] = None
        self.is_executing = False
        self.current_object_position: Optional[Dict] = None

        self.get_logger().info('系统初始化完成！')
        self.hri.speak("系统就绪，请告诉我您需要什么帮助。")

    def _register_tools(self) -> Dict:
        """注册Agent可调用的工具函数"""
        return {
            'detect_object': self.tool_detect_object,
            'get_scene_info': self.tool_get_scene_info,
            'navigate_to': self.tool_navigate_to,
            'move_arm_to': self.tool_move_arm_to,
            'move_arm_relative': self.tool_move_arm_relative,
            'grasp': self.tool_grasp,
            'release': self.tool_release,
            'open_gripper': self.tool_open_gripper,
            'close_gripper': self.tool_close_gripper,
            'check_grasp_success': self.tool_check_grasp_success,
            'say': self.tool_say,
            'report': self.tool_report,
        }

    # ========== 工具函数实现 ==========

    def tool_detect_object(self, object_name: str, **kwargs) -> Dict:
        """检测指定物体"""
        self.get_logger().info(f'检测物体: {object_name}')

        # 从视觉模块获取最新检测结果
        # 实际使用中，需要订阅ROS话题获取实时检测结果
        detections = self.vision.get_latest_detections()

        # 查找匹配物体
        for det in detections:
            if object_name in det.get('class_name', ''):
                self.current_object_position = det['center_3d']
                return {
                    'found': True,
                    'object': object_name,
                    'position_3d': det['center_3d'],
                    'bbox': det['bbox'],
                    'confidence': det['confidence'],
                }

        return {'found': False, 'object': object_name}

    def tool_get_scene_info(self, **kwargs) -> Dict:
        """获取当前场景信息"""
        detections = self.vision.get_latest_detections()
        objects = [d['class_name'] for d in detections]
        return {
            'objects_detected': objects,
            'object_count': len(objects),
            'timestamp': time.time(),
        }

    def tool_navigate_to(self, position: Dict, **kwargs) -> str:
        """导航到指定位置"""
        x = position.get('x', 0.0)
        y = position.get('y', 0.0)
        z = position.get('z', 0.0)

        self.get_logger().info(f'导航到位置: ({x}, {y}, {z})')
        # 实际使用Nav2进行导航
        # nav_result = self.navigation_client.go_to(x, y, theta)
        return f'已导航到位置 ({x:.2f}, {y:.2f})'

    def tool_move_arm_to(self, position: Dict, **kwargs) -> str:
        """移动机械臂到绝对位置"""
        x = position.get('x', 0.0)
        y = position.get('y', 0.0)
        z = position.get('z', 0.0)

        self.get_logger().info(f'移动机械臂到: ({x}, {y}, {z})')

        # 安全检查
        if not self.safety_monitor.check_position_safe(x, y, z):
            return '错误：目标位置不安全'

        # 调用机械臂控制
        success = self.arm.move_to_pose((x, y, z, 0, 0, 0, 1))
        return '机械臂移动完成' if success else '机械臂移动失败'

    def tool_move_arm_relative(self, delta: Dict, **kwargs) -> str:
        """相对移动机械臂"""
        dx = delta.get('dx', 0.0)
        dy = delta.get('dy', 0.0)
        dz = delta.get('dz', 0.0)

        current = self.arm.get_current_pose()
        if current is None:
            return '错误：无法获取当前位姿'

        new_x = current[0] + dx
        new_y = current[1] + dy
        new_z = current[2] + dz

        return self.tool_move_arm_to({'x': new_x, 'y': new_y, 'z': new_z})

    def tool_grasp(self, force: float = 10.0, **kwargs) -> str:
        """执行抓取"""
        self.get_logger().info(f'执行抓取，力度: {force}N')
        success = self.arm.grasp(force=force)
        return '抓取成功' if success else '抓取失败'

    def tool_release(self, width: float = None, **kwargs) -> str:
        """释放物体"""
        self.get_logger().info('释放夹爪')
        success = self.arm.release(width=width)
        return '释放完成' if success else '释放失败'

    def tool_open_gripper(self, **kwargs) -> str:
        """打开夹爪"""
        self.gripper.open()
        return '夹爪已打开'

    def tool_close_gripper(self, **kwargs) -> str:
        """关闭夹爪"""
        self.gripper.close()
        return '夹爪已关闭'

    def tool_check_grasp_success(self, **kwargs) -> Dict:
        """检查抓取是否成功"""
        force = self.gripper.get_current_force()
        position = self.gripper.get_current_position()

        success = (force > 2.0) and (position < self.gripper.max_position * 0.9)
        return {
            'success': success,
            'force': force,
            'gripper_position': position,
        }

    def tool_say(self, text: str, **kwargs) -> str:
        """语音反馈"""
        self.hri.speak(text)
        return f'已说: {text}'

    def tool_report(self, status: str, **kwargs) -> str:
        """报告状态"""
        self.get_logger().info(f'状态报告: {status}')
        self.ros2_pub.publish_status(status)
        return f'已报告: {status}'

    # ========== 主控制流程 ==========

    def on_user_command(self, command: str):
        """处理用户指令"""
        if self.is_executing:
            self.hri.speak("我正在处理上一个任务，请稍候。")
            return

        self.current_task = command
        self.is_executing = True

        # 在后台线程执行任务
        thread = threading.Thread(target=self.execute_task, args=(command,), daemon=True)
        thread.start()

    def execute_task(self, task: str):
        """执行任务的完整流程"""
        try:
            self.get_logger().info(f'开始执行任务: {task}')

            # Step 1: 解析用户指令
            self.hri.speak("我正在理解您的指令...")
            parsed_command = self.language_understanding.parse_instruction(task)
            self.get_logger().info(f'解析结果: intent={parsed_command.intent}')

            # Step 2: 构建上下文（环境状态 + 知识库）
            scene_info = self.tool_get_scene_info()
            rag_knowledge = self.rag.retrieve_for_task(
                task=parsed_command.intent.value,
                object_type=parsed_command.target_object
            )

            context = {
                'scene': scene_info,
                'knowledge': rag_knowledge,
                'parsed_command': {
                    'intent': parsed_command.intent.value,
                    'target': parsed_command.target_object,
                    'source': parsed_command.source_location,
                    'destination': parsed_command.destination,
                }
            }

            # Step 3: 任务规划和执行
            self.hri.speak("好的，我开始执行任务。")
            result = self.task_planner.plan_and_execute(task, context)

            # Step 4: 处理执行结果
            if result['success']:
                self.hri.speak("任务执行完成！还有其他需要帮忙的吗？")
            else:
                self.hri.speak("任务执行过程中遇到了一些问题，请查看详情。")

            self.get_logger().info(f"任务完成: {result}")

        except Exception as e:
            self.get_logger().error(f'任务执行出错: {str(e)}')
            self.hri.speak("抱歉，执行过程中出现了错误。")
            self.arm.stop()  # 紧急停止

        finally:
            self.is_executing = False

    def shutdown(self):
        """系统关闭"""
        self.get_logger().info('正在关闭系统...')
        self.is_executing = False
        self.safety_monitor.stop_monitoring()
        self.arm.go_home()
        self.hri.speak("系统已关闭，再见！")
```

## 5. 代码实战：LLM任务规划Agent

### 5.1 完整Agent实现

```python
# agent_complete.py - 完整的ReAct Agent实现

"""
大模型任务规划Agent完整实现

功能：
1. 接收高层自然语言指令
2. 使用ReAct范式进行推理和规划
3. 调用各种工具执行子任务
4. 根据执行反馈动态调整计划
"""

import json
import re
import time
from typing import Dict, List, Optional, Any
from dataclasses import dataclass, field
from enum import Enum


class TaskState(Enum):
    """任务执行状态"""
    IDLE = "idle"
    PLANNING = "planning"
    EXECUTING = "executing"
    WAITING = "waiting"  # 等待用户确认
    SUCCEEDED = "succeeded"
    FAILED = "failed"
    CANCELLED = "cancelled"


@dataclass
class Step:
    """执行步骤"""
    step_id: int
    thought: str
    action_name: str
    action_params: Dict[str, Any]
    observation: str = ""
    result: Any = None
    success: bool = True
    timestamp: float = field(default_factory=time.time)


@dataclass
class ExecutionContext:
    """执行上下文"""
    user_command: str
    parsed_intent: str
    target_object: Optional[str] = None
    source_location: Optional[str] = None
    destination: Optional[str] = None
    constraints: Dict[str, Any] = field(default_factory=dict)
    scene_objects: List[str] = field(default_factory=list)
    retrieved_knowledge: str = ""
    execution_history: List[Step] = field(default_factory=list)


class EmbodiedAgent:
    """
    具身智能任务规划Agent

    核心思想：Thought → Action → Observation 循环
    """

    # 工具函数注册表
    TOOL_REGISTRY = {
        # 感知类
        'detect_object': {
            'description': '检测指定物体，返回位置信息',
            'params': {'object_name': 'str'},
        },
        'get_scene_info': {
            'description': '获取当前场景中所有物体',
            'params': {},
        },
        'get_depth_at': {
            'description': '获取指定像素位置的深度',
            'params': {'u': 'int', 'v': 'int'},
        },

        # 导航类
        'navigate_to': {
            'description': '移动机器人到指定位置',
            'params': {'x': 'float', 'y': 'float', 'theta': 'float'},
        },
        'navigate_to_object': {
            'description': '导航到指定物体附近',
            'params': {'object_name': 'str', 'distance': 'float'},
        },

        # 机械臂类
        'move_arm_to': {
            'description': '移动机械臂末端到指定位置',
            'params': {'x': 'float', 'y': 'float', 'z': 'float'},
        },
        'move_arm_to_pose': {
            'description': '移动机械臂到指定位姿',
            'params': {'position': 'dict', 'orientation': 'dict'},
        },
        'move_arm_relative': {
            'description': '相对移动机械臂',
            'params': {'dx': 'float', 'dy': 'float', 'dz': 'float'},
        },

        # 抓取类
        'plan_grasp': {
            'description': '计算抓取位姿',
            'params': {'object_name': 'str'},
        },
        'execute_grasp': {
            'description': '执行抓取动作',
            'params': {'force': 'float'},
        },
        'check_grasp': {
            'description': '检查抓取是否成功',
            'params': {},
        },
        'release': {
            'description': '释放物体',
            'params': {'width': 'float'},
        },

        # 夹爪类
        'open_gripper': {
            'description': '打开夹爪',
            'params': {},
        },
        'close_gripper': {
            'description': '关闭夹爪',
            'params': {},
        },

        # 交互类
        'say': {
            'description': '语音播报',
            'params': {'text': 'str'},
        },
        'ask_user': {
            'description': '向用户询问',
            'params': {'question': 'str'},
        },
        'confirm': {
            'description': '请求用户确认',
            'params': {'action': 'str'},
        },

        # 状态类
        'report': {
            'description': '报告当前状态',
            'params': {'status': 'str'},
        },
        'wait': {
            'description': '等待指定时间',
            'params': {'seconds': 'float'},
        },

        # 结束类
        'finish': {
            'description': '任务完成',
            'params': {},
        },
    }

    def __init__(self, llm_client, tool_executors: Dict[str, callable]):
        self.llm = llm_client
        self.tool_executors = tool_executors  # {tool_name: callable}

        self.max_iterations = 20
        self.current_context: Optional[ExecutionContext] = None
        self.state = TaskState.IDLE

    def execute(self, command: str, context: Dict = None) -> Dict:
        """
        执行任务的主入口

        参数:
            command: 用户指令
            context: 额外上下文（场景信息、知识库等）
        返回:
            执行结果字典
        """
        self.state = TaskState.PLANNING
        context = context or {}

        # 构建初始上下文
        self.current_context = ExecutionContext(
            user_command=command,
            parsed_intent=context.get('intent', 'unknown'),
            target_object=context.get('target_object'),
            destination=context.get('destination'),
            scene_objects=context.get('scene_objects', []),
            retrieved_knowledge=context.get('knowledge', ''),
        )

        self.state = TaskState.EXECUTING

        # ReAct循环
        step_id = 0
        while step_id < self.max_iterations:
            step_id += 1

            # 构建ReAct提示
            prompt = self._build_react_prompt()

            # LLM生成下一步
            llm_output = self.llm.generate(prompt)

            # 解析动作
            action_name, action_params, thought = self._parse_action(llm_output)

            # 检查是否是结束动作
            if action_name == 'finish':
                self.state = TaskState.SUCCEEDED
                break

            # 执行动作
            observation, result, success = self._execute_action(action_name, action_params)

            # 记录步骤
            step = Step(
                step_id=step_id,
                thought=thought,
                action_name=action_name,
                action_params=action_params,
                observation=observation,
                result=result,
                success=success,
            )
            self.current_context.execution_history.append(step)

            # 检查执行结果
            if not success:
                self.state = TaskState.FAILED
                break

        # 构建返回结果
        return self._build_result()

    def _build_react_prompt(self) -> str:
        """构建ReAct推理提示"""
        # 可用工具说明
        tools_desc = []
        for name, info in self.TOOL_REGISTRY.items():
            params = ", ".join([f"{k}: {v}" for k, v in info['params'].items()])
            tools_desc.append(f"- {name}({params}): {info['description']}")
        tools_text = "\n".join(tools_desc)

        # 执行历史
        history_text = ""
        for step in self.current_context.execution_history[-5:]:  # 最近5步
            history_text += f"\nStep {step.step_id}:\n"
            history_text += f"  Thought: {step.thought}\n"
            history_text += f"  Action: {step.action_name}({step.action_params})\n"
            history_text += f"  Observation: {step.observation}\n"

        prompt = f"""你是智能服务机器人的任务规划Agent，负责将高层指令分解为可执行的动作序列。

## 当前任务

## 输出格式

Action: <动作名>(<参数>)
"""

        return prompt

    def _parse_action(self, llm_output: str) -> tuple:
        """从LLM输出中解析动作"""
        # 提取Thought
        thought_match = re.search(r'Thought:\s*(.*?)(?:\n|$)', llm_output, re.DOTALL)
        thought = thought_match.group(1).strip() if thought_match else ""

        # 提取Action
        action_match = re.search(
            r'Action:\s*(\w+)\s*\((.*?)\)\s*(?:\n|$)',
            llm_output,
            re.DOTALL
        )

        if not action_match:
            return 'report', {'status': '无法解析动作'}, thought

        action_name = action_match.group(1).strip()
        params_str = action_match.group(2).strip()

        # 解析参数
        params = self._parse_params(params_str)

        return action_name, params, thought

    def _parse_params(self, params_str: str) -> Dict[str, Any]:
        """解析参数字符串"""
        if not params_str.strip():
            return {}

        # 尝试JSON格式
        try:
            # 处理可能的嵌套字典
            if '{' in params_str and '}' in params_str:
                # 找到最外层的大括号
                start = params_str.find('{')
                end = params_str.rfind('}') + 1
                json_str = params_str[start:end]
                return json.loads(json_str)
        except json.JSONDecodeError:
            pass

        # 处理 key=value 格式
        params = {}
        pattern = r'(\w+)\s*=\s*(?:(["\'])(.*?)\2|(\d+\.?\d*)|(\w+))'
        matches = re.findall(pattern, params_str)

        for match in matches:
            key = match[0]
            # 根据捕获组判断类型
            if match[2]:  # 带引号的字符串
                params[key] = match[2]
            elif match[3]:  # 数字
                try:
                    params[key] = int(match[3])
                except ValueError:
                    params[key] = float(match[3])
            elif match[4]:  # 普通字符串
                params[key] = match[4]

        return params

    def _execute_action(self, action_name: str, params: Dict) -> tuple:
        """执行动作"""
        # 检查工具是否存在
        if action_name not in self.tool_executors:
            return f"未知工具: {action_name}", None, False


        try:
            executor = self.tool_executors[action_name]
            result = executor(**params)

            # 格式化观察结果
            if isinstance(result, dict):
                observation = json.dumps(result, ensure_ascii=False)
            else:
                observation = str(result)

            return observation, result, True

        except Exception as e:
            return f"执行错误: {str(e)}", None, False

    def _build_result(self) -> Dict:
        """构建执行结果"""
        success = self.state == TaskState.SUCCEEDED

        return {
            'success': success,
            'state': self.state.value,
            'task': self.current_context.user_command,
            'total_steps': len(self.current_context.execution_history),
            'steps': [
                {
                    'id': s.step_id,
                    'thought': s.thought,
                    'action': s.action_name,
                    'params': s.action_params,
                    'observation': s.observation,
                    'success': s.success,
                }
                for s in self.current_context.execution_history
            ],
        }
```

```python
# vision_language_alignment.py - 视觉-语言对齐模块

"""
视觉-语言对齐（Vision-Language Alignment）

将视觉感知结果与语言描述对齐，
实现"在图像中找到用户描述的物体"等功能
"""

import torch
import torch.nn as nn
import numpy as np
from typing import List, Dict, Tuple, Optional
from dataclasses import dataclass


@dataclass
class GroundingResult:
    """视觉语言对齐结果"""
    bbox: Tuple[int, int, int, int]  # x1, y1, x2, y2
    confidence: float
    phrase: str  # 对应的语言描述
    image_features: np.ndarray  # 图像区域特征
    text_features: np.ndarray   # 文本特征


class VisionLanguageAligner:
    """
    视觉-语言对齐器

    使用CLIP风格的模型实现图像区域和文本描述的匹配
    """

    def __init__(self, model_name: str = "openai/clip-vit-base-patch32"):
        # 实际使用CLIP模型
        # self.model, self.preprocess = clip.load(model_name, device="cuda")

        self.feature_dim = 512

    def encode_image(self, image: np.ndarray) -> np.ndarray:
        """
        编码图像为特征向量

        参数:
            image: H x W x 3 的RGB图像数组
        返回:

        """
        # 实际使用CLIP:
        # from PIL import Image

        # preprocessed = self.preprocess(pil_image).unsqueeze(0).to(self.device)
        # with torch.no_grad():
        #     features = self.model.encode_image(preprocessed)
        # return features.cpu().numpy().flatten()

        # 模拟：返回随机特征
        return np.random.randn(self.feature_dim)

    def encode_text(self, text: str) -> np.ndarray:
        """
        编码文本为特征向量

        参数:
            text: 文本描述
        返回:
            文本特征向量 (feature_dim,)
        """
        # 实际使用CLIP:

        # return features.cpu().numpy().flatten()

        # 模拟：返回随机特征
        return np.random.randn(self.feature_dim)

    def compute_similarity(self, image_features: np.ndarray,
                         text_features: np.ndarray) -> float:
        """计算图像和文本特征的余弦相似度"""
        dot_product = np.dot(image_features, text_features)
        norm_product = np.linalg.norm(image_features) * np.linalg.norm(text_features)
        return dot_product / (norm_product + 1e-8)

    def ground_phrase_in_image(self, image: np.ndarray,
                               phrases: List[str]) -> List[GroundingResult]:
        """
        在图像中定位文本描述的物体

        使用滑动窗口+CLIP特征匹配的方法

        参数:
            image: RGB图像数组
            phrases: 要匹配的短语列表，如 ["红色杯子", "蓝色的书"]
        返回:
            每个短语的匹配结果
        """
        H, W = image.shape[:2]

        # 编码所有文本短语
        text_features_list = [self.encode_text(p) for p in phrases]

        results = []
        step_size = 32  # 滑动窗口步长
        window_sizes = [128, 192, 256]  # 多尺度窗口

        for phrase, text_feat in zip(phrases, text_features_list):
            best_similarity = -1.0
            best_bbox = (0, 0, 100, 100)

            # 多尺度滑动窗口
            for w_size in window_sizes:
                for y in range(0, H - w_size, step_size):
                    for x in range(0, W - w_size, step_size):
                        # 提取窗口图像
                        window = image[y:y+w_size, x:x+w_size]

                        # 编码窗口图像
                        img_feat = self.encode_image(window)

                        # 计算相似度
                        similarity = self.compute_similarity(img_feat, text_feat)

                        if similarity > best_similarity:
                            best_similarity = similarity
                            best_bbox = (x, y, x + w_size, y + w_size)

            results.append(GroundingResult(
                bbox=best_bbox,
                confidence=best_similarity,
                phrase=phrase,
                image_features=self.encode_image(image),
                text_features=text_feat,
            ))

        return results


class ReferringExpressionComprehension:
    """
    指代表达理解（Referring Expression Comprehension）

    理解"那个红色的杯子"这样的指代表达，
    在场景中找到被指代的物体
    """

    def __init__(self, aligner: VisionLanguageAligner):
        self.aligner = aligner

    def comprehend(self, image: np.ndarray,
                   referring_expression: str,
                   candidate_objects: List[Dict]) -> Optional[Dict]:
        """
        理解指代表达

        参数:
            image: 图像数组
            referring_expression: 指代表达，如 "那个红色的杯子"
            candidate_objects: 候选物体列表，每个包含 {'bbox', 'label', 'features'}
        返回:
            匹配的物体，或None
        """
        # Step 1: 解析指代表达，提取关键属性
        attributes = self._parse_attributes(referring_expression)
        # attributes = {'color': '红色', 'type': '杯子'}

        # Step 2: 过滤候选物体
        filtered = []
        for obj in candidate_objects:
            match = True
            for attr, value in attributes.items():
                if not self._match_attribute(obj, attr, value):
                    match = False
                    break
            if match:
                filtered.append(obj)

        if not filtered:
            return None

        # Step 3: 在过滤后的候选中，选择与指代表达最相似的
        text_feat = self.aligner.encode_text(referring_expression)
        best_obj = None
        best_similarity = -1.0

        for obj in filtered:
            obj_feat = obj.get('features')
            if obj_feat is None:
                continue

            similarity = self.aligner.compute_similarity(obj_feat, text_feat)
            if similarity > best_similarity:
                best_similarity = similarity
                best_obj = obj

        if best_obj:
            best_obj['similarity'] = best_similarity
            best_obj['confidence'] = best_similarity

        return best_obj

    def _parse_attributes(self, expression: str) -> Dict[str, str]:
        """从指代表达中解析属性"""
        attributes = {}

        # 颜色词
        colors = ['红色', '蓝色', '绿色', '黄色', '白色', '黑色', '粉色', '紫色']
        for color in colors:
            if color in expression:
                attributes['color'] = color
                break

        # 尺寸词
        if '大的' in expression or '大的那个' in expression:
            attributes['size'] = 'large'
        elif '小的' in expression or '小的那个' in expression:
            attributes['size'] = 'small'

        # 物体类型（提取名词短语）
        import re
        objects = ['杯子', '瓶子', '书', '苹果', '橙子', '碗', '盒子', '瓶子']
        for obj in objects:
            if obj in expression:
                attributes['type'] = obj
                break

        return attributes

    def _match_attribute(self, obj: Dict, attr: str, value: str) -> bool:
        """匹配物体属性"""
        obj_label = obj.get('label', '').lower()
        obj_color = obj.get('color', '')

        if attr == 'color':
            return value.replace('色', '') in obj_color
        elif attr == 'type':
            return value in obj_label
        elif attr == 'size':
            obj_size = obj.get('size', 'medium')
            return obj_size == value

        return True
```

### 6.2 视觉-语言-动作闭环

```python
# vision_language_action_loop.py - 视觉-语言-动作闭环

"""
视觉-语言-动作闭环（Vision-Language-Action Loop）

将视觉感知、语言理解和动作执行整合为闭环控制系统
"""

import numpy as np
from typing import Dict, List, Optional, Tuple
from dataclasses import dataclass
from enum import Enum


class ActionPhase(Enum):
    """动作执行阶段"""
    IDLE = "idle"
    APPROACH = "approach"      # 接近目标
    ALIGN = "align"            # 对齐
    DESCEND = "descend"        # 下降
    GRASP = "grasp"            # 抓取
    LIFT = "lift"              # 抬起
    RETRACT = "retract"        # 撤回
    TRANSPORT = "transport"    # 搬运
    RELEASE = "release"       # 释放
    COMPLETE = "complete"      # 完成


@dataclass
class VLAControlState:
    """视觉-语言-动作控制状态"""
    phase: ActionPhase
    target_position: Tuple[float, float, float]  # 目标3D位置
    current_position: Tuple[float, float, float]   # 当前位置
    error: float                                    # 位置误差
    visual_tracking_active: bool                   # 是否在视觉跟踪
    last_visual_update: float                      # 上次视觉更新时间
    grasp_success: bool = False                     # 抓取是否成功


class VLAControlLoop:
    """
    视觉-语言-动作闭环控制器

    核心思想：在动作执行过程中持续利用视觉反馈进行误差校正
    """

    def __init__(self, vision_module, arm_controller, gripper_controller):
        self.vision = vision_module
        self.arm = arm_controller
        self.gripper = gripper_controller

        # 控制参数
        self.position_threshold = 0.01  # 位置误差阈值 (m)
        self.visual_update_rate = 10.0   # 视觉更新频率 (Hz)
        self.max_descend_speed = 0.05    # 最大下降速度 (m/s)

        # 状态
        self.state = VLAControlState(
            phase=ActionPhase.IDLE,
            target_position=(0.0, 0.0, 0.0),
            current_position=(0.0, 0.0, 0.0),
            error=float('inf'),
            visual_tracking_active=False,
            last_visual_update=0.0,
        )

    def execute_grasp_sequence(self, object_position: Tuple[float, float, float],
                              grasp_height: float = 0.15) -> bool:
        """
        执行完整的抓取序列

        包含：接近 → 对齐 → 下降 → 抓取 → 抬起 → 撤回

        参数:
            object_position: 物体3D位置 (x, y, z)
            grasp_height: 预抓取高度（物体上方安全距离）
        返回:
            success: 抓取是否成功
        """
        x, y, z = object_position

        # 更新目标位置
        self.state.target_position = object_position

        print(f"[VLA] 开始抓取序列，目标位置: {object_position}")

        # ===== Phase 1: 接近目标 =====
        self.state.phase = ActionPhase.APPROACH
        approach_pos = (x, y, z + grasp_height + 0.1)
        if not self._move_to_position(approach_pos):
            return False

        # ===== Phase 2: 对齐（视觉校正） =====
        self.state.phase = ActionPhase.ALIGN
        corrected_pos = self._visual_alignment(approach_pos)
        if not self._move_to_position(corrected_pos):
            return False

        # ===== Phase 3: 下降到抓取高度 =====
        self.state.phase = ActionPhase.DESCEND
        descent_pos = (x, y, z + grasp_height)
        if not self._controlled_descent(object_position, grasp_height):
            return False

        # ===== Phase 4: 执行抓取 =====
        self.state.phase = ActionPhase.GRASP
        self.gripper.close()
        import time
        time.sleep(0.5)  # 等待夹爪关闭

        # 检查抓取是否成功
        if not self._check_grasp_success():
            print("[VLA] 抓取失败，尝试重新抓取...")
            self.gripper.open()
            # 重新对齐并抓取
            self.state.phase = ActionPhase.ALIGN
            corrected_pos = self._visual_alignment(descent_pos)
            self._move_to_position(corrected_pos)
            self.gripper.close()
            time.sleep(0.5)
            if not self._check_grasp_success():
                self.state.phase = ActionPhase.IDLE
                return False

        self.state.grasp_success = True
        print("[VLA] 抓取成功！")

        # ===== Phase 5: 抬起 =====
        self.state.phase = ActionPhase.LIFT
        lift_pos = (x, y, z + grasp_height + 0.15)
        self._move_to_position(lift_pos)

        # ===== Phase 6: 撤回 =====
        self.state.phase = ActionPhase.RETRACT
        retract_pos = (x, y, z + grasp_height + 0.3)
        self._move_to_position(retract_pos)

        self.state.phase = ActionPhase.COMPLETE
        print("[VLA] 抓取序列完成")
        return True

    def _move_to_position(self, position: Tuple[float, float, float]) -> bool:
        """移动机械臂到指定位置"""
        x, y, z = position
        pose = (x, y, z, 0.0, 0.0, 0.0, 1.0)  # 默认姿态
        success = self.arm.move_to_pose(pose)
        self.state.current_position = position
        return success

    def _visual_alignment(self, current_pos: Tuple) -> Tuple:
        """
        视觉对齐校正

        利用视觉反馈校正目标位置

        参数:
            current_pos: 当前目标位置
        返回:
           校正后的位置
        """
        # 检测目标物体，获取最新位置
        detections = self.vision.get_latest_detections()
        for det in detections:
            if det.get('confidence', 0) > 0.7:
                # 使用视觉检测到的位置进行校正
                visual_pos = det['center_3d']
                # 滤波平滑
                alpha = 0.7  # 滤波系数
                corrected = (
                    alpha * visual_pos[0] + (1 - alpha) * current_pos[0],
                    alpha * visual_pos[1] + (1 - alpha) * current_pos[1],
                    current_pos[2],  # z方向保持不变
                )
                print(f"[VLA] 视觉校正: {current_pos} -> {corrected}")
                return corrected

        return current_pos

    def _controlled_descent(self, target_pos: Tuple,
                            grasp_height: float) -> bool:
        """
        受控下降

        采用视觉反馈，实现精确定位

        参数:
            target_pos: 目标位置（物体实际位置）
            grasp_height: 预抓取高度
        返回:
            success: 是否成功
        """
        # 分步下降，每步都进行视觉校正
        steps = 5
        z_start = target_pos[2] + grasp_height
        z_end = target_pos[2]

        for i in range(steps):
            z_current = z_start + (z_end - z_start) * (i + 1) / steps
            target = (target_pos[0], target_pos[1], z_current)

            # 视觉校正x,y方向
            if i > 0:  # 从第二步开始校正
                target = self._visual_alignment(target)

            self._move_to_position(target)

            # 检查是否触碰到物体（通过力矩检测）
            force = self.gripper.get_current_force()
            if force > 2.0:
                print(f"[VLA] 检测到接触，力矩: {force}N")
                break

        return True

    def _check_grasp_success(self) -> bool:
        """检查抓取是否成功"""
        # 检查夹爪力矩
        force = self.gripper.get_current_force()
        position = self.gripper.get_current_position()

        # 成功条件：有力矩反馈，且夹爪未完全打开
        success = (force > 2.0) and (position < self.gripper.max_position * 0.85)

        print(f"[VLA] 抓取检查: force={force:.2f}N, position={position:.4f}m, success={success}")
        return success

    def execute_place_sequence(self, place_position: Tuple) -> bool:
        """
        执行放置序列

        参数:
            place_position: 放置位置 (x, y, z)
        返回:
            success: 是否成功
        """
        print(f"[VLA] 开始放置序列，位置: {place_position}")

        # 移动到放置位置上方
        approach_pos = (place_position[0], place_position[1], place_position[2] + 0.2)
        self._move_to_position(approach_pos)

        # 下降
        descend_pos = (place_position[0], place_position[1], place_position[2] + 0.05)
        self._move_to_position(descend_pos)

        # 释放
        self.gripper.open()

        # 撤回
        retract_pos = (place_position[0], place_position[1], place_position[2] + 0.3)
        self._move_to_position(retract_pos)

        print("[VLA] 放置序列完成")
        return True

    def execute_pick_and_place(self, pick_pos: Tuple, place_pos: Tuple) -> bool:
        """
        执行完整的拾取-放置任务

        参数:
            pick_pos: 抓取位置
            place_pos: 放置位置
        返回:
            success: 是否成功
        """
        print(f"[VLA] 执行拾取-放置任务: {pick_pos} -> {place_pos}")

        # 抓取
        if not self.execute_grasp_sequence(pick_pos):
            print("[VLA] 抓取失败，任务终止")
            return False

        # 搬运并放置
        if not self.execute_place_sequence(place_pos):
            print("[VLA] 放置失败")
            return False

        print("[VLA] 拾取-放置任务完成")
        return True
```

```python
"""
安全保障机制

确保机器人系统在安全范围内运行

    """
    安全监控器

    多层次安全保障
    """

    def __init__(self):
        self.workspace_bounds = {
            'x': (-1.0, 1.0),
            'y': (-1.0, 1.0),
            'z': (0.0, 1.5),
        }
        self.max_linear_velocity = 0.5  # m/s

        self.on_emergency_stop: Optional[Callable] = None
        self.is_monitoring = False
        self.emergency_stopped = False

    def check_position_safe(self, position: Tuple[float, float, float]) -> Tuple[bool, str]:
        """检查目标位置是否安全"""
        x, y, z = position
        for axis, val in zip(['x', 'y', 'z'], [x, y, z]):
            bounds = self.workspace_bounds[axis]
            if val < bounds[0] or val > bounds[1]:
                return False, f"位置超出{axis}轴工作范围"
        return True, "安全"

    def check_velocity_safe(self, velocity: Tuple[float, float, float],
                          angular_velocity: Tuple[float, float, float]) -> Tuple[bool, str]:
        """检查速度是否安全"""
        linear_norm = np.linalg.norm(velocity)
        angular_norm = np.linalg.norm(angular_velocity)
        if linear_norm > self.max_linear_velocity:
            return False, f"线速度超过限制"
        if angular_norm > self.max_angular_velocity:
            return False, f"角速度超过限制"
        return True, "安全"

    def emergency_stop(self):
        """触发紧急停止"""
        if self.emergency_stopped:
            return
        print("[Safety] ！！！紧急停止！！！")
        self.emergency_stopped = True
        if self.on_emergency_stop:
            self.on_emergency_stop()

    def recover_from_emergency(self):
        """从紧急停止恢复"""
        print("[Safety] 执行安全恢复程序...")
        self.emergency_stopped = False


class CollisionDetector:
    """
    碰撞检测器

    基于以下方法检测碰撞
    """

    def __init__(self, robot_model):
        self.robot_model = robot_model
        self.contact_threshold = 20.0  # N
        self.torque_surge_threshold = 5.0  # Nm
        self.last_torques = None

    def check_self_collision(self, joint_positions: List[float]) -> Tuple[bool, Optional[str]]:
        """检查自碰撞"""
        if len(joint_positions) >= 7:
            shoulder = joint_positions[0]
            elbow = joint_positions[2]
            if abs(shoulder) > 2.5 and abs(elbow) > 2.0:
                return True, "肩部与肘部可能碰撞"
        return False, None

    def check_torque_anomaly(self, current_torques: List[float]) -> Tuple[bool, Optional[str]]:
        """检测力矩异常（可能的碰撞）"""
        if self.last_torques is None:
            self.last_torques = current_torques
            return False, None
        for i, (curr, last) in enumerate(zip(current_torques, self.last_torques)):
            surge = abs(curr - last)
            if surge > self.torque_surge_threshold:
                self.last_torques = current_torques
                return True, f"关节{i}力矩突变"
        self.last_torques = current_torques
        return False, None

    def is_collision_detected(self, joint_positions: List[float],
                             joint_torques: List[float]) -> Tuple[bool, Optional[str]]:
        """综合碰撞检测"""
        self_collision, desc = self.check_self_collision(joint_positions)
        if self_collision:
            return True, f"自碰撞: {desc}"
        torque_anomaly, desc = self.check_torque_anomaly(joint_torques)
        if torque_anomaly:
            return True, f"碰撞检测(力矩异常): {desc}"
        return False, None
```

```
  5. 逐步恢复运动
```

```python
    异常恢复策略

    def handle_collision(self):

        self.arm.stop()

        return 'COLLISION_HANDLED'

    def handle_execution_timeout(self, action):

        self.arm.stop()

        return 'TIMEOUT_HANDLED'

    def handle_perception_failure(self):

        self.arm.go_home()
        self.retry_perception(max_attempts=3)
```
