### 1.2 机器人任务定义

**任务：Fetch 机械臂目标物抓取**

- **环境**：Gazebo 仿真 + 真实 Fetch 机器人
- **观测**：RGB-D 相机图像 + 机械臂关节角度 + 末端执行器位置
- **动作**：7-DOF 机械臂关节速度控制
- **目标**：将桌面上的目标物抓取并放置到指定区域
- **成功标准**：物体被移动到目标区域，且夹爪闭合

```
        相机视角
       ┌──────┐
       │Target│
       └──────┘
          │
    ┌─────────────┐
    │  Fetch 臂   │
    └─────────────┘
```

**状态空间**：

| 传感器 | 维度 | 说明 |
|--------|------|------|
| 视觉图像 | 224×224×3 | RGB 相机 |
| 深度图像 | 224×224×1 | 深度相机 |
| 关节角度 | 7 | 机械臂 7 个关节 |
| 末端位置 | 3 | 末端执行器 xyz |
| 目标位置 | 3 | 目标物 xyz（可选） |

**动作空间**：7 维关节速度，范围 $[-1.0, 1.0]$ rad/s

**本项目采用遥操作方式**，使用 PS4 手柄控制 Fetch 机械臂：

```python
"""
data_collection/joystick_controller.py
PS4 手柄遥操作控制器
"""
import pygame
import numpy as np
import time


class JoystickController:
    """
    PS4/DualShock4 手柄遥操作控制器
    映射：
    - 左摇杆（LX, LY）：机械臂基座旋转 + 俯仰
    - 右摇杆（RX, RY）：机械臂末端位置
    - L2/R2：夹爪开/合
    - 十字键：目标物选择（如有多个）
    """
    def __init__(self):
        pygame.init()
        pygame.joystick.init()
        
        if pygame.joystick.get_count() == 0:
            raise RuntimeError("未检测到手柄，请连接 PS4/DS4 手柄")
        
        self.joystick = pygame.joystick.Joystick(0)
        self.joystick.init()
        print(f"已连接手柄: {self.joystick.get_name()}")
        
        # 动作缩放系数
        self.speed_scale = 0.05  # 基础移动速度
        self.rotation_scale = 0.03  # 旋转速度
        
        # 夹爪状态（0=开, 1=闭合）
        self.gripper_state = 0.0
        self.gripper_target = 0.0
        
        # 滤波器参数（平滑控制）
        self.action_history = []
        self.history_len = 5
        
    def get_action(self):
        """
        读取手柄输入，转换为机械臂控制动作
        Returns:
            action: 8 维动作 [v_base, v_shoulder, v_elbow, 
                              v_wrist1, v_wrist2, v_wrist3,
                              gripper, reset]
        """
        pygame.event.pump()  # 更新手柄状态
        
        # ===== 左摇杆：基座旋转 + 肩部俯仰 =====
        lx = self.joystick.get_axis(0)  # 左摇杆 X
        ly = self.joystick.get_axis(1)  # 左摇杆 Y
        
        # ===== 右摇杆：末端位置控制 =====
        rx = self.joystick.get_axis(2)  # 右摇杆 X
        ry = self.joystick.get_axis(3)  # 右摇杆 Y
        
        # ===== L2/R2：夹爪控制 =====
        l2 = self.joystick.get_axis(4)  # L2 扳机 [0, 1]
        r2 = self.joystick.get_axis(5)  # R2 扳机 [0, 1]
        
        # L2 夹爪打开，R2 夹爪闭合
        self.gripper_target = r2 - (l2 + 1) / 2
        self.gripper_target = np.clip(self.gripper_target, 0, 1)
        
        # 平滑夹爪动作
        self.gripper_state += 0.1 * (self.gripper_target - self.gripper_state)
        
        # ===== 按钮：重置 / 保存 =====
        btn_cross = self.joystick.get_button(0)  # × 按钮
        btn_circle = self.joystick.get_button(1)  # ○ 按钮
        btn_triangle = self.joystick.get_button(2)  # △ 按钮
        btn_square = self.joystick.get_button(3)  # □ 按钮
        
        # ===== 动作组装 =====
        action = np.array([
            lx * self.rotation_scale,   # 基座旋转速度
            -ly * self.speed_scale,     # 肩部俯仰速度
            -rx * self.speed_scale,     # 肘部俯仰速度
            -ry * self.speed_scale,     # 腕部俯仰速度
            lx * self.speed_scale * 0.5,  # 腕部旋转速度
            0.0,                          # 预留轴
            self.gripper_state,           # 夹爪开度 [0, 1]
            1.0 if btn_cross else 0.0    # 重置标志
        ], dtype=np.float32)
        
        return action
    
    def close(self):
        pygame.quit()
```

### 2.2 传感器同步

多传感器数据同步是采集系统的关键技术。本项目需要同步视觉、关节状态和动作指令。

```python
"""
data_collection/synchronized_recorder.py
多传感器同步数据记录器
"""
import numpy as np
import h5py
import time
import threading
from collections import deque


class SynchronizedRecorder:
    """
    多传感器同步记录器
    核心设计：
    - 视觉数据：30 Hz 采集
    - 关节数据：100 Hz 采集
    - 动作用于动作标签
    - 时间戳对齐到最近视觉帧
    """
    def __init__(self, record_dir="./demonstrations", buffer_size=100):
        self.record_dir = record_dir
        self.buffer_size = buffer_size
        
        # 数据缓冲区（FIFO）
        self.rgb_buffer = deque(maxlen=buffer_size)
        self.depth_buffer = deque(maxlen=buffer_size)
        self.joint_buffer = deque(maxlen=buffer_size)
        self.action_buffer = deque(maxlen=buffer_size)
        self.timestamp_buffer = deque(maxlen=buffer_size)
        
        # 时间同步
        self.last_rgb_stamp = 0
        self.last_joint_stamp = 0
        
        # 录制状态
        self.is_recording = False
        self.episode_buffer = []
        
    def rgb_callback(self, rgb_image, timestamp):
        """
        RGB 图像回调（来自 ROS 话题或仿真）
        Args:
            rgb_image: H×W×3 numpy 数组，uint8
            timestamp: 时间戳（秒）
        """
        if not self.is_recording:
            return
        self.rgb_buffer.append({
            'image': rgb_image.copy(),
            'timestamp': timestamp,
            'type': 'rgb'
        })
        self.last_rgb_stamp = timestamp
        
    def depth_callback(self, depth_image, timestamp):
        """深度图像回调"""
        if not self.is_recording:
            return
        self.depth_buffer.append({
            'image': depth_image.copy(),
            'timestamp': timestamp,
            'type': 'depth'
        })
        
    def joint_callback(self, joint_positions, joint_velocities, timestamp):
        """
        关节状态回调
        Args:
            joint_positions: 7 维关节角度 (rad)
            joint_velocities: 7 维关节速度 (rad/s)
            timestamp: 时间戳（秒）
        """
        if not self.is_recording:
            return
        self.joint_buffer.append({
            'positions': np.array(joint_positions),
            'velocities': np.array(joint_velocities),
            'timestamp': timestamp,
            'type': 'joint'
        })
        self.last_joint_stamp = timestamp
        
    def action_callback(self, action, timestamp):
        """
        动作指令回调
        Args:
            action: 8 维动作向量
            timestamp: 时间戳（秒）
        """
        if not self.is_recording:
            return
        self.action_buffer.append({
            'action': np.array(action),
            'timestamp': timestamp,
            'type': 'action'
        })
        
    def _synchronize_data(self):
        """
        时间同步算法：
        将关节和动作数据对齐到最近的视觉帧时间戳
        使用最近邻插值
        """
        synchronized = []
        
        while self.rgb_buffer:
            rgb_data = self.rgb_buffer.popleft()
            ref_stamp = rgb_data['timestamp']
            
            # 找最近的关节数据
            joint_data = self._find_nearest(self.joint_buffer, ref_stamp)
            action_data = self._find_nearest(self.action_buffer, ref_stamp)
            
            # 找最近的深度数据（允许更大时间差）
            depth_data = self._find_nearest(
                self.depth_buffer, ref_stamp, max_delta=0.1
            )
            
            if joint_data is not None:
                synchronized.append({
                    'rgb': rgb_data['image'],
                    'depth': depth_data['image'] if depth_data else None,
                    'joint_positions': joint_data['positions'],
                    'joint_velocities': joint_data['velocities'],
                    'action': action_data['action'] if action_data else None,
                    'timestamp': ref_stamp
                })
        
        return synchronized
    
    def _find_nearest(self, buffer, ref_stamp, max_delta=0.05):
        """在 buffer 中找到时间戳最接近 ref_stamp 的数据"""
        if not buffer:
            return None
        nearest = min(buffer, 
                      key=lambda x: abs(x['timestamp'] - ref_stamp))
        if abs(nearest['timestamp'] - ref_stamp) <= max_delta:
            return nearest
        return None
    
    def start_episode(self):
        """开始一个新的示教回合"""
        self.is_recording = True
        self.episode_buffer = []
        print("[Recorder] 开始录制回合")
        
    def stop_episode(self):
        """停止录制并同步数据"""
        self.is_recording = False
        synchronized = self._synchronize_data()
        self.episode_buffer = synchronized
        print(f"[Recorder] 回合结束，采集 {len(synchronized)} 帧有效数据")
        return synchronized
    
    def save_episode(self, episode_id, file_path=None):
        """
        将回合数据保存为 HDF5 格式
        Args:
            episode_id: 回合编号
            file_path: 保存路径（默认自动生成）
        """
        if file_path is None:
            file_path = f"{self.record_dir}/episode_{episode_id:04d}.h5"
        
        with h5py.File(file_path, 'w') as f:
            # 保存配置
            f.attrs['episode_id'] = episode_id
            f.attrs['num_frames'] = len(self.episode_buffer)
            f.attrs['timestamp'] = time.time()
            
            # 创建数据集
            n = len(self.episode_buffer)
            
            rgb_shape = (n, 224, 224, 3)
            depth_shape = (n, 224, 224, 1)
            joint_shape = (n, 7)
            action_shape = (n, 8)
            
            rgb_ds = f.create_dataset('rgb', shape=rgb_shape, dtype='uint8')
            depth_ds = f.create_dataset('depth', shape=depth_shape, 
                                        dtype='float32', chunks=True)
            joint_pos_ds = f.create_dataset('joint_positions', 
                                             shape=joint_shape, dtype='float32')
            joint_vel_ds = f.create_dataset('joint_velocities', 
                                             shape=joint_shape, dtype='float32')
            action_ds = f.create_dataset('actions', shape=action_shape, 
                                         dtype='float32')
            ts_ds = f.create_dataset('timestamps', shape=(n,), 
                                      dtype='float64')
            
            # 填充数据
            for i, frame in enumerate(self.episode_buffer):
                rgb_ds[i] = frame['rgb']
                if frame['depth'] is not None:
                    depth_ds[i] = frame['depth']
                joint_pos_ds[i] = frame['joint_positions']
                joint_vel_ds[i] = frame['joint_velocities']
                action_ds[i] = frame['action'] if frame['action'] is not None else np.zeros(8)
                ts_ds[i] = frame['timestamp']
        
        print(f"[Recorder] 数据已保存至: {file_path}")
        return file_path
```

### 2.3 数据存储

```python
"""
data_collection/collect_demonstrations.py
主采集脚本：连接仿真环境，采集示教数据
"""
import gymnasium as gym
import numpy as np
import time
import os
import argparse
from joystick_controller import JoystickController
from synchronized_recorder import SynchronizedRecorder


class DemonstrationCollector:
    """
    示教数据采集器
    将遥操作、仿真环境和数据记录整合在一起
    """
    def __init__(self, env_name='FetchPickAndPlace-v2', record_dir='./demonstrations'):
        self.env = gym.make(env_name)
        self.recorder = SynchronizedRecorder(record_dir=record_dir)
        self.joystick = None
        self.episode_count = 0
        
        # 创建保存目录
        os.makedirs(record_dir, exist_ok=True)
        
    def connect_joystick(self):
        """连接手柄"""
        try:
            self.joystick = JoystickController()
            print("[Collector] 手柄已连接")
        except RuntimeError as e:
            print(f"[Collector] 警告: {e}，将使用键盘控制")
            self.joystick = None
            
    def collect_episode(self, max_steps=500, verbose=True):
        """
        采集一个示教回合
        Args:
            max_steps: 最大步数
            verbose: 是否打印状态
        """
        obs, info = self.env.reset()
        self.recorder.start_episode()
        
        episode_reward = 0
        for step in range(max_steps):
            # 获取动作
            if self.joystick:
                action = self.joystick.get_action()
                # 如果是重置信号
                if action[7] > 0.5:
                    obs, info = self.env.reset()
                    if verbose:
                        print(f"[Step {step}] 环境重置")
                    continue
            else:
                # 键盘回退：使用纯随机动作
                action = self.env.action_space.sample()
            
            # 记录当前状态和动作
            rgb_image = obs['observation']  # 本项目使用状态观测
            joint_pos = obs['achieved_goal']
            joint_vel = obs.get('joint_velocities', np.zeros(7))
            
            # 采集回调（异步写入缓冲区）
            timestamp = time.time()
            self.recorder.action_callback(action, timestamp)
            self.recorder.joint_callback(joint_pos, joint_vel, timestamp)
            
            # 执行动作
            next_obs, reward, terminated, truncated, info = self.env.step(action)
            
            episode_reward += reward
            obs = next_obs
            
            if verbose and step % 20 == 0:
                print(f"[Step {step}] reward={reward:.3f} "
                      f"gripper={action[6]:.2f}")
            
            if terminated or truncated:
                if verbose:
                    print(f"[Episode End] 总 reward={episode_reward:.2f}")
                break
        
        # 停止录制并保存
        self.recorder.stop_episode()
        self.episode_count += 1
        self.recorder.save_episode(self.episode_count)
        
        return episode_reward
    
    def run(self, num_episodes=10, episodes_per_save=5):
        """
        运行采集主循环
        Args:
            num_episodes: 总共采集的回合数
            episodes_per_save: 每采集多少回合保存一次
        """
        print(f"开始采集 {num_episodes} 个示教回合...")
        print("按 Ctrl+C 中断采集")
        
        try:
            for i in range(num_episodes):
                print(f"\n{'='*50}")
                print(f"采集回合 {i+1}/{num_episodes}")
                print(f"{'='*50}")
                
                self.collect_episode()
                
                # 定期保存
                if (i + 1) % episodes_per_save == 0:
                    print(f"\n[Checkpoint] 已完成 {i+1}/{num_episodes} 回合")
                    
        except KeyboardInterrupt:
            print("\n[Collector] 用户中断采集")
        finally:
            self.env.close()
            if self.joystick:
                self.joystick.close()
            print(f"\n采集完成！共 {self.episode_count} 个回合")


if __name__ == '__main__':
    parser = argparse.ArgumentParser(description='Fetch 机器人示教数据采集')
    parser.add_argument('--env', type=str, default='FetchPickAndPlace-v2')
    parser.add_argument('--output', type=str, default='./demonstrations')
    parser.add_argument('--episodes', type=int, default=20)
    args = parser.parse_args()
    
    collector = DemonstrationCollector(
        env_name=args.env, 
        record_dir=args.output
    )
    collector.connect_joystick()
    collector.run(num_episodes=args.episodes)
```

## 3. 算法选择

### 3.1 DAgger 算法

**DAgger（Dataset Aggregation）** 的核心思想是通过迭代式在线查询专家，修正策略的分布偏移问题。

**算法流程**：

```
1. 用专家策略 π* 采集示范数据集 D
2. 用 D 训练策略 π_θ（行为克隆）
3. 用 π_θ 收集轨迹，得到状态分布 p_π
4. 在这些状态上查询专家得到最优动作 a* = π*(s)
5. 将 (s, a*) 加入 D，回到步骤 2
```

**数学表达**：

$$
\pi_{i+1} = \arg\min_{\pi} \mathbb{E}_{s \sim d_{\pi_i}} [\ell(\pi(s), \pi^*(s))]
$$

其中 $d_{\pi_i}$ 是策略 $\pi_i$ 访问的状态分布，而非专家分布。

**DAgger 的优势**：

| 特性 | 行为克隆 | DAgger |
|------|----------|--------|
| 数据分布 | 固定（仅专家分布） | 动态（覆盖策略分布） |
| 分布偏移 | 严重，误差累积 | 通过在线查询修正 |
| 样本复杂度 | 低（一次性采集） | 中等（需要多轮采集） |
| 专家需求 | 仅初始采集 | 需要持续访问专家 |

### 3.2 GAIL 算法

**GAIL（Generative Adversarial Imitation Learning）** 通过对抗训练区分专家轨迹和策略轨迹，隐式学习奖励函数。

**核心思想**：

- 判别器 $D_\omega(s,a)$ 学习区分专家轨迹和策略轨迹
- 策略 $\pi_\theta$ 被训练为"骗过"判别器
- 判别器输出作为奖励信号驱动策略优化

**优化目标**：

$$
\min_{\pi} \max_{D} \mathbb{E}_{\pi}[\log(1 - D(s,a))] + \mathbb{E}_{\pi_E}[\log D(s,a)]
$$

**GAIL 的优势**：

- 不需要显式学习奖励函数
- 可处理高维状态和动作空间
- 训练好的判别器可迁移到相似任务

### 3.3 性能对比

| 维度 | BC | DAgger | GAIL |
|------|-----|--------|------|
| **实现复杂度** | ⭐ 低 | ⭐⭐ 中 | ⭐⭐⭐ 高 |
| **样本效率** | ⭐ 高（一次性采集） | ⭐⭐ 中（多轮采集） | ⭐⭐ 中（需策略采样） |
| **训练稳定性** | ⭐ 高 | ⭐⭐ 中 | ⭐⭐ 中 |
| **分布偏移处理** | ❌ 无 | ✅ 在线修正 | ✅ 对抗适应 |
| **泛化能力** | ⭐ 低 | ⭐⭐ 中 | ⭐⭐⭐ 高 |
| **计算资源** | ⭐ 低 | ⭐⭐ 中 | ⭐⭐⭐ 高 |
| **适合场景** | 数据稀缺 | 中等数据 | 大量在线数据 |

**本项目选择 DAgger 作为主算法**，原因如下：

1. **Fetch 任务可获取专家策略**（Gymnasium 内置）
2. **DAgger 实现相对简单**，易于调试和迭代
3. **样本效率适中**，适合 10-20 个示教回合的数据量
4. **可与 BC 对比验证**，直观看出分布偏移修正效果

class PolicyConverter:

        导出为 ONNX 格式
        Args:
            save_path: 保存路径
            input_dims: 输入维度元组
            dynamic_axes: 动态轴定义（如批量大小可变）

            export_params=True,
            opset_version=14,
            do_constant_folding=True,
            input_names=['observation'],
            output_names=['action'],
            dynamic_axes=dynamic_axes or {
                'observation': {0: 'batch_size'},
                'action': {0: 'batch_size'}
            }
        )
        
        print(f"[Converter] ONNX 模型已保存: {save_path}")
        
    def export_to_torchscript(self, save_path, input_dims):
        """
        导出为 TorchScript 格式（更适合 ROS/Python 环境）
        """
        dummy_input = torch.randn(1, *input_dims).to(self.device)
        
        # TorchScript tracing
        traced = torch.jit.trace(self.policy, dummy_input)
        traced.save(save_path)
        
        print(f"[Converter] TorchScript 模型已保存: {save_path}")
        
    def verify_onnx(self, onnx_path, test_input):
        """
        验证 ONNX 模型输出是否与原始 PyTorch 模型一致
        """
        import onnxruntime as ort
        
        # PyTorch 输出
        self.policy.eval()
        with torch.no_grad():
            pt_output = self.policy(test_input).numpy()
        
        # ONNX 输出
        session = ort.InferenceSession(onnx_path)
        onnx_output = session.run(
            None, 
            {'observation': test_input.numpy()}
        )[0]
        
        # 比较
        diff = np.abs(pt_output - onnx_output).max()
        print(f"[Verifier] PyTorch vs ONNX 最大差异: {diff:.6f}")
        
        if diff < 1e-5:
            print("[Verifier] ✅ ONNX 模型验证通过")
            return True
        else:
            print("[Verifier] ❌ ONNX 模型验证失败")

### 5.2 实时推理

```python
"""
deployment/robot_inference.py
机器人实时推理模块
"""
import torch
import numpy as np
import time
import os


class RobotPolicyRunner:
    """
    机器人策略运行器
    负责：模型加载 → 观测处理 → 动作推理 → 控制输出
    """
    def __init__(self, model_path, device='cpu', action_scale=1.0):
        """
        Args:
            model_path: 模型文件路径（.pt, .onnx, .torchscript）
            device: 推理设备
            action_scale: 动作输出缩放系数
        """
        self.device = torch.device(device)
        self.action_scale = action_scale
        self.model_path = model_path
        
        # 加载模型
        self.policy = self._load_model(model_path)
        self.policy.eval()
        
        # 性能统计
        self.inference_times = []
        self.total_calls = 0
        
        print(f"[Runner] 模型已加载: {model_path}")
        print(f"[Runner] 推理设备: {self.device}")
        
    def _load_model(self, model_path):
        """根据文件扩展名加载模型"""
        ext = os.path.splitext(model_path)[1].lower()
        
        if ext == '.pt':
            # PyTorch 模型
            from networks import PolicyNetwork  # 导入网络定义
            policy = PolicyNetwork()
            policy.load_state_dict(torch.load(model_path, map_location=self.device))
            policy.to(self.device)
            
        elif ext == '.onnx':
            # ONNX 模型（需要 onnxruntime）
            import onnxruntime as ort
            sess_options = ort.SessionOptions()
            sess_options.graph_optimization_level = (
                ort.GraphOptimizationLevel.ORT_ENABLE_ALL
            )
            policy = ort.InferenceSession(model_path, sess_options)
            self.onnx_session = policy
            return policy  # ONNX 模型直接返回 session
            
        elif ext == '.torchscript':
            # TorchScript 模型
            policy = torch.jit.load(model_path, map_location=self.device)
            
        else:
            raise ValueError(f"不支持的模型格式: {ext}")
        
        return policy
        
    def predict(self, observation):
        """
        策略推理（单步）
        Args:
            observation: numpy 数组或 torch.Tensor，观测向量
        Returns:
            action: numpy 数组，动作向量
        """
        start_time = time.perf_counter()
        
        # 转换为 torch.Tensor 并添加批次维度
        if isinstance(observation, np.ndarray):
            obs = torch.from_numpy(observation).float()
        else:
            obs = observation.float()
            
        if obs.dim() == 1:
            obs = obs.unsqueeze(0)  # 添加批次维度
        obs = obs.to(self.device)
        
        # 推理
        with torch.no_grad():
            if hasattr(self, 'onnx_session'):
                # ONNX 推理
                action = self.onnx_session.run(
                    None, {'observation': obs.cpu().numpy()}
                )[0]
                action = torch.from_numpy(action)
            else:
                # PyTorch / TorchScript 推理
                action = self.policy(obs)
        
        action = action.cpu().numpy().squeeze()
        action = np.clip(action * self.action_scale, -1, 1)
        
        # 统计推理时间
        inference_time = time.perf_counter() - start_time
        self.inference_times.append(inference_time)
        self.total_calls += 1
        
        return action
    
    def get_stats(self):
        """获取推理性能统计"""
        if not self.inference_times:
            return {'mean_ms': 0, 'std_ms': 0, 'calls': 0}
        
        times_ms = np.array(self.inference_times) * 1000
        return {
            'mean_ms': times_ms.mean(),
            'std_ms': times_ms.std(),
            'min_ms': times_ms.min(),
            'max_ms': times_ms.max(),
            'calls': self.total_calls
        }


def normalize_observation(obs, mean=None, std=None):
    """
    观测归一化
    Args:
        obs: 原始观测 numpy 数组
        mean: 均值（若为 None，从 obs 计算）
        std: 标准差（若为 None，从 obs 计算）
    Returns:
        normalized_obs: 归一化后的观测
        mean, std: 用于后续反归一化
    """
    if mean is None:
        mean = np.mean(obs, axis=0)
    if std is None:
        std = np.std(obs, axis=0) + 1e-8  # 避免除零
    
    return (obs - mean) / std, mean, std


def denormalize_action(action, action_min, action_max):
    """将归一化的动作反归一化到实际物理范围"""
    return ((action + 1) / 2) * (action_max - action_min) + action_min


### 5.3 误差校正

实物部署时，仿真环境与真实环境之间存在**仿真-现实差距（Sim-to-Real Gap）**，需要通过误差校正来弥补。

**常见误差来源**：

| 误差类型 | 描述 | 校正方法 |
|----------|------|----------|
| 观测误差 | 相机标定差异、光照变化 | 在线自适应、相机标定 |
| 物理参数差异 | 摩擦力、惯性、延迟 | 系统辨识、Domain Randomization |
| 传感器延迟 | 视觉处理延迟、通信延迟 | 时间戳对齐、预测补偿 |
| 执行延迟 | 控制器采样周期、动作延迟 | 动作缓冲、前馈补偿 |

**误差校正策略**：

```python
"""
deployment/error_correction.py
误差校正模块
"""
import numpy as np
from scipy.ndimage import uniform_filter1d


class ErrorCorrector:
    """
    误差校正器
    采用多种策略修正仿真到真实的差距
    """
    def __init__(self, robot_state_dim, action_dim):
        self.state_dim = robot_state_dim
        self.action_dim = action_dim
        
        # 在线自适应参数
        self.observation_history = []
        self.action_history = []
        self.observation_mean = None
        self.observation_std = None
        
        # 延迟补偿
        self.action_buffer = []
        self.action_buffer_size = 5
        self.delay_steps = 2
        
        # 传感器偏差
        self.bias = np.zeros(robot_state_dim)
        self.bias_correction_alpha = 0.01  # 偏差更新系数
        
    def update_observation_stats(self, obs):
        """
        在线更新观测统计（用于自适应归一化）
        使用滑动均值和标准差
        """
        self.observation_history.append(obs)
        
        # 保留最近 1000 帧
        if len(self.observation_history) > 1000:
            self.observation_history.pop(0)
        
        if len(self.observation_history) > 100:
            stacked = np.stack(self.observation_history)
            self.observation_mean = np.mean(stacked, axis=0)
            self.observation_std = np.std(stacked, axis=0) + 1e-8
            
    def correct_observation(self, obs, use_adaptive=True):
        """
        观测误差校正
        Args:
            obs: 原始观测
            use_adaptive: 是否使用自适应归一化
        Returns:
            corrected_obs: 校正后的观测
        """
        # 偏差校正
        obs_corrected = obs - self.bias
        
        # 自适应归一化
        if use_adaptive and self.observation_mean is not None:
            obs_corrected = (
                (obs_corrected - self.observation_mean) / 
                self.observation_std
            )
        
        return obs_corrected
    
    def update_bias(self, expected_obs, actual_obs):
        """
        根据观测误差更新偏差估计
        Args:
            expected_obs: 期望观测（来自模型预测）
            actual_obs: 实际观测
        """
        error = actual_obs - expected_obs
        self.bias += self.bias_correction_alpha * error
        
    def compensate_action_delay(self, action):
        """
        动作延迟补偿
        使用动作缓冲区的历史动作加权平均
        """
        self.action_buffer.append(action.copy())
        
        if len(self.action_buffer) > self.action_buffer_size:
            self.action_buffer.pop(0)
        
        if len(self.action_buffer) <= self.delay_steps:
            return action
        
        # 使用 delay_steps 前的动作（前馈补偿）
        compensated = self.action_buffer[-self.delay_steps - 1]
        return compensated
    
    def correct_policy_output(self, action, robot_state):
        """
        综合策略输出校正
        Args:
            action: 策略原始输出
            robot_state: 当前机器人状态
        Returns:
            corrected_action: 校正后的动作
        """
        # 延迟补偿
        action = self.compensate_action_delay(action)
        
        # 动作限幅
        action = np.clip(action, -1, 1)
        
        # 平滑处理（防止突变）
        if len(self.action_history) > 0:
            last_action = self.action_history[-1]
            action = 0.9 * action + 0.1 * last_action
        
        self.action_history.append(action.copy())
        return action

## 6. 评估与优化

### 6.1 成功率评估

**评估指标体系**：

| 指标 | 定义 | 计算方式 |
|------|------|----------|
| **成功率（SR）** | 任务完成的回合比例 | $SR = \frac{N_{success}}{N_{total}}$ |
| **平均奖励（AR）** | 回合平均累计奖励 | $AR = \frac{1}{N}\sum R_i$ |
| **平均步数（AS）** | 成功回合的平均步数 | 越少越好 |
| **位置误差（PE）** | 末端执行器到目标距离 | 实时监控 |
| **碰撞率（CR）** | 发生碰撞的回合比例 | 环境内置检测 |

**评估脚本**：

```python
"""

策略评估器
"""
import numpy as np
import gymnasium as gym


class PolicyEvaluator:
    """
    策略评估器
    支持 BC、DAgger、GAIL 等各种策略的评估
    """
    def __init__(self, env_name, num_episodes=20, render=False):
        self.env_name = env_name
        self.num_episodes = num_episodes
        self.render = render
        
        self.env = gym.make(env_name)
        
    def evaluate(self, policy_runner, verbose=True):
        """
        评估策略
        Args:
            policy_runner: RobotPolicyRunner 实例
        Returns:
            metrics: 评估指标字典
        """
        episode_rewards = []
        episode_successes = []
        episode_steps = []
        final_distances = []
        
        for ep in range(self.num_episodes):
            obs, info = self.env.reset()
            done = False
            total_reward = 0
            steps = 0
            
            while not done:
                # 策略推理
                action = policy_runner.predict(obs)
                
                # 执行动作
                next_obs, reward, terminated, truncated, info = (
                    self.env.step(action)
                )
                
                total_reward += reward
                obs = next_obs
                steps += 1
                
                done = terminated or truncated
                
                if self.render:
                    self.env.render()
            
            # 获取本回合结果
            is_success = info.get('is_success', False)
            episode_rewards.append(total_reward)
            episode_successes.append(is_success)
            episode_steps.append(steps)
            final_distances.append(
                info.get('goal_distance', 0.0)
            )
            
            if verbose:
                status = "✅ 成功" if is_success else "❌ 失败"
                print(f"Episode {ep+1}/{self.num_episodes}: "
                      f"{status}, reward={total_reward:.2f}, "
                      f"steps={steps}, dist={final_distances[-1]:.3f}")
        
        # 计算统计指标
        metrics = {
            'success_rate': np.mean(episode_successes),
            'avg_reward': np.mean(episode_rewards),
            'std_reward': np.std(episode_rewards),
            'avg_steps': np.mean(episode_steps),
            'std_steps': np.std(episode_steps),
            'avg_final_distance': np.mean(final_distances),
            'episode_rewards': episode_rewards,
            'episode_successes': episode_successes,
            'episode_steps': episode_steps
        }
        
        return metrics
    
    def compare_policies(self, policy_runners, policy_names):
        """
        对比多个策略的性能
        Args:
            policy_runners: RobotPolicyRunner 实例列表
            policy_names: 策略名称列表
        Returns:
            comparison_table: 对比结果表格
        """
        results = []
        
        for runner, name in zip(policy_runners, policy_names):
            print(f"\n{'='*50}")
            print(f"评估策略: {name}")
            print(f"{'='*50}")
            
            metrics = self.evaluate(runner, verbose=True)
            metrics['policy_name'] = name
            results.append(metrics)
        
        # 打印对比表格
        print(f"\n{'='*70}")
        print(f"{'策略':<20} {'成功率':<12} {'平均奖励':<12} {'平均步数':<12}")
        print(f"{'='*70}")
        
        for r in results:
            print(f"{r['policy_name']:<20} "
                  f"{r['success_rate']:.2%}       "
                  f"{r['avg_reward']:.2f}       "
                  f"{r['avg_steps']:.1f}")
        print(f"{'='*70}")
        
        return results
    
    def close(self):
        self.env.close()


def statistical_significance(results_a, results_b, metric='success_rate'):
    """
    检验两个策略在某一指标上是否有显著差异
    使用 Welch's t-test
    """
    from scipy import stats
    
    data_a = [r[metric] for r in results_a]
    data_b = [r[metric] for r in results_b]
    
    t_stat, p_value = stats.ttest_ind(data_a, data_b, equal_var=False)
    
    significance = "显著" if p_value < 0.05 else "不显著"
    print(f"t-test ({metric}): t={t_stat:.3f}, p={p_value:.4f} ({significance})")
    
    return p_value < 0.05


### 6.2 泛化性测试

泛化性是衡量策略质量的重要指标——策略在训练任务上表现良好，不意味着能泛化到新任务。

**泛化性测试维度**：

| 测试维度 | 描述 | 方法 |
|----------|------|------|
| **目标位置泛化** | 目标移动到新位置 | 在不同位置测试成功率 |
| **物体形状泛化** | 换成不同形状的物体 | 更换抓取物体 |
| **光照变化泛化** | 改变环境光照 | 调整光照参数 |
| **背景干扰泛化** | 背景中有其他物体 | 添加干扰物 |
| **噪声鲁棒性** | 传感器噪声增强 | 注入人工噪声 |

class DomainRandomization:
    """
    Domain Randomization：训练时随机化仿真参数
    使策略对真实世界的不确定性更鲁棒
    """
    def __init__(self):
        # 随机化参数范围
        self.param_ranges = {
            'mass_scale': (0.8, 1.2),
            'friction': (0.5, 1.5),
            'action_delay': (0.0, 0.05),
            'camera_noise': (0.0, 0.05),
            'light_intensity': (0.7, 1.3),
        }
        
    def randomize(self, param_name):
        """对指定参数进行随机化"""
        if param_name not in self.param_ranges:
            return 1.0
        lo, hi = self.param_ranges[param_name]
        return np.random.uniform(lo, hi)
    
    def get_randomized_env_params(self):
        """获取所有随机化后的环境参数"""
        return {name: self.randomize(name) for name in self.param_ranges}


### 6.3 优化迭代

基于评估结果进行策略优化迭代，是提升模仿学习效果的关键环节。

**优化方向决策树**：

```
评估结果分析
├── 成功率低 → 诊断
│   ├── 训练损失高 → 增加训练 epochs，检查网络结构
│   ├── 训练损失低但评估低 → 分布偏移 → 使用 DAgger
│   └── 评估初期成功，后期失败 → 误差累积 → 引入时序模型
│
├── 泛化性差 → 改进
│   ├── 增加训练数据多样性
│   ├── 使用 Domain Randomization
│   └── 引入数据增强（视觉噪声、光照变化）
│
└── 推理速度慢 → 加速
    ├── 模型剪枝
    ├── 量化（FP16/INT8）
    └── 知识蒸馏
```

**常见优化策略**：

| 优化方向 | 具体方法 | 预期效果 |
|----------|----------|----------|
| 数据质量 | 移除噪声样本，提高示教一致性 | +5~15% 成功率 |
| 数据多样性 | 采集不同目标位置/光照下的数据 | +10~20% 泛化性 |
| 网络容量 | 增大网络，添加残差连接 | 适应更复杂任务 |
| 训练技巧 | Label smoothing、Mixup | 减少过拟合 |
| 在线微调 | 部署后在线收集失败样本微调 | 持续改进 |

```python
"""
dagger/train_dagger.py
完整 DAgger 训练脚本
"""
import os
import time
import numpy as np
import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader, TensorDataset
import gymnasium as gym
import argparse
import json

from networks import PolicyNetwork
from data_utils import load_demonstrations, BCM losses
from evaluation.evaluator import PolicyEvaluator
from evaluation.robot_inference import RobotPolicyRunner


class DAggerTrainer:
    """
    DAgger 训练器
    实现完整的 DAgger 迭代训练流程
    """
    def __init__(self, config):
        self.config = config
        self.device = torch.device(config.device)
        
        # 创建环境
        self.env = gym.make(config.env_name)
        state_dim = self.env.observation_space.shape[0]
        action_dim = self.env.action_space.shape[0]
        
        # 创建策略网络
        self.policy = PolicyNetwork(state_dim, action_dim,
                                     hidden_dims=config.policy_hidden_dims)
        self.policy.to(self.device)
        
        # 专家策略（使用环境的内置策略作为专家）
        self.expert = self.env.unwrapped._get_default_policy()
        
        # 优化器
        self.optimizer = optim.Adam(
            self.policy.parameters(), 
            lr=config.lr, 
            weight_decay=config.weight_decay
        )
        
        # 数据集
        self.dataset = {'observations': [], 'actions': []}
        
        # 监控器
        self.train_losses = []
        self.eval_results = []
        
        # 创建输出目录
        os.makedirs(config.output_dir, exist_ok=True)
        
    def collect_expert_demonstrations(self, num_episodes=10, max_steps=500):
        """
        采集专家示教数据
        """
        demonstrations = {'observations': [], 'actions': []}
        
        for ep in range(num_episodes):
            obs, info = self.env.reset()
            done = False
            steps = 0
            
            while not done and steps < max_steps:
                # 获取专家动作
                with torch.no_grad():
                    obs_t = torch.FloatTensor(obs).unsqueeze(0).to(self.device)
                    expert_action = self.expert.get_action(obs_t).cpu().numpy().squeeze()
                
                demonstrations['observations'].append(obs.copy())
                demonstrations['actions'].append(expert_action)
                
                # 执行动作
                obs, reward, terminated, truncated, info = self.env.step(expert_action)
                done = terminated or truncated
                steps += 1
            
            print(f"[Expert Demo] Episode {ep+1}/{num_episodes}, steps={steps}")
        
        return demonstrations
    
    def collect_rollouts(self, num_rollouts=5, max_steps=500):
        """
        用当前策略收集轨迹，并在访问的状态上查询专家
        这是 DAgger 的核心步骤
        """
        rollouts = {'observations': [], 'expert_actions': []}
        
        for ep in range(num_rollouts):
            obs, info = self.env.reset()
            done = False
            steps = 0
            
            while not done and steps < max_steps:
                # 策略推理
                with torch.no_grad():
                    obs_t = torch.FloatTensor(obs).unsqueeze(0).to(self.device)
                    policy_action = self.policy(obs_t).cpu().numpy().squeeze()
                
                # 记录当前状态
                rollouts['observations'].append(obs.copy())
                
                # 查询专家动作（DAgger 关键）
                with torch.no_grad():
                    expert_action = self.expert.get_action(obs_t).cpu().numpy().squeeze()
                rollouts['expert_actions'].append(expert_action)
                
                # 用策略动作执行（不是专家动作！）
                obs, reward, terminated, truncated, info = self.env.step(policy_action)
                done = terminated or truncated
                steps += 1
            
            print(f"[Rollout] Episode {ep+1}/{num_rollouts}, steps={steps}")
        
        return rollouts
    
    def update_policy(self, epochs=10, batch_size=256):
        """
        用行为克隆损失更新策略
        """
        all_obs = np.array(self.dataset['observations'])
        all_act = np.array(self.dataset['actions'])
        
        # 创建 DataLoader
        dataset = TensorDataset(
            torch.FloatTensor(all_obs),
            torch.FloatTensor(all_act)
        )
        loader = DataLoader(dataset, batch_size=batch_size, shuffle=True)
        
        epoch_losses = []
        self.policy.train()
        
        for epoch in range(epochs):
            epoch_loss = 0
            for batch_obs, batch_act in loader:
                batch_obs = batch_obs.to(self.device)
                batch_act = batch_act.to(self.device)
                
                # 前向传播
                pred_act = self.policy(batch_obs)
                
                # MSE 损失（回归任务）
                loss = nn.functional.mse_loss(pred_act, batch_act)
                
                # 反向传播
                self.optimizer.zero_grad()
                loss.backward()
                self.optimizer.step()
                
                epoch_loss += loss.item()
            
            avg_loss = epoch_loss / len(loader)
            epoch_losses.append(avg_loss)
        
        self.policy.eval()
        return np.mean(epoch_losses)
    
    def train(self):
        """
        DAgger 主训练循环
        """
        print("=" * 60)
        print("开始 DAgger 训练")
        print("=" * 60)
        
        # ====== 步骤 1: 初始 BC 预训练 ======
        print("\n[阶段 1] 采集专家示教数据...")
        init_demos = self.collect_expert_demonstrations(
            num_episodes=self.config.bc_pretrain_epochs // 5
        )
        self.dataset['observations'].extend(init_demos['observations'])
        self.dataset['actions'].extend(init_demos['actions'])
        
        print(f"[BC 预训练] 共 {len(self.dataset['observations'])} 个样本")
        self.update_policy(epochs=self.config.bc_pretrain_epochs)
        
        # ====== DAgger 迭代 ======
        for iteration in range(self.config.num_iterations):
            print(f"\n{'='*60}")
            print(f"DAgger 迭代 {iteration + 1}/{self.config.num_iterations}")
            print(f"{'='*60}")
            
            # 步骤 2: 用当前策略收集轨迹
            print("\n[步骤 2] 用当前策略收集轨迹...")
            rollouts = self.collect_rollouts(
                num_rollouts=self.config.num_rollouts_per_iter
            )
            
            # 步骤 3: 查询专家获取校正动作
            print("[步骤 3] 查询专家动作（标签数据扩展）")
            self.dataset['observations'].extend(rollouts['observations'])
            self.dataset['actions'].extend(rollouts['expert_actions'])
            print(f"[Dataset] 累计 {len(self.dataset['observations'])} 个样本")
            
            # 步骤 4: 更新策略
            print("[步骤 4] 更新策略...")
            train_loss = self.update_policy(
                epochs=self.config.epochs_per_iteration,
                batch_size=self.config.batch_size
            )
            self.train_losses.append(train_loss)
            print(f"[训练] 当前轮损失: {train_loss:.6f}")
            
            # 定期评估
            if (iteration + 1) % self.config.eval_freq == 0:
                print("\n[评估] 运行评估...")
                evaluator = PolicyEvaluator(
                    self.config.env_name,
                    num_episodes=self.config.num_eval_episodes
                )
                runner = RobotPolicyRunner(
                    None, device=self.config.device
                )
                runner.policy = self.policy
                
                metrics = evaluator.evaluate(runner, verbose=False)
                self.eval_results.append(metrics)
                
                print(f"[评估] 成功率: {metrics['success_rate']:.2%}, "
                      f"平均奖励: {metrics['avg_reward']:.2f}")
                evaluator.close()
            
            # 定期保存
            if (iteration + 1) % self.config.save_freq == 0:
                self.save_checkpoint(iteration + 1)
        
        print("\n" + "=" * 60)
        print("DAgger 训练完成！")
        print("=" * 60)
        
        return self.policy
    
    def save_checkpoint(self, iteration):
        """保存模型检查点"""
        ckpt_path = os.path.join(
            self.config.output_dir,
            f"dagger_iter{iteration}.pt"
        )
        torch.save({
            'iteration': iteration,
            'policy_state_dict': self.policy.state_dict(),
            'optimizer_state_dict': self.optimizer.state_dict(),
            'train_losses': self.train_losses,
            'eval_results': self.eval_results,
            'dataset_size': len(self.dataset['observations'])
        }, ckpt_path)
        print(f"[Checkpoint] 已保存: {ckpt_path}")
    
    def save_final(self):
        """保存最终模型"""
        final_path = os.path.join(self.config.output_dir, "dagger_final.pt")
        torch.save(self.policy.state_dict(), final_path)
        
        # 保存训练记录
        record_path = os.path.join(self.config.output_dir, "training_record.json")
        with open(record_path, 'w') as f:
            json.dump({
                'train_losses': self.train_losses,
                'eval_results': self.eval_results
            }, f, indent=2)
        
        print(f"[Final] 模型已保存: {final_path}")


def main():
    parser = argparse.ArgumentParser(description='DAgger 训练')

```python
"""
deployment/deploy_policy.py
将训练好的策略部署到机器人
"""
import os
import time
import numpy as np
import gymnasium as gym
import argparse

from deployment.robot_inference import RobotPolicyRunner
from deployment.error_correction import ErrorCorrector


class RobotDeployer:
    """
    机器人策略部署器
    整合推理和误差校正
    """
    def __init__(self, model_path, env_name='FetchReach-v2',
                 use_correction=True, device='cpu'):
        self.runner = RobotPolicyRunner(
            model_path=model_path,
            device=device
        )
        self.env = gym.make(env_name)
        self.use_correction = use_correction
        
        if use_correction:
            state_dim = self.env.observation_space.shape[0]
            action_dim = self.env.action_space.shape[0]
            self.corrector = ErrorCorrector(state_dim, action_dim)
        
        self.episode_count = 0
        
    def run_episode(self, max_steps=500, render=True, verbose=True):
        """
        运行一个回合
        """
        obs, info = self.env.reset()
        total_reward = 0
        steps = 0
        done = False
        
        while not done and steps < max_steps:
            # 误差校正
            if self.use_correction:
                # 在线更新观测统计
                self.corrector.update_observation_stats(obs)
                
                # 校正观测
                obs_corrected = self.corrector.correct_observation(obs)
            else:
                obs_corrected = obs
            
            # 策略推理
            action = self.runner.predict(obs_corrected)
            
            # 综合校正
            if self.use_correction:
                action = self.corrector.correct_policy_output(action, obs)
            
            # 执行
            next_obs, reward, terminated, truncated, info = self.env.step(action)
            
            total_reward += reward
            obs = next_obs
            steps += 1
            done = terminated or truncated
            
            if render:

            # 实时显示
            if verbose and steps % 20 == 0:

        stats = self.runner.get_stats()
        print(f"[推理性能] 平均: {stats['mean_ms']:.2f}ms, "
              f"最大: {stats['max_ms']:.2f}ms")
        
        return {
            'success': success,
            'reward': total_reward,

### 9.3 简答题答案

**11. DAgger 完整训练流程及分布偏移解决机制**

DAgger 的完整流程：

```
1. 用专家策略采集初始示范数据集 D
2. 用 D 训练策略 π（行为克隆）
3. 用 π 收集轨迹，得到策略访问的状态分布 p_π
4. 在这些状态上查询专家，得到最优动作 a* = π*(s)
5. 将 (s, a*) 加入 D，回到步骤 2
```

**为什么能解决分布偏移**：行为克隆的问题在于只在专家见过的状态分布上训练，而策略执行时会访问自己的状态分布 $p_\pi$，两者不同。DAgger 通过步骤 3-4 显式覆盖了 $p_\pi$，使得策略在所有自己会遇到的状态上都有正确的专家标签，因此不会发生"策略偏离后遇到未知状态"的误差累积问题。

**12. 误差校正的必要性及主要方法**

**必要性**：仿真环境与真实环境之间存在物理参数、传感器特性、通信延迟等多方面差异，这些差异会导致"仿真-现实差距（Sim-to-Real Gap）"，即使在仿真中表现良好的策略，直接部署到真实机器人上也可能失效。

**主要方法**：

1. **观测偏差校正**：记录实际观测与期望观测的差异，在线估计并补偿传感器偏差
2. **延迟补偿**：考虑视觉处理和控制通信的延迟，使用动作缓冲区前馈补偿
3. **动作平滑**：对策略输出进行低通滤波，防止高频抖动损坏机器人
4. **自适应归一化**：在线更新观测的均值和标准差，适应不同环境
5. **Domain Randomization**：训练时随机化仿真参数，增强策略鲁棒性

**13. BC、DAgger、GAIL 适用场景对比**

| 方法 | 适用场景 | 不适用场景 |
|------|----------|------------|
| **BC** | 数据稀缺、需要快速部署、专家可离线获取 | 任务复杂、状态空间大、分布偏移敏感 |
| **DAgger** | 有在线专家可用、任务需要适应性、BC 效果不理想 | 无法访问专家、专家成本高（遥操作等） |
| **GAIL** | 无显式奖励函数、有大量在线交互数据、任务复杂 | 数据稀缺、训练资源有限、需要可解释奖励 |

**实战建议**：先用 BC 建立基线，如果分布偏移严重，再用 DAgger 迭代修正。GAIL 适合作为进阶方法。

**14. 仿真良好但真实失效的原因及解决方案**

1. **仿真-现实物理差异（摩擦力、惯性等）**
   - 解决：使用 Domain Randomization 训练；进行系统辨识校准物理参数

2. **传感器差异（相机噪声、光照、标定）**
   - 解决：在图像上添加人工噪声和光照变化；使用实际传感器数据微调

3. **动作执行延迟（控制周期、通信延迟）**
   - 解决：添加延迟补偿模块；使用动作缓冲和预测

4. **视觉领域偏移（背景、物体外观）**
   - 解决：使用领域适应技术；在真实数据上微调

5. **夹爪/末端执行器物理特性差异**
   - 解决：调整夹爪开合阈值；在真实机器人上做少量在线微调

def dagger_training_loop(env, expert_policy, policy_network,
                         num_iterations=50, rollouts_per_iter=5,
                         epochs_per_iter=10, batch_size=256):
    """
    简化 DAgger 训练循环
    Args:
        env: Gymnasium 环境
        expert_policy: 专家策略（可调用 get_action(state)）
        policy_network: 待训练的策略网络
        num_iterations: DAgger 迭代次数
        rollouts_per_iter: 每轮用策略收集的轨迹数
        epochs_per_iter: 每轮训练的 epoch 数
    """
    dataset_obs = []   # 状态数据集
    dataset_act = []   # 动作数据集（专家标签）
    
    optimizer = optim.Adam(policy_network.parameters(), lr=3e-4)
    
    for iteration in range(num_iterations):
        # ====== 1. 如果是初始迭代，先采集专家数据 ======
        if iteration == 0:
            print(f"[迭代 0] 采集初始专家数据...")
            for ep in range(5):  # 5 回合专家示教
                obs, _ = env.reset()
                done = False
                while not done:
                    # 专家动作
                    expert_a = expert_policy.get_action(obs)
                    dataset_obs.append(obs)
                    dataset_act.append(expert_a)
                    obs, r, terminated, truncated, _ = env.step(expert_a)
                    done = terminated or truncated
            print(f"初始数据集大小: {len(dataset_obs)}")
        
        # ====== 2. 用当前策略收集轨迹 ======
        print(f"[迭代 {iteration+1}] 用策略收集轨迹...")
        for ep in range(rollouts_per_iter):
            obs, _ = env.reset()
            done = False
            while not done:
                # 策略推理
                with torch.no_grad():
                    obs_t = torch.FloatTensor(obs).unsqueeze(0)
                    policy_a = policy_network(obs_t).numpy().squeeze()
                
                # 执行策略动作
                obs, r, terminated, truncated, _ = env.step(policy_a)
                done = terminated or truncated
        
        # ====== 3. 在策略访问的状态上查询专家 ======
        print(f"[迭代 {iteration+1}] 查询专家获取校正动作...")
        # 重新执行轨迹，收集状态并查询专家
        for ep in range(rollouts_per_iter):
            obs, _ = env.reset()
            done = False
            while not done:
                # 记录策略遇到的状态
                dataset_obs.append(obs.copy())
                # 查询专家动作（关键步骤）
                expert_a = expert_policy.get_action(obs)
                dataset_act.append(expert_a)
                # 执行专家动作推进环境
                obs, r, terminated, truncated, _ = env.step(expert_a)
                done = terminated or truncated
        
        print(f"数据集已扩展: {len(dataset_obs)} 样本")
        
        # ====== 4. 用扩展数据集更新策略 ======
        print(f"[迭代 {iteration+1}] 更新策略...")
        dataset_obs_arr = np.array(dataset_obs)
        dataset_act_arr = np.array(dataset_act)
        
        loader = DataLoader(
            TensorDataset(torch.FloatTensor(dataset_obs_arr),
                          torch.FloatTensor(dataset_act_arr)),
            batch_size=batch_size, shuffle=True
        )
        
        policy_network.train()
        for epoch in range(epochs_per_iter):
            epoch_loss = 0
            for batch_obs, batch_act in loader:
                pred_act = policy_network(batch_obs)
                loss = nn.functional.mse_loss(pred_act, batch_act)
                optimizer.zero_grad()
                loss.backward()
                optimizer.step()
                epoch_loss += loss.item()
            print(f"  Epoch {epoch+1}/{epochs_per_iter}, Loss: {epoch_loss/len(loader):.6f}")
        
        policy_network.eval()
    
    return policy_network
```

class ErrorCorrector:
    """
    误差校正器
    支持：观测偏差校正、动作延迟补偿、动作平滑
    """
    def __init__(self, state_dim, action_dim,
                 delay_steps=2, smooth_alpha=0.1,
                 bias_alpha=0.01):
        self.state_dim = state_dim
        self.action_dim = action_dim
        self.delay_steps = delay_steps
        self.smooth_alpha = smooth_alpha
        self.bias_alpha = bias_alpha
        
        # 动作延迟缓冲
        self.action_buffer = []
        
        # 观测统计（用于自适应归一化）
        self.obs_buffer = []
        self.obs_mean = None
        self.obs_std = None
        
        # 动作平滑
        self.last_action = None
        
        # 偏差估计
        self.bias = np.zeros(state_dim)
        
    def update_observation_stats(self, obs):
        """在线更新观测统计"""
        self.obs_buffer.append(obs)
        if len(self.obs_buffer) > 500:
            self.obs_buffer.pop(0)
        if len(self.obs_buffer) > 50:
            stacked = np.stack(self.obs_buffer)
            self.obs_mean = np.mean(stacked, axis=0)
            self.obs_std = np.std(stacked, axis=0) + 1e-8
    
    def correct_observation(self, obs):
        """校正观测"""
        obs_corr = obs - self.bias  # 偏差校正
        if self.obs_mean is not None:
            obs_corr = (obs_corr - self.obs_mean) / self.obs_std  # 自适应归一化
        return obs_corr
    
    def compensate_delay(self, action):
        """动作延迟补偿"""
        self.action_buffer.append(action.copy())
        if len(self.action_buffer) > self.delay_steps + 1:
            self.action_buffer.pop(0)
        if len(self.action_buffer) <= self.delay_steps:
            return action
        return self.action_buffer[-self.delay_steps - 1]
    
    def smooth_action(self, action):
        """动作平滑"""
        if self.last_action is None:
            self.last_action = action.copy()
            return action
        smoothed = self.smooth_alpha * action + (1 - self.smooth_alpha) * self.last_action
        self.last_action = action.copy()
        return smoothed
    
    def correct_action(self, raw_action, current_obs):
        """
        综合校正：延迟补偿 → 偏差更新 → 平滑
        """
        # 更新观测统计
        self.update_observation_stats(current_obs)
        
        # 延迟补偿
        action = self.compensate_delay(raw_action)
        
        # 限幅
        action = np.clip(action, -1, 1)
        
        # 平滑
        action = self.smooth_action(action)
        
        return action
